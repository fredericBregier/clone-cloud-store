17  package io.clonecloudstore.topology.client.api;
18  
19  import java.io.Closeable;
20  import java.util.Collection;
21  
22  import io.clonecloudstore.common.quarkus.client.ClientResponseExceptionMapper;
23  import io.clonecloudstore.common.quarkus.client.RequestHeaderFactory;
24  import io.clonecloudstore.topology.model.Topology;
25  import io.clonecloudstore.topology.model.TopologyStatus;
26  import io.smallrye.mutiny.Uni;
27  import jakarta.ws.rs.Consumes;
28  import jakarta.ws.rs.DELETE;
29  import jakarta.ws.rs.DefaultValue;
30  import jakarta.ws.rs.GET;
31  import jakarta.ws.rs.POST;
32  import jakarta.ws.rs.PUT;
33  import jakarta.ws.rs.Path;
34  import jakarta.ws.rs.Produces;
35  import jakarta.ws.rs.QueryParam;
36  import jakarta.ws.rs.core.Response;
37  import org.eclipse.microprofile.openapi.annotations.Operation;
38  import org.eclipse.microprofile.openapi.annotations.responses.APIResponse;
39  import org.eclipse.microprofile.openapi.annotations.tags.Tag;
40  import org.eclipse.microprofile.rest.client.annotation.RegisterClientHeaders;
41  import org.eclipse.microprofile.rest.client.annotation.RegisterProvider;
42  import org.eclipse.microprofile.rest.client.inject.RegisterRestClient;
43  import org.jboss.resteasy.reactive.RestPath;
44  
45  import static io.clonecloudstore.accessor.config.AccessorConstants.Api.TAG_ADMINISTRATION;
46  import static io.clonecloudstore.replicator.config.ReplicatorConstants.Api.BASE;
47  import static io.clonecloudstore.replicator.config.ReplicatorConstants.Api.TOPOLOGIES;
48  import static jakarta.ws.rs.core.MediaType.APPLICATION_JSON;
49  
50  @Path(BASE + TOPOLOGIES)
51  @RegisterRestClient
52  @RegisterProvider(ClientResponseExceptionMapper.class)
53  @RegisterClientHeaders(RequestHeaderFactory.class)
54  @RegisterProvider(ResponseClientFilter.class)
55  public interface TopologyApi extends Closeable {
56  
57    @GET
58    @Tag(name = TAG_ADMINISTRATION)
59    @Produces(APPLICATION_JSON)
60    @APIResponse(responseCode = "200", description = "Successfully retrieved list of topology", headers = {})
61    @APIResponse(responseCode = "500", description = "Internal server error", headers = {})
62    @Operation(summary = "Get list of replicators", description = "Get list of replicators in the  topology")
63    Uni<Collection<Topology>> listAll(@QueryParam("status") @DefaultValue("UNKNOWN") final TopologyStatus status);
64  
65    @POST
66    @Tag(name = TAG_ADMINISTRATION)
67    @APIResponse(responseCode = "201", description = "Successfully added topology", headers = {})
68    @APIResponse(responseCode = "400", description = "Topology not valid", headers = {})
69    @APIResponse(responseCode = "500", description = "Internal server error", headers = {})
70    @Produces(APPLICATION_JSON)
71    @Consumes(APPLICATION_JSON)
72    @Operation(summary = "Add a replicator to topology", description = "Add a replicator to topology")
73    Uni<Topology> add(Topology topology);
74  
75    @PUT
76    @Tag(name = TAG_ADMINISTRATION)
77    @APIResponse(responseCode = "202", description = "Successfully added topology", headers = {})
78    @APIResponse(responseCode = "400", description = "Topology not valid", headers = {})
79    @APIResponse(responseCode = "404", description = "Topology not found", headers = {})
80    @APIResponse(responseCode = "500", description = "Internal server error", headers = {})
81    @Produces(APPLICATION_JSON)
82    @Consumes(APPLICATION_JSON)
83    @Operation(summary = "Update a replicator in the topology", description = "Update a replicator in the topology")
84    Uni<Topology> update(Topology topology);
85  
86    @GET
87    @Tag(name = TAG_ADMINISTRATION)
88    @Path("/{site}")
89    @APIResponse(responseCode = "200", description = "Successfully retrieved topology", headers = {})
90    @APIResponse(responseCode = "400", description = "Topology id not valid", headers = {})
91    @APIResponse(responseCode = "404", description = "Topology not found", headers = {})
92    @APIResponse(responseCode = "500", description = "Internal server error", headers = {})
93    @Produces(APPLICATION_JSON)
94    @Operation(summary = "Get a replicator from topology", description = "Get a replicator full details from topology " +
95        "based on its site")
96    Uni<Topology> findBySite(@RestPath String site);
97  
98    @Path("/{site}")
99    @Tag(name = TAG_ADMINISTRATION)
100   @DELETE
101   @APIResponse(responseCode = "204", description = "Successfully deleted topology", headers = {})
102   @APIResponse(responseCode = "400", description = "Topology id not valid", headers = {})
103   @APIResponse(responseCode = "404", description = "Topology not found", headers = {})
104   @APIResponse(responseCode = "500", description = "Internal server error", headers = {})
105   @Produces(APPLICATION_JSON)
106   @Operation(summary = "Delete a replicator from topology", description = "Delete a replicator from topology")
107   Uni<Response> delete(@RestPath String site);
108 }
