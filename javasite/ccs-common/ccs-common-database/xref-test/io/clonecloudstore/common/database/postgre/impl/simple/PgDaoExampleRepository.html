17  package io.clonecloudstore.common.database.postgre.impl.simple;
18  
19  import io.clonecloudstore.common.database.model.dao.DaoExample;
20  import io.clonecloudstore.common.database.model.dao.DaoExampleRepository;
21  import io.clonecloudstore.common.database.postgre.ExtendedPanacheRepositoryBase;
22  import io.clonecloudstore.common.database.utils.DbQuery;
23  import io.clonecloudstore.common.database.utils.exception.CcsDbException;
24  import io.quarkus.arc.lookup.LookupIfProperty;
25  import jakarta.enterprise.context.ApplicationScoped;
26  import jakarta.transaction.Transactional;
27  
28  import static io.clonecloudstore.common.database.utils.DbType.CCS_DB_TYPE;
29  import static io.clonecloudstore.common.database.utils.DbType.POSTGRE;
30  
31  @LookupIfProperty(name = CCS_DB_TYPE, stringValue = POSTGRE)
32  @ApplicationScoped
33  @Transactional
34  public class PgDaoExampleRepository extends ExtendedPanacheRepositoryBase<DaoExample, PgDaoExample>
35      implements DaoExampleRepository {
36    public PgDaoExampleRepository() {
37      super(new PgDaoExample());
38    }
39  
40    @Override
41    public String getTable() {
42      return TABLE_NAME;
43    }
44  
45    @Override
46    public DaoExample createEmptyItem() {
47      return new PgDaoExample();
48    }
49  
50    /**
51     * Example of Stream usage: note that Stream cannot be used outside @Transactional
52     *
53     * @param dbQuery
54     * @return the first item
55     */
56    public DaoExample findUsingStream(final DbQuery dbQuery) throws CcsDbException {
57      // Stream cannot be used outside @Transactional
58      final var stream = findStream(dbQuery);
59      final var optionalPgDbDtoExample = stream.findFirst();
60      if (optionalPgDbDtoExample.isPresent()) {
61        return optionalPgDbDtoExample.get();
62      }
63      return null;
64    }
65  }
