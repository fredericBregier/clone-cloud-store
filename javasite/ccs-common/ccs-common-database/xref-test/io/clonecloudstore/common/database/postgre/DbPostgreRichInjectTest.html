49  @QuarkusTest
50  @TestProfile(PostgresProfile.class)
51  class DbPostgreRichInjectTest {
52    private static final Logger LOG = Logger.getLogger(DbPostgreRichInjectTest.class);
53    @Inject
54    Instance<DaoRichExampleRepository> repositoryInstance;
55    DaoRichExampleRepository repository;
56  
57    @BeforeEach
58    void setup() {
59      repository = repositoryInstance.get();
60    }
61  
62    @Test
63    void checkDb() throws JsonProcessingException, CcsDbException {
64      Assertions.assertEquals(DaoRichExampleRepository.TABLE_NAME, repository.getTable());
65      assertEquals(ID_PG, repository.getPkName());
66      assertTrue(repository.isSqlRepository());
67      assertEquals(0, repository.countAll());
68      final var opId = GuidLike.getGuid();
69      final Map<String, String> map = new HashMap<>();
70      map.put("key1", "val1");
71      map.put("key2", "val2");
72      final Set<String> set = new HashSet<>();
73      set.add("setval1");
74      set.add("setval2");
75      final var dtoExample = new DtoRichExample().setGuid(opId).setField1("field1").setMap1(map).setSet1(set);
76      var dao = repository.createEmptyItem();
77      dao.fromDto(dtoExample);
78      repository.insert(dao);
79      repository.flushAll();
80      assertEquals(1, repository.countAll());
81      var pgDbDtoExample = repository.findUsingStream(new DbQuery());
82      assertEquals(opId, pgDbDtoExample.getGuid());
83      assertEquals(map, pgDbDtoExample.getMap1());
84      assertEquals(set, pgDbDtoExample.getSet1());
85      streamCommand(new DbQuery());
86      pgDbDtoExample = repository.findWithPk(opId);
87      assertEquals(opId, pgDbDtoExample.getGuid());
88      final var dbQuery = new DbQuery(RestQuery.QUERY.EQ, ID_PG, opId);
89      assertEquals(1, repository.count(dbQuery));
90      final var dbUpdate = new DbUpdate().addToSet(DaoRichExampleRepository.ARRAY1, "setval3");
91      assertEquals(1, repository.update(dbQuery, dbUpdate));
92      pgDbDtoExample = repository.findOne(dbQuery);
93      assertEquals(opId, pgDbDtoExample.getGuid());
94      assertTrue(pgDbDtoExample.getSet1().contains("setval3"));
95      final var dbUpdate2 = new DbUpdate().setArray(DaoRichExampleRepository.ARRAY1, "setval4", set);
96      assertEquals(1, repository.update(dbQuery, dbUpdate2));
97      pgDbDtoExample = repository.findOne(dbQuery);
98      assertEquals(opId, pgDbDtoExample.getGuid());
99      assertTrue(pgDbDtoExample.getSet1().contains("setval4"));
100     pgDbDtoExample.addMapValue("key3", "val3");
101     assertTrue(repository.updateFull(pgDbDtoExample));
102     repository.deleteWithPk(opId);
103     assertEquals(0, repository.countAll());
104     final var example = repository.createEmptyItem();
105     example.fromDto(dtoExample);
106     repository.insert(example);
107     final var json = JsonUtil.getInstance().writeValueAsString(dtoExample);
108     final var dtoExample1 = JsonUtil.getInstance().readValue(json, DtoRichExample.class);
109     assertEquals(dtoExample, dtoExample1);
110     assertEquals(1, repository.countAll());
111     assertEquals(1, repository.delete(dbQuery));
112   }
113 
114   /**
115    * Example of usage of Stream within a Transactional method
116    *
117    * @param dbQuery the dbQuery
118    */
119   @Transactional
120   void streamCommand(final DbQuery dbQuery) throws CcsDbException {
121     final var stream = repository.findStream(dbQuery);
122     final var optionalPgDbDtoExample = stream.findFirst();
123     assertTrue(optionalPgDbDtoExample.isPresent());
124     final var pgDbDtoExample = optionalPgDbDtoExample.get();
125     assertNotNull(pgDbDtoExample);
126     assertEquals(1, repository.findStream(dbQuery).toList().size());
127   }
