34  import jakarta.persistence.Index;

41  import static io.clonecloudstore.common.database.postgre.PostgreSqlHelper.JSON_TYPE;
42  import static io.clonecloudstore.common.database.postgre.model.rich.DaoRichExampleRepository.ARRAY1;
43  import static io.clonecloudstore.common.database.postgre.model.rich.DaoRichExampleRepository.FIELD1;
44  import static io.clonecloudstore.common.database.postgre.model.rich.DaoRichExampleRepository.ID_PG;
45  import static io.clonecloudstore.common.database.postgre.model.rich.DaoRichExampleRepository.MAP1;
46  import static io.clonecloudstore.common.database.postgre.model.rich.DaoRichExampleRepository.TABLE_NAME;
47  import static io.clonecloudstore.common.standard.guid.GuidLike.UUID_B32_SIZE;
48  
49  @Entity
50  @Table(name = TABLE_NAME, indexes = {@Index(name = TABLE_NAME + "_filter_idx", columnList = FIELD1)})
51  public class PgDaoRichExample extends DaoRichExample {
52    private static final String[] EMPTY = new String[0];
53    /**
54     * To get a Set internally instead of an array
55     */
56    @IgnoreProperty
57    @Transient
58    @JsonIgnore
59    private final Set<String> set1 = new HashSet<>();
60    @Column(name = MAP1, columnDefinition = JSON_TYPE)
61    @JdbcTypeCode(SqlTypes.JSON)
62    private final Map<String, String> map1 = new HashMap<>();
63    @Id
64    @Column(name = ID_PG, nullable = false, length = UUID_B32_SIZE)
65    private String guid;
66    @Column(columnDefinition = "text[]", name = ARRAY1)
67    @Type(value = PostgreStringArrayType.class)
68    @JdbcTypeCode(SqlTypes.ARRAY)
69    private String[] array1;
70    /**
71     * To ensure array ans set are correctly initialized
72     */
73    @IgnoreProperty
74    @Transient
75    @JsonIgnore
76    private boolean checked;
77  
78  
79    public PgDaoRichExample() {
80      // Empty
81    }
82  
83    public PgDaoRichExample(final DtoRichExample dto) {
84      fromDto(dto);
85    }
86  
87    String[] getArray1() {
88      check();
89      return array1;
90    }
91  
92    PgDaoRichExample setArray1(final String[] array1) {
93      this.array1 = array1;
94      setStringToSet();
95      return this;
96    }
97  
98    /**
99     * To check if the array and set are correctly setup
100    *
101    * @return True if ok
102    */
103   @IgnoreProperty
104   @Transient
105   @JsonIgnore
106   public PgDaoRichExample check() {
107     if (checked) {
108       return this;
109     }
110     if (array1 != null) {
111       if (array1.length < set1.size()) {
112         setSetToString();
113       } else if (array1.length > set1.size()) {
114         setStringToSet();
115       }
116     } else {
117       setSetToString();
118     }
119     checked = true;
120     return this;
121   }
122 
123   private void setSetToString() {
124     array1 = set1.toArray(EMPTY);
125     checked = true;
126   }
127 
128   private void setStringToSet() {
129     set1.clear();
130     Collections.addAll(set1, array1);
131     checked = true;
132   }
133 
134   @IgnoreProperty
135   @Transient
136   @JsonIgnore
137   public PgDaoRichExample addItems(final Set<String> set1) {
138     check();
139     this.set1.addAll(set1);
140     setSetToString();
141     return this;
142   }
143 
144   @IgnoreProperty
145   @Transient
146   @JsonIgnore
147   @Override
148   public PgDaoRichExample addItem(final String item) {
149     if (this.set1 == null) {
150       return this;
151     }
152     check();
153     set1.add(item);
154     setSetToString();
155     return this;
156   }
157 
158   @IgnoreProperty
159   @Transient
160   @JsonIgnore
161   @Override
162   public PgDaoRichExample removeItem(final String item) {
163     check();
164     set1.remove(item);
165     setSetToString();
166     return this;
167   }
168 
169   @Override
170   @IgnoreProperty
171   @JsonIgnore
172   @Transient
173   public String getMapValue(final String key) {
174     return map1.get(key);
175   }
176 
177   @Override
178   @IgnoreProperty
179   @Transient
180   @JsonIgnore
181   public PgDaoRichExample addMapValue(final String key, final String value) {
182     map1.put(key, value);
183     return this;
184   }
185 
186   @Override
187   public String getGuid() {
188     return guid;
189   }
190 
191   @Override
192   public PgDaoRichExample setGuid(final String guid) {
193     this.guid = guid;
194     return this;
195   }
196 
197   @Override
198   @IgnoreProperty
199   @Transient
200   public Set<String> getSet1() {
201     check();
202     return set1;
203   }
204 
205   @Override
206   public PgDaoRichExample setSet1(final Collection<String> items) {
207     if (this.set1 == null) {
208       return this;
209     }
210     set1.clear();
211     set1.addAll(items);
212     setSetToString();
213     return this;
214   }
215 
216   @Override
217   public Map<String, String> getMap1() {
218     return map1;
219   }
220 
221   @Override
222   public PgDaoRichExample setMap1(final Map<String, String> map) {
223     if (this.map1 == null) {
224       return this;
225     }
226     this.map1.clear();
227     this.map1.putAll(map);
228     return this;
229   }
230 }
