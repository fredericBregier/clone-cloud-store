61    @Test
62    void checkDb() throws JsonProcessingException, CcsDbException {
63      assertEquals(TABLE_NAME, repository.getTable());
64      assertEquals(ID_PG, repository.getPkName());
65      assertTrue(repository.isSqlRepository());
66      assertEquals(0, repository.countAll());
67      final var instant = Instant.now();
68      final var opId = GuidLike.getGuid();
69      final var dtoExample = new DtoExample().setGuid(opId).setField1("field1").setField2("field2").setTimeField(instant);
70      repository.insert(new PgDaoExample(dtoExample));
71      repository.flushAll();
72      assertEquals(1, repository.countAll());
73      var pgDbDtoExample = repository.findUsingStream(new DbQuery());
74      assertEquals(opId, pgDbDtoExample.getGuid());
75      assertEquals(SystemTools.toMillis(instant), pgDbDtoExample.getTimeField());
76      streamCommand(new DbQuery());
77      pgDbDtoExample = repository.findWithPk(opId);
78      assertEquals(opId, pgDbDtoExample.getGuid());
79      final var dbQuery = DbQuery.idEquals(opId);
80      assertEquals(1, repository.count(dbQuery));
81      var dbUpdate = new DbUpdate().set(FIELD2, "value2");
82      assertEquals(1, repository.update(dbQuery, dbUpdate));
83      pgDbDtoExample = repository.findOne(dbQuery);
84      assertEquals(opId, pgDbDtoExample.getGuid());
85      assertEquals("value2", pgDbDtoExample.getField2());
86      dbUpdate = new DbUpdate().set(FIELD2, "value3");
87      Assertions.assertEquals(1, repository.update(PostgreSqlHelper.update(dbUpdate, dbQuery),
88          PostgreSqlHelper.getUpdateParamsAsArray(dbUpdate, dbQuery)));
89      pgDbDtoExample = repository.find(PostgreSqlHelper.query(dbQuery), dbQuery.getSqlParamsAsArray()).firstResult();
90      assertEquals(opId, pgDbDtoExample.getGuid());
91      assertEquals("value3", pgDbDtoExample.getField2());
92      assertEquals(1,
93          repository.getEntityManager().createNativeQuery(PostgreSqlHelper.select(repository.getTable(), dbQuery))
94              .setParameter(1, dbQuery.getSqlParams().get(0)).getResultList().size());
95      assertEquals(1,
96          repository.getEntityManager().createNativeQuery(PostgreSqlHelper.select(repository.getTable(), new DbQuery()))
97              .getResultList().size());
98      assertEquals(1,
99          repository.getEntityManager().createNativeQuery(PostgreSqlHelper.count(repository.getTable(), dbQuery))
100             .setParameter(1, dbQuery.getSqlParams().get(0)).getResultList().size());
101     assertEquals(1,
102         repository.getEntityManager().createNativeQuery(PostgreSqlHelper.count(repository.getTable(), new DbQuery()))
103             .getResultList().size());
104     repository.deleteWithPk(opId);
105     assertEquals(0, repository.countAll());
106     final var example = repository.createEmptyItem();
107     example.fromDto(dtoExample);
108     repository.insert(example);
109     final var json = JsonUtil.getInstance().writeValueAsString(dtoExample);
110     final var dtoExample1 = JsonUtil.getInstance().readValue(json, DtoExample.class);
111     assertEquals(dtoExample, dtoExample1);
112     example.setField1("newValue");
113     repository.updateFull(example);
114     pgDbDtoExample = repository.findOne(DbQuery.idEquals(example.getGuid()));
115     assertEquals(opId, pgDbDtoExample.getGuid());
116     assertEquals("newValue", pgDbDtoExample.getField1());
117     assertEquals(1, repository.countAll());
118     assertEquals(1, repository.delete(dbQuery));
119     repository.insert(example);
120     pgDbDtoExample = repository.findOne(DbQuery.idEquals(example.getGuid()));
121     var dbQueryEmpty = new DbQuery();
122     Assertions.assertEquals(1, repository.update(PostgreSqlHelper.update(dbUpdate, dbQueryEmpty),
123         PostgreSqlHelper.getUpdateParamsAsArray(dbUpdate, dbQueryEmpty)));
124     assertEquals(1, deleteFromNativeQuery(dbQueryEmpty));
125     assertEquals(0, repository.countAll());
126     repository.insert(example);
127     pgDbDtoExample = repository.findOne(DbQuery.idEquals(example.getGuid()));
128     var dbQueryId = DbQuery.idEquals(example.getGuid());
129     Assertions.assertEquals(1, repository.update(PostgreSqlHelper.update(dbUpdate, dbQueryId),
130         PostgreSqlHelper.getUpdateParamsAsArray(dbUpdate, dbQueryId)));
131     assertEquals(1, deleteFromNativeQuery(dbQueryId));
132     assertEquals(0, repository.countAll());
133   }

135   @Transactional
136   protected int deleteFromNativeQuery(final DbQuery dbQuery) {
137     if (dbQuery.isEmpty()) {
138       return repository.getEntityManager().createNativeQuery(PostgreSqlHelper.delete(repository.getTable(), dbQuery))
139           .executeUpdate();
140     }
141     return repository.getEntityManager().createNativeQuery(PostgreSqlHelper.delete(repository.getTable(), dbQuery))
142         .setParameter(1, dbQuery.getSqlParams().get(0)).executeUpdate();
143   }

145   /**
146    * Example of usage of Stream within a Transactional method
147    *
148    * @param dbQuery the dbQuery
149    */
150   @Transactional
151   void streamCommand(final DbQuery dbQuery) throws CcsDbException {
152     final var stream = repository.findStream(dbQuery);
153     final var optionalPgDbDtoExample = stream.findFirst();
154     assertTrue(optionalPgDbDtoExample.isPresent());
155     final var pgDbDtoExample = optionalPgDbDtoExample.get();
156     assertNotNull(pgDbDtoExample);
157     assertEquals(1, repository.findQuery(dbQuery).list().size());
158   }

168   void dbInsertBulk() throws CcsDbException {
169     assertEquals(TABLE_NAME, repository.getTable());
170     assertEquals(ID_PG, repository.getPkName());
171     assertFalse(DbType.getInstance().isMongoDbType());
172     assertEquals(0, repository.countAll());
173     final var dtoExample = new DtoExample();
174     dtoExample.setField1("field1").setField2("field2").setTimeField(Instant.now());
175     final var start = System.nanoTime();
176     for (var i = 0; i < MAX_BULK; i++) {
177       final var pgDbDtoExample = new PgDaoExample(dtoExample).setGuid(GuidLike.getGuid());
178       repository.insert(pgDbDtoExample);
179     }
180     repository.flushAll();
181     final var stop = System.nanoTime();
182     final var start2 = System.nanoTime();
183     insertBulk(dtoExample);
184     final var stop2 = System.nanoTime();
185     LOG.info("Standard Insert: " + (stop - start) / 1000000);
186     LOG.info("Bulk Insert: " + (stop2 - start2) / 1000000);
187     final var start3 = System.nanoTime();
188     streamAllCommand(new DbQuery());
189     final var stop3 = System.nanoTime();
190     final var start4 = System.nanoTime();
191     countAll();
192     final var stop4 = System.nanoTime();
193 
194     LOG.info("Stream all: " + (stop3 - start3) / 1000000 + " vs " + (stop4 - start4) / 1000000);
195     repository.deleteAllDb();
196   }

198   @Transactional
199   void insertBulk(final DtoExample dtoExample) throws CcsDbException {
200     for (var i = 0; i < MAX_BULK; i++) {
201       final var pgDbDtoExample = new PgDaoExample(dtoExample).setGuid(GuidLike.getGuid());
202       repository.addToInsertBulk(pgDbDtoExample);
203     }
204     repository.flushAll();
205   }

215   void dbUpsertBulk() throws CcsDbException {
216     assertEquals(TABLE_NAME, repository.getTable());
217     assertEquals(ID_PG, repository.getPkName());
218     assertEquals(0, repository.countAll());
219     final var dtoExample = new DtoExample();
220     dtoExample.setField1("field1").setField2("field2").setTimeField(Instant.now());
221     List<PgDaoExample> example2List = new ArrayList<>(MAX_BULK);
222     insertBulkForUpdate(dtoExample, example2List);
223     final var start = System.nanoTime();
224     for (final var pgDaoExample : example2List) {
225       pgDaoExample.setField1("value2");
226       repository.updateFull(pgDaoExample);
227     }
228     repository.flushAll();
229     final var stop = System.nanoTime();
230     final var start2 = System.nanoTime();
231     updateBulk(example2List);
232     final var stop2 = System.nanoTime();
233 
234     LOG.info("Standard Update: " + (stop - start) / 1000000);
235     LOG.info("Bulk Update: " + (stop2 - start2) / 1000000);
236     repository.deleteAllDb();
237     repository.flushAll();
238 
239     final var start3 = System.nanoTime();
240     upsertBulk(example2List);
241     final var stop3 = System.nanoTime();
242 
243     LOG.info("Bulk Update Insert: " + (stop3 - start3) / 1000000);
244     repository.deleteAllDb();
245 
246     assertTrue(stop - start > stop2 - start2);
247   }

249   @Transactional
250   void upsertBulk(final List<PgDaoExample> example2List) throws CcsDbException {
251     for (final var pgDaoExample : example2List) {
252       pgDaoExample.setField1("value3");
253       repository.addToUpdateBulk(pgDaoExample);
254     }
255     repository.flushAll();
256   }

258   @Transactional
259   void updateBulk(final List<PgDaoExample> example2List) throws CcsDbException {
260     for (final var pgDaoExample : example2List) {
261       pgDaoExample.setField1("value2");
262       repository.addToUpdateBulk(pgDaoExample);
263     }
264     repository.flushAll();
265   }
266 
267   @Transactional
268   void insertBulkForUpdate(final DtoExample dtoExample, final List<PgDaoExample> example2List) throws CcsDbException {
269     for (var i = 0; i < MAX_BULK; i++) {
270       final var pgDbDtoExample = new PgDaoExample(dtoExample).setGuid(GuidLike.getGuid());
271       repository.addToInsertBulk(pgDbDtoExample);
272       example2List.add(pgDbDtoExample);
273     }
274     repository.flushAll();
275   }

277   void countAll() {
278     final var count = repository.findAll().stream().count();

281   @Transactional
282   void streamAllCommand(final DbQuery dbQuery) throws CcsDbException {
283     final var countReal = repository.count(dbQuery);
284     final var stream = repository.findStream(dbQuery);
285     final var count = stream.count();
286     assertEquals(count, countReal);
