17  package io.clonecloudstore.common.standard.system;
18  
19  import java.io.OutputStream;
20  import java.io.PrintStream;
21  import java.nio.charset.StandardCharsets;
22  
23  import io.quarkus.test.junit.QuarkusTest;
24  import org.junit.jupiter.api.AfterAll;
25  import org.junit.jupiter.api.BeforeAll;
26  import org.junit.jupiter.api.Test;
27  
28  import static io.clonecloudstore.common.standard.system.SysErrLogger.ANSI_BLACK;
29  import static io.clonecloudstore.common.standard.system.SysErrLogger.ANSI_BLUE;
30  import static io.clonecloudstore.common.standard.system.SysErrLogger.ANSI_CYAN;
31  import static io.clonecloudstore.common.standard.system.SysErrLogger.ANSI_GREEN;
32  import static io.clonecloudstore.common.standard.system.SysErrLogger.ANSI_PURPLE;
33  import static io.clonecloudstore.common.standard.system.SysErrLogger.ANSI_RED;
34  import static io.clonecloudstore.common.standard.system.SysErrLogger.ANSI_RESET;
35  import static io.clonecloudstore.common.standard.system.SysErrLogger.ANSI_WHITE;
36  import static io.clonecloudstore.common.standard.system.SysErrLogger.ANSI_YELLOW;
37  import static org.junit.jupiter.api.Assertions.assertEquals;
38  import static org.junit.jupiter.api.Assertions.assertTrue;
39  
40  @QuarkusTest
41  class SysErrLoggerTest {
42    private static final String NOT_EMPTY = "Not empty";
43    private static final StringBuilder buf = new StringBuilder();
44    private static PrintStream err;
45    private static PrintStream out;
46  
47    @BeforeAll
48    static void setUpBeforeClass() {
49      err = System.err; // NOSONAR since Logger test
50      System.setErr(new PrintStream(new OutputStream() {
51        @Override
52        public void write(final int b) {
53          buf.append((char) b);
54        }
55      }, true, StandardCharsets.UTF_8));
56      out = System.out; // NOSONAR since Logger test
57      System.setOut(new PrintStream(new OutputStream() {
58        @Override
59        public void write(final int b) {
60          buf.append((char) b);
61        }
62      }, true, StandardCharsets.UTF_8));
63    }
64  
65    @AfterAll
66    static void tearDownAfterClass() {
67      System.setErr(err);
68      System.setOut(out);
69    }
70  
71    @Test
72    void testSyserr() {
73      buf.setLength(0);
74      SysErrLogger.FAKE_LOGGER.ignoreLog(new Exception("Fake exception"));
75      assertEquals(0, buf.length());
76      SysErrLogger.FAKE_LOGGER.syserr(NOT_EMPTY);
77      assertTrue(buf.length() > 0);
78      buf.setLength(0);
79      SysErrLogger.FAKE_LOGGER.syserr();
80      assertTrue(buf.length() > 0);
81      buf.setLength(0);
82      SysErrLogger.FAKE_LOGGER.syserr(NOT_EMPTY, new Exception("Fake exception"));
83      assertTrue(buf.length() > NOT_EMPTY.length() + 5);
84      buf.setLength(0);
85      SysErrLogger.FAKE_LOGGER.syserr(new Exception("Fake exception"));
86      assertTrue(buf.length() > 5);
87      buf.setLength(0);
88    }
89  
90    @Test
91    void testSysout() {
92      buf.setLength(0);
93      SysErrLogger.FAKE_LOGGER.ignoreLog(new Exception("Fake exception"));
94      assertEquals(0, buf.length());
95      SysErrLogger.FAKE_LOGGER.sysout(NOT_EMPTY);
96      assertTrue(buf.length() > 0);
97      buf.setLength(0);
98      SysErrLogger.FAKE_LOGGER.sysout();
99      assertTrue(buf.length() > 0);
100     buf.setLength(0);
101   }
102 
103   @Test
104   void testColor() {
105     buf.setLength(0);
106     SysErrLogger.FAKE_LOGGER.syserr(
107         SysErrLogger.cyan("cyan") + " " + SysErrLogger.black("black") + " " + SysErrLogger.blue("blue") + " " +
108             SysErrLogger.green("green") + " " + SysErrLogger.purple("purple") + " " + SysErrLogger.red("red") + " " +
109             SysErrLogger.white("white") + " " + SysErrLogger.yellow("yellow"));
110     assertTrue(buf.length() > 0);
111     assertTrue(buf.indexOf(ANSI_CYAN) >= 0);
112     assertTrue(buf.indexOf(ANSI_BLACK) >= 0);
113     assertTrue(buf.indexOf(ANSI_BLUE) >= 0);
114     assertTrue(buf.indexOf(ANSI_GREEN) >= 0);
115     assertTrue(buf.indexOf(ANSI_PURPLE) >= 0);
116     assertTrue(buf.indexOf(ANSI_RED) >= 0);
117     assertTrue(buf.indexOf(ANSI_WHITE) >= 0);
118     assertTrue(buf.indexOf(ANSI_YELLOW) >= 0);
119     assertTrue(buf.indexOf(ANSI_RED) >= 0);
120     assertTrue(buf.indexOf(ANSI_RED) >= 0);
121     assertTrue(buf.indexOf(ANSI_RESET) >= 0);
122     buf.setLength(0);
123   }
124 }
