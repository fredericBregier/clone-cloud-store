17  package io.clonecloudstore.common.standard.guid;
18  
19  import java.util.Arrays;
20  import java.util.concurrent.atomic.AtomicLong;
21  
22  import io.clonecloudstore.common.standard.exception.CcsInvalidArgumentRuntimeException;
23  import io.clonecloudstore.common.standard.system.BaseXx;
24  
25  /**
26   * UUID Generator with 2 longs (128 bits), compatible multi instances (through the PID,
27   * but not multi servers, neither multi-sites due to lack of Mac address/Platform Id.
28   * <br>
29   * <br>
30   * Inspired from com.groupon locality-uuid which used combination of internal
31   * counter value, process id and
32   * Timestamp. see https://github.com/groupon/locality-uuid.java <br>
33   * <br>
34   * Benchmark shows about 20 millions/s generated GuidLike, 8 M/s getId().
35   */
36  public final class GuidLike {
37    /**
38     * Bits size of Counter
39     */
40    private static final int SIZE_COUNTER = 20;
41    /**
42     * Min Counter value
43     */
44    private static final int MIN_COUNTER = 0;
45    /**
46     * Max Counter value
47     */
48    private static final int MAX_COUNTER = (1 << SIZE_COUNTER);
49    /**
50     * Counter part
51     */
52    private static final AtomicLong COUNTER = new AtomicLong(MIN_COUNTER);
53    /**
54     * Byte size of UUID
55     */
56    private static final int UUID_SIZE = 16;
57    /**
58     * GUID Like Base 32 size
59     */
60    public static final int UUID_B32_SIZE = 26;
61    /**
62     * GUID Like Base 32 size
63     */
64    private static final int UUID_B64_SIZE = 22;
65    private static final short BYTE_MASK = 0xFF;
66    private static final short BYTE_SIZE = 8;
67    private static final short VERSION = 1;
68    /**
69     * real UUID
70     */
71    private final byte[] uuid;
72  
73    /**
74     * @return Convenient method to get String GUID (similar to new GuidLike().getId())
75     */
76    public static String getGuid() {
77      return BaseXx.getBase32(new GuidLike().uuid);
78    }
79  
80    /**
81     * Constructor that generates a new UUID using the current process id and
82     * MAC address, timestamp and a counter
83     */
84    public GuidLike() {
85      uuid = new byte[UUID_SIZE];
86      // Platform and PID in 6 bytes
87      // Header 1 byte
88      System.arraycopy(JvmProcessMacIds.getMacPid(), 0, uuid, 7, 6);
89      uuid[6] = VERSION & BYTE_MASK;
90      // Counter with 3 bytes
91      for (int pos = 13, valuei = (int) (COUNTER.getAndIncrement() % MAX_COUNTER); pos < 16;
92           pos++, valuei >>>= BYTE_SIZE) {
93        uuid[pos] = (byte) (valuei & BYTE_MASK);
94      }
95      // Timestamp with 6 bytes
96      var value = System.currentTimeMillis();
97      for (int pos = 0; pos < 6; pos++, value >>>= BYTE_SIZE) {
98        uuid[pos] = (byte) (value & BYTE_MASK);
99      }
100   }
101 
102   public GuidLike(final long valueLow, final long valueHigh) {
103     uuid = new byte[UUID_SIZE];
104     for (int pos = 0, move = 56; pos < 7; pos++, move -= 8) {
105       uuid[pos] = (byte) ((valueLow >> move) & BYTE_MASK);
106     }
107     uuid[7] = (byte) (valueLow & BYTE_MASK);
108     for (int pos = 8, move = 56; pos < 15; pos++, move -= 8) {
109       uuid[pos] = (byte) ((valueHigh >> move) & BYTE_MASK);
110     }
111     uuid[15] = (byte) (valueHigh & BYTE_MASK);
112   }
113 
114   /**
115    * Constructor that takes a byte array as UUID's content
116    *
117    * @param bytes UUID content
118    */
119   public GuidLike(final byte[] bytes) {
120     if (bytes.length != UUID_SIZE) {
121       throw new CcsInvalidArgumentRuntimeException("Attempted to parse malformed UUID: " + Arrays.toString(bytes));
122     }
123     uuid = new byte[UUID_SIZE];
124     System.arraycopy(bytes, 0, uuid, 0, UUID_SIZE);
125   }
126 
127   /**
128    * Constructor that takes a String representation of a UUID in Base 16
129    *
130    * @param idsource UUID content as String
131    */
132   public GuidLike(final String idsource) {
133     final var id = idsource.trim();
134     uuid = new byte[UUID_SIZE];
135     if (id.length() == UUID_SIZE * 2) {
136       System.arraycopy(BaseXx.getFromBase16(id), 0, uuid, 0, UUID_SIZE);
137     } else if (id.length() == UUID_B32_SIZE) {
138       System.arraycopy(BaseXx.getFromBase32(id), 0, uuid, 0, UUID_SIZE);
139     } else if (id.length() == UUID_B64_SIZE) {
140       System.arraycopy(BaseXx.getFromBase64(id), 0, uuid, 0, UUID_SIZE);
141     } else {
142       throw new CcsInvalidArgumentRuntimeException("Attempted to parse malformed UUID: " + id);
143     }
144   }
145 
146   /**
147    * @return (pseudo) time
148    */
149   public long getTime() {
150     long value = uuid[0];
151     for (var pos = 1; pos < 6; pos++) {
152       value <<= BYTE_SIZE;
153       value += uuid[pos];
154     }
155     return value;
156   }
157 
158   /**
159    * @return the LongUuid size in bytes
160    */
161   public static short getKeySize() {
162     return UUID_SIZE;
163   }
164 
165   /**
166    * copy the uuid of this UUID, so that it can't be changed, and return it
167    *
168    * @return raw byte array of UUID
169    */
170   public byte[] getBytes() {
171     return Arrays.copyOf(uuid, UUID_SIZE);
172   }
173 
174   @Override
175   public int hashCode() {
176     return Arrays.hashCode(uuid);
177   }
178 
179   @Override
180   public boolean equals(final Object o) {
181     if (!(o instanceof GuidLike)) {
182       return false;
183     }
184     return this == o || Arrays.equals(uuid, ((GuidLike) o).uuid);
185   }
186 
187   /**
188    * @return the Base 32 representation
189    */
190   @Override
191   public String toString() {
192     return getId();
193   }
194 
195   /**
196    * @return the Base 32 representation
197    */
198   public String getId() {
199     return toBase32();
200   }
201 
202   public String toHex() {
203     return BaseXx.getBase16(uuid);
204   }
205 
206   public String toBase64() {
207     return BaseXx.getBase64(uuid);
208   }
209 
210   public String toBase32() {
211     return BaseXx.getBase32(uuid);
212   }
213 
214   /**
215    * @return the equivalent Low UUID as long
216    */
217   public long getLongLow() {
218     var value = ((long) uuid[0] & BYTE_MASK) << 56;
219     for (int pos = 1, move = 48; pos < 7; pos++, move -= 8) {
220       value |= ((long) uuid[pos] & BYTE_MASK) << move;
221     }
222     value |= (long) uuid[7] & BYTE_MASK;
223     return value;
224   }
225 
226   /**
227    * @return the equivalent High UUID as long
228    */
229   public long getLongHigh() {
230     var value = ((long) uuid[8] & BYTE_MASK) << 56;
231     for (int pos = 9, move = 48; pos < 15; pos++, move -= 8) {
232       value |= ((long) uuid[pos] & BYTE_MASK) << move;
233     }
234     value |= (long) uuid[15] & BYTE_MASK;
235     return value;
236   }
