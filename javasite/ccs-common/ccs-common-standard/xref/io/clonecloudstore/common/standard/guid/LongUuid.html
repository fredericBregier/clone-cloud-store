17  package io.clonecloudstore.common.standard.guid;
18  
19  import java.util.Arrays;
20  import java.util.concurrent.atomic.AtomicInteger;
21  
22  import io.clonecloudstore.common.standard.exception.CcsInvalidArgumentRuntimeException;
23  import io.clonecloudstore.common.standard.system.BaseXx;
24  
25  /**
26   * UUID Generator but limited to 1 Long (64 bits), compatible multi instances (through the PID,
27   * but not multi servers, neither multi-sites due to lack of Mac address/Platform Id.
28   * <br>
29   * <br>
30   * Inspired from com.groupon locality-uuid which used combination of internal
31   * counter value, process id and
32   * Timestamp. see https://github.com/groupon/locality-uuid.java <br>
33   * <br>
34   * Benchmark shows about 20 millions/s generated LongUuid, 8 M/s getId().
35   */
36  public final class LongUuid {
37    /**
38     * Bits size of Counter
39     */
40    private static final int SIZE_COUNTER = 20;
41    /**
42     * Min Counter value
43     */
44    private static final int MIN_COUNTER = 0;
45    /**
46     * Max Counter value
47     */
48    private static final int MAX_COUNTER = (1 << SIZE_COUNTER);
49    /**
50     * Counter part
51     */
52    private static final AtomicInteger COUNTER = new AtomicInteger(MIN_COUNTER);
53    /**
54     * Byte size of UUID
55     */
56    private static final int UUID_SIZE = 8;
57  
58  
59    /**
60     * real UUID
61     */
62    private final byte[] uuid = {0, 0, 0, 0, 0, 0, 0, 0};
63  
64    /**
65     * Constructor that generates a new UUID using the current process id and
66     * MAC address, timestamp and a counter
67     */
68    public LongUuid() {
69      this(getLongUuid());
70    }
71  
72    public LongUuid(final long value) {
73      uuid[0] = (byte) (value >> 56);
74      uuid[1] = (byte) (value >> 48);
75      uuid[2] = (byte) (value >> 40);
76      uuid[3] = (byte) (value >> 32);
77      uuid[4] = (byte) (value >> 24);
78      uuid[5] = (byte) (value >> 16);
79      uuid[6] = (byte) (value >> 8);
80      uuid[7] = (byte) value;
81    }
82  
83    /**
84     * @return a new Uuid as long
85     */
86    public static long getLongUuid() {
87      final var time = System.currentTimeMillis();
88      // atomically
89      final var count = COUNTER.getAndIncrement() % MAX_COUNTER;
90      // Jvmd Id on 4 first bits
91      // Timestamp on 40 bits (2^40 ms = 35 years rolling)
92      // Count on 20 bits => 2^20 (1M / ms)
93      var uuidAsLong = (JvmProcessMacIds.getJvmByteId() & 0xF0L) << 56;
94      uuidAsLong |= (time & 0xFFFFFFFFFFL) << 20;
95      uuidAsLong |= count & 0xFFFFFL;
96      return uuidAsLong;
97    }
98  
99    static int getCounter() {
100     return COUNTER.getAndIncrement() % MAX_COUNTER;
101   }
102 
103   /**
104    * Constructor that takes a byte array as UUID's content
105    *
106    * @param bytes UUID content
107    */
108   public LongUuid(final byte[] bytes) {
109     if (bytes.length != UUID_SIZE) {
110       throw new CcsInvalidArgumentRuntimeException("Attempted to parse malformed UUID: " + Arrays.toString(bytes));
111     }
112     System.arraycopy(bytes, 0, uuid, 0, UUID_SIZE);
113   }
114 
115   /**
116    * Constructor that takes a String representation of a UUID in Base 16
117    *
118    * @param idsource UUID content as String
119    */
120   public LongUuid(final String idsource) {
121     final var id = idsource.trim();
122 
123     if (id.length() != UUID_SIZE * 2) {
124       throw new CcsInvalidArgumentRuntimeException("Attempted to parse malformed UUID: " + id);
125     }
126     System.arraycopy(BaseXx.getFromBase16(id), 0, uuid, 0, UUID_SIZE);
127   }
128 
129   /**
130    * @return the LongUuid size in bytes
131    */
132   public static short getKeySize() {
133     return UUID_SIZE;
134   }
135 
136   /**
137    * copy the uuid of this UUID, so that it can't be changed, and return it
138    *
139    * @return raw byte array of UUID
140    */
141   public byte[] getBytes() {
142     return Arrays.copyOf(uuid, UUID_SIZE);
143   }
144 
145   /**
146    * Extract process id from raw UUID bytes and return as int
147    *
148    * @return id of process that generated the UUID
149    */
150   public int getProcessId() {
151     return uuid[0] >> 4 & 0x0F;
152   }
153 
154   /**
155    * Extract timestamp from raw UUID bytes and return as int
156    *
157    * @return millisecond UTC timestamp from generation of the UUID
158    */
159   public long getTimestamp() {
160     long time;
161     time = ((long) uuid[0] & 0x0F) << 36;
162     time |= ((long) uuid[1] & 0xFF) << 28;
163     time |= ((long) uuid[2] & 0xFF) << 20;
164     time |= ((long) uuid[3] & 0xFF) << 12;
165     time |= ((long) uuid[4] & 0xFF) << 4;
166     time |= ((long) uuid[5] & 0xF0) >> 4;
167     return time;
168   }
169 
170   @Override
171   public int hashCode() {
172     return Arrays.hashCode(uuid);
173   }
174 
175   @Override
176   public boolean equals(final Object o) {
177     if (!(o instanceof LongUuid)) {
178       return false;
179     }
180     return this == o || Arrays.equals(uuid, ((LongUuid) o).uuid);
181   }
182 
183   /**
184    * @return the Base 16 representation
185    */
186   @Override
187   public String toString() {
188     return getId();
189   }
190 
191   /**
192    * @return the Base 16 representation
193    */
194   public String getId() {
195     return BaseXx.getBase16(uuid);
196   }
197 
198   /**
199    * @return the equivalent UUID as long
200    */
201   public long getLong() {
202     var value = ((long) uuid[0] & 0xFF) << 56;
203     value |= ((long) uuid[1] & 0xFF) << 48;
204     value |= ((long) uuid[2] & 0xFF) << 40;
205     value |= ((long) uuid[3] & 0xFF) << 32;
206     value |= ((long) uuid[4] & 0xFF) << 24;
207     value |= ((long) uuid[5] & 0xFF) << 16;
208     value |= ((long) uuid[6] & 0xFF) << 8;
209     value |= (long) uuid[7] & 0xFF;
210     return value;
211   }
212 }
