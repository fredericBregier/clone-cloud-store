29  import static org.junit.jupiter.api.Assertions.assertEquals;
30  
31  @TestMethodOrder(MethodOrderer.MethodName.class)
32  @QuarkusTest
33  @Disabled("Only for checking")
34  class ApiFullDoubleLongRunIT extends ApiFullServerDoubleAbstract {
35    private static final Logger LOG = Logger.getLogger(ApiFullDoubleLongRunIT.class);
36    private static final int NB_ITER_SEQUENTIAL = 5;
37    private static final int NB_TEST = 8;
38    private static final int NB_STREAM_PER_TEST = 3;
39  
40    @Test
41    void checkStabilityManualTest() throws InterruptedException {
42      final var start = System.nanoTime();
43      final var res1 = checkForStabilityManual();
44      final var stop = System.nanoTime();
45      double value1 = ((float) ApiQuarkusService.LEN) * (res1 * NB_STREAM_PER_TEST) /
46          (1024 * 1024.0 * ((stop - start) / 1000000000.0));
47      LOG.infof("Global Bandwidth on %d transfers: %f", res1 * NB_STREAM_PER_TEST, value1);
48  
49      assertEquals(NB_ITER_SEQUENTIAL * NB_TEST, res1);
50    }
51  
52    long checkForStabilityManual() {
53      LOG.infof("START UNITARY");
54      AtomicLong cpt = new AtomicLong();
55      for (int i = 0; i < NB_ITER_SEQUENTIAL; i++) {
56        executeTests(cpt);
57      }
58      LOG.infof("END UNITARY");
59      return cpt.get();
60    }
61  
62    private void executeTests(final AtomicLong cpt) {
63      check30PostInputStreamQuarkusDoubleTest();
64      cpt.incrementAndGet();
65      check31PostInputStreamQuarkusDoubleNoSizeTest();
66      cpt.incrementAndGet();
67      check33PostInputStreamQuarkusShaDoubleNoSizeTest();
68      cpt.incrementAndGet();
69      check34GetInputStreamQuarkusDoubleTest();
70      cpt.incrementAndGet();
71      check35GetInputStreamQuarkusDoubleNoSizeTest();
72      cpt.incrementAndGet();
73      check41PostInputStreamQuarkusDoubleCompressedIntraTest();
74      cpt.incrementAndGet();
75      check43PostInputStreamQuarkusShaDoubleCompressedIntraTest();
76      cpt.incrementAndGet();
77      check51GetInputStreamQuarkusDoubleNoSizeCompressedIntraTest();
78      cpt.incrementAndGet();
79    }
80  }
