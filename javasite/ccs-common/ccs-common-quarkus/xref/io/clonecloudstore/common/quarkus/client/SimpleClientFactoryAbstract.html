17  package io.clonecloudstore.common.quarkus.client;
18  
19  import java.io.Closeable;
20  import java.net.URI;
21  
22  import io.clonecloudstore.common.quarkus.properties.QuarkusSystemPropertyUtil;
23  import io.clonecloudstore.common.standard.properties.StandardProperties;
24  import io.clonecloudstore.common.standard.system.ParametersChecker;
25  import io.quarkus.rest.client.reactive.QuarkusRestClientBuilder;
26  import io.vertx.mutiny.core.Vertx;
27  import jakarta.annotation.PreDestroy;
28  import jakarta.enterprise.inject.spi.CDI;
29  import jakarta.inject.Inject;
30  import org.jboss.logging.Logger;
31  
32  /**
33   * The Abstract implementation for the Client Factory.
34   * Note that the implementation can use ApplicationScoped annotation.
35   * Use property quarkus.rest-client."org.acme.rest.client.ExtensionsService".url to setup the right URL for Quarkus
36   * service.
37   *
38   * @param <S> the type for the Rest Service as Quarkus definition
39   */
40  public abstract class SimpleClientFactoryAbstract<S extends Closeable> implements Closeable {
41    private static final Logger LOGGER = Logger.getLogger(SimpleClientFactoryAbstract.class);
42    private static final String PREFIX_URL = "quarkus.rest-client.\"";
43    private static final String POSTFIX_URL = "\".url";
44    public static final String DEFAULT_VALUE = "http://127.0.0.1:8081";
45    private static final String QUARKUS_REST_CLIENT_INTERFACE = "$$QuarkusRestClientInterface";
46    Vertx vertx = null;
47    private boolean tls;
48    private URI uri;
49  
50    /**
51     * Empty constructor
52     */
53    protected SimpleClientFactoryAbstract() {
54      // Empty
55    }
56  
57    @Inject
58    void setup() {
59      vertx = CDI.current().select(Vertx.class).get();
60      // Load quarkus.rest-client."org.acme.rest.client.ExtensionsService".url
61      try {
62        final String className;
63        final var serviceClass = getServiceClass();
64        if (serviceClass != null) {
65          className = serviceClass.getName();
66        } else {
67          try (final var service = getService(URI.create(DEFAULT_VALUE))) {
68            // Remove $$QuarkusRestClientInterface from class name to get the real Interface name
69            className = service.getClass().getName().replace(QUARKUS_REST_CLIENT_INTERFACE, "");
70          }
71        }
72        final var uriString =
73            QuarkusSystemPropertyUtil.getStringConfig(PREFIX_URL + className + POSTFIX_URL, DEFAULT_VALUE);
74        final var uriSimple = URI.create(uriString);
75        prepare(uriSimple);
76      } catch (final Exception ignore) {
77        // Ignore
78      }
79    }
80  
81    /**
82     * Shall be overridden
83     *
84     * @return the class of API Interface
85     */
86    protected abstract Class<?> getServiceClass();
87  
88    /**
89     * Allows to change the target after initialization.
90     * This shall be called from Properties at runtime for each Factory (1 per destination).
91     * Not necessary if the configuration contains and if the factory is for only one target:<br/>
92     * quarkus.rest-client."org.acme.rest.client.ExtensionsService".url=https://hostname:port/api
93     */
94    public SimpleClientFactoryAbstract<S> prepare(final URI uri) {
95      final var scheme = uri.getScheme() == null ? "http" : uri.getScheme();
96      final var isTls = "https".equals(scheme);
97      final var host = uri.getHost() == null ? "127.0.0.1" : uri.getHost();
98      final var port = uri.getPort() <= 0 ? 8081 : uri.getPort();
99      final var path = uri.getPath();
100     return prepare(isTls, host, port, path);
101   }
102 
103   /**
104    * Allows to change the target after initialization.
105    * This shall be called from Properties at runtime for each Factory (1 per destination).
106    * Not necessary if the configuration contains and if the factory is for only one target:<br/>
107    * quarkus.rest-client."org.acme.rest.client.ExtensionsService".url=https://hostname:port/api
108    */
109   public SimpleClientFactoryAbstract<S> prepare(final boolean tls, final String hostname, final int quarkusPort,
110                                                 final String path) {
111     if (vertx == null) {
112       setup();
113     }
114     this.tls = tls;
115     final var finalPath = '/' + (ParametersChecker.isNotEmpty(path) ? path : "");
116     final var base = finalPath.replaceAll("\\/+", "/");
117     uri = URI.create((tls ? "https://" : "http://") + hostname + ':' + quarkusPort + base);
118     return this;
119   }
120 
121   /**
122    * @return the Vertx CDI instance
123    */
124   public Vertx getVertx() {
125     return vertx;
126   }
127 
128   /**
129    * @return the URI as specified by tls, hostname and quarkusPort
130    */
131   public URI getUri() {
132     return uri;
133   }
134 
135   @Override
136   @PreDestroy
137   public void close() {
138     // Empty
139   }
140 
141   /**
142    * @return True if TLS is required
143    */
144   public boolean isTls() {
145     return tls;
146   }
147 
148   /**
149    * In general, implementation is: QuarkusRestClientBuilder.newBuilder().baseUri(uri).build(ServiceRest.class)
150    * where ServiceRest is your Java Interface of the Rest Service.
151    *
152    * @return the Rest Service associated
153    */
154   protected S getService(final URI uri) {
155     final var apiClass = getServiceClass();
156     LOGGER.debugf("Uri %s Class %s", uri, apiClass);
157     return (S) QuarkusRestClientBuilder.newBuilder().baseUri(uri)
158         .property("io.quarkus.rest.client.max-chunk-size", StandardProperties.getBufSize()).build(apiClass);
159   }
160 
161   /**
162    * In general, implementation is simply: return new ClientImpl(this);
163    *
164    * @return the new Client with this associated factory
165    */
166   public abstract SimpleClientAbstract<S> newClient();
167 
168   /**
169    * For Factory with multiple targets, to ensure correctness of URI
170    *
171    * @return the new Client with this associated factory for this URI
172    */
173   public synchronized SimpleClientAbstract<S> newClient(final URI uri) {
174     return prepare(uri).newClient();
175   }
176 }
