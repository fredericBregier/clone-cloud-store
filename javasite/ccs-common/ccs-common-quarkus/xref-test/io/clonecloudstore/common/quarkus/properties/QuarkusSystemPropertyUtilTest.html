48    @Test
49    final void testSystemPropertyDefault() {
50      QuarkusSystemPropertyUtil.refresh();
51      assertNotNull(QuarkusSystemPropertyUtil.get(QuarkusProperties.CCS_MACHINE_ID));
52      QuarkusSystemPropertyUtil.set(QuarkusProperties.CCS_MACHINE_ID, "AE1632");
53      assertEquals("AE1632", QuarkusProperties.getCcsMachineId());
54      QuarkusSystemPropertyUtil.clear(QuarkusProperties.CCS_MACHINE_ID);
55      QuarkusSystemPropertyUtil.refresh();
56      assertNull(QuarkusProperties.getCcsMachineId());
57    }

59    @Test
60    final void testSystemPropertyString() {
61      QuarkusSystemPropertyUtil.refresh();
62      QuarkusSystemPropertyUtil.set(KEY_TEST, KEY_VALUE);
63      assertTrue(QuarkusSystemPropertyUtil.contains(KEY_TEST));
64      assertEquals(KEY_VALUE, QuarkusSystemPropertyUtil.get(KEY_TEST));
65      assertEquals(KEY_VALUE, QuarkusSystemPropertyUtil.get(KEY_TEST, OTHER));
66      assertEquals(KEY_VALUE, QuarkusSystemPropertyUtil.getAndSet(KEY_TEST, OTHER));
67      assertEquals(OTHER, QuarkusSystemPropertyUtil.getAndSet(KEY_TEST + '2', OTHER));
68      assertEquals(OTHER, QuarkusSystemPropertyUtil.set(KEY_TEST + '2', OTHER));
69    }

71    @Test
72    final void testSystemPropertyBoolean() {
73      QuarkusSystemPropertyUtil.set(KEY_BTEST, KEY_BVALUE);
74      assertTrue(QuarkusSystemPropertyUtil.contains(KEY_BTEST));
75      assertEquals(Boolean.toString(KEY_BVALUE), QuarkusSystemPropertyUtil.get(KEY_BTEST));
76      assertEquals(KEY_BVALUE, QuarkusSystemPropertyUtil.get(KEY_BTEST, false));
77      assertEquals(KEY_BVALUE, QuarkusSystemPropertyUtil.getAndSet(KEY_BTEST, false));
78      assertFalse(QuarkusSystemPropertyUtil.getAndSet(KEY_BTEST + '2', false));
79      assertFalse(QuarkusSystemPropertyUtil.set(KEY_BTEST + '2', false));
80      assertFalse(QuarkusSystemPropertyUtil.get(KEY_BTEST + '3', false));
81      assertNull(QuarkusSystemPropertyUtil.set(KEY_BTEST + '3', "true"));
82      assertTrue(QuarkusSystemPropertyUtil.get(KEY_BTEST + '3', false));
83      assertEquals("true", QuarkusSystemPropertyUtil.set(KEY_BTEST + '3', "yes"));
84      assertTrue(QuarkusSystemPropertyUtil.get(KEY_BTEST + '3', false));
85      assertEquals("yes", QuarkusSystemPropertyUtil.set(KEY_BTEST + '3', "1"));
86      assertTrue(QuarkusSystemPropertyUtil.get(KEY_BTEST + '3', false));
87      assertEquals("1", QuarkusSystemPropertyUtil.set(KEY_BTEST + '3', "yes2"));
88      assertFalse(QuarkusSystemPropertyUtil.get(KEY_BTEST + '3', false));
89      assertEquals("yes2", QuarkusSystemPropertyUtil.set(KEY_BTEST + '3', ""));
90      assertTrue(QuarkusSystemPropertyUtil.get(KEY_BTEST + '3', false));
91    }

161     final var bool = new AtomicBoolean(false);
162     final var outputStream = new OutputStream() {
163       @Override
164       public void write(final int i) throws IOException {
165         bool.set(true);
166       }
167 
168       @Override
169       public void write(final byte[] b) throws IOException {
170         bool.set(true);
171       }
172 
173       @Override
174       public void write(final byte[] b, final int off, final int len) throws IOException {
175         bool.set(true);
176       }
177     };
178     final var out = new PrintStream(outputStream);
179     QuarkusSystemPropertyUtil.debug(out);
180     assertTrue(bool.get());

183   @Test
184   final void testSystemPropertyOs() {
185     QuarkusSystemPropertyUtil.get(OS_NAME);
186     final var platform = QuarkusSystemPropertyUtil.getOS();
187     switch (platform) {
188       case MAC: {
189         assertTrue(QuarkusSystemPropertyUtil.isMac());
190         assertFalse(QuarkusSystemPropertyUtil.isWindows());
191         assertFalse(QuarkusSystemPropertyUtil.isUnix());
192         assertFalse(QuarkusSystemPropertyUtil.isSolaris());
193         break;
194       }
195       case SOLARIS: {
196         assertFalse(QuarkusSystemPropertyUtil.isMac());
197         assertFalse(QuarkusSystemPropertyUtil.isWindows());
198         assertFalse(QuarkusSystemPropertyUtil.isUnix());
199         assertTrue(QuarkusSystemPropertyUtil.isSolaris());
200         break;
201       }
202       case UNIX: {
203         assertFalse(QuarkusSystemPropertyUtil.isMac());
204         assertFalse(QuarkusSystemPropertyUtil.isWindows());
205         assertTrue(QuarkusSystemPropertyUtil.isUnix());
206         assertFalse(QuarkusSystemPropertyUtil.isSolaris());
207         break;
208       }
209       case UNSUPPORTED: {
210         assertFalse(QuarkusSystemPropertyUtil.isMac());
211         assertFalse(QuarkusSystemPropertyUtil.isWindows());
212         assertFalse(QuarkusSystemPropertyUtil.isUnix());
213         assertFalse(QuarkusSystemPropertyUtil.isSolaris());
214         break;
215       }
216       case WINDOWS: {
217         assertFalse(QuarkusSystemPropertyUtil.isMac());
218         assertTrue(QuarkusSystemPropertyUtil.isWindows());
219         assertFalse(QuarkusSystemPropertyUtil.isUnix());
220         assertFalse(QuarkusSystemPropertyUtil.isSolaris());
221         break;
222       }
223       default: {
224       }
225     }
226   }

228   @Test
229   final void testSystemPropertyError() {
230     assertThrows(CcsInvalidArgumentRuntimeException.class, () -> QuarkusSystemPropertyUtil.contains(null));
231     assertThrows(CcsInvalidArgumentRuntimeException.class, () -> QuarkusSystemPropertyUtil.get(null));
232     assertThrows(CcsInvalidArgumentRuntimeException.class, () -> QuarkusSystemPropertyUtil.get(null, KEY_IVALUE));
233     assertThrows(CcsInvalidArgumentRuntimeException.class, () -> QuarkusSystemPropertyUtil.get(null, KEY_BVALUE));
234     assertThrows(CcsInvalidArgumentRuntimeException.class, () -> QuarkusSystemPropertyUtil.get(null, KEY_LVALUE));
235     assertThrows(CcsInvalidArgumentRuntimeException.class, () -> QuarkusSystemPropertyUtil.getAndSet(null, KEY_VALUE));
236     assertThrows(CcsInvalidArgumentRuntimeException.class, () -> QuarkusSystemPropertyUtil.getAndSet(null, KEY_BVALUE));
237     assertThrows(CcsInvalidArgumentRuntimeException.class, () -> QuarkusSystemPropertyUtil.getAndSet(null, KEY_IVALUE));
238     assertThrows(CcsInvalidArgumentRuntimeException.class, () -> QuarkusSystemPropertyUtil.getAndSet(null, KEY_LVALUE));
239     assertThrows(CcsInvalidArgumentRuntimeException.class, () -> QuarkusSystemPropertyUtil.set(null, KEY_VALUE));
240     assertThrows(CcsInvalidArgumentRuntimeException.class, () -> QuarkusSystemPropertyUtil.set(null, KEY_BVALUE));
241     assertThrows(CcsInvalidArgumentRuntimeException.class, () -> QuarkusSystemPropertyUtil.set(null, KEY_IVALUE));
242     assertThrows(CcsInvalidArgumentRuntimeException.class, () -> QuarkusSystemPropertyUtil.set(null, KEY_LVALUE));
243   }
