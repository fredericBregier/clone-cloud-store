41  @ApplicationScoped
42  public class LocalReplicatorRequestTopicConsumer {
43    private static final Logger LOGGER = Logger.getLogger(LocalReplicatorRequestTopicConsumer.class);
44    private final RemoteReplicatorApiClientFactory remoteReplicatorApiClientFactory;
45    private final LocalReplicatorService replicatorService;
46  
47    public LocalReplicatorRequestTopicConsumer(final RemoteReplicatorApiClientFactory remoteReplicatorApiClientFactory,
48                                               final LocalReplicatorService replicatorService) {
49      this.remoteReplicatorApiClientFactory = remoteReplicatorApiClientFactory;
50      this.replicatorService = replicatorService;
51    }
52  
53    private void prepareSend(final ReplicatorOrder replicatorOrder,
54                             final Map<String, List<ReplicatorOrder>> mapItemsPerTarget) {
55      final var list = mapItemsPerTarget.computeIfAbsent(replicatorOrder.toSite(), k -> new ArrayList<>());
56      list.add(replicatorOrder);
57    }
58  
59    @Incoming(ReplicatorConstants.Topic.REPLICATOR_REQUEST_IN)
60    @Blocking
61    public void consumeOrder(final List<ReplicatorOrder> replicatorOrders) {
62      QuarkusProperties.refreshModuleMdc();
63      final Map<String, Topology> mapTopologies;
64      try {
65        final var topologiesTemp = replicatorService.getTopologies(null).stream().distinct().toList();
66        final var optional =
67            topologiesTemp.stream().filter(topology -> Objects.equals(topology.id(), ServiceProperties.getAccessorSite()))
68                .findFirst();
69        mapTopologies = topologiesTemp.stream().collect(Collectors.toMap(Topology::id, topology -> topology));
70        optional.ifPresent(topology -> mapTopologies.remove(topology.id()));
71      } catch (final CcsWithStatusException e) {
72        LOGGER.error(e.getMessage());
73        // Element in error will be ignored, letting the reconciliation doing the job
74        return;
75      }
76      final var mapItemsPerTarget = new HashMap<String, List<ReplicatorOrder>>();
77      try {
78        // Element in error will be ignored, letting the reconciliation doing the job
79        // TODO However could optimize reconciliation by storing the ReplicatorOrders in error to redo them
80        prepareOrders(replicatorOrders, mapItemsPerTarget, mapTopologies);
81        for (final var set : mapItemsPerTarget.entrySet()) {
82          final var list = set.getValue();
83          if (list != null && !list.isEmpty()) {
84            final var topology = mapTopologies.get(set.getKey());
85            createOrderWithClientAssociatedWithTopology(topology, list);
86            list.clear();
87          }
88        }
89      } finally {
90        mapItemsPerTarget.clear();
91        mapTopologies.clear();
92      }
93    }
94  
95    private void createOrderWithClientAssociatedWithTopology(final Topology topology, final List<ReplicatorOrder> list) {
96      try (final var client = remoteReplicatorApiClientFactory.newClient(URI.create(topology.uri()))) {
97        try {
98          if (list.size() == 1) {
99            client.createOrder(list.getFirst()).await().atMost(QuarkusProperties.getDurationResponseTimeout()).close();
100         } else {
101           client.createOrders(list).await().atMost(QuarkusProperties.getDurationResponseTimeout()).close();
102         }
103       } catch (final RuntimeException e) {
104         // Ignored
105         LOGGER.infof(e.getMessage());
106       }
107     }
108   }
109 
110   private void prepareOrders(final List<ReplicatorOrder> replicatorOrders,
111                              final HashMap<String, List<ReplicatorOrder>> mapItemsPerTarget,
112                              final Map<String, Topology> mapTopologies) {
113     for (final var replicatorOrder : replicatorOrders) {
114       LOGGER.debugf("Will Send to all %s", replicatorOrder);
115       final var toSite = replicatorOrder.toSite();
116       if (ParametersChecker.isNotEmpty(toSite)) {
117         prepareSend(replicatorOrder, mapItemsPerTarget);
118       } else {
119         for (final var topology : mapTopologies.keySet()) {
120           final var clone = new ReplicatorOrder(replicatorOrder, topology);
121           prepareSend(clone, mapItemsPerTarget);
122         }
123       }
124     }
125   }
