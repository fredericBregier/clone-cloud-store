129       "local accessor")
130   @Blocking
131   public Uni<Response> remoteReadObject(@PathParam("bucketName") final String bucketName,
132                                         @PathParam("objectName") final String objectName,
133                                         @HeaderParam(ACCEPT) final String acceptHeader,
134                                         @DefaultValue(MediaType.APPLICATION_OCTET_STREAM) @HeaderParam(ACCEPT_ENCODING) final String acceptEncodingHeader,
135                                         @Parameter(name = AccessorConstants.Api.X_CLIENT_ID, description = "Client " +

137                                             , required = true) @HeaderParam(X_CLIENT_ID) final String xClientId,
138                                         @Parameter(name = X_OP_ID, description = "Operation ID", in =
139                                             ParameterIn.HEADER, schema = @Schema(type = SchemaType.STRING), required
140                                             = false) @HeaderParam(X_OP_ID) final String xOpId,
141                                         final HttpServerRequest request, @Context final Closer closer) {
142     final var decodedName = ParametersChecker.getSanitizedName(objectName);
143     LOGGER.debugf("Remote read object [%s] from bucket [%s]", decodedName, bucketName);
144     final var replicatorObject =
145         new ReplicatorOrder(xOpId, ServiceProperties.getAccessorSite(), ServiceProperties.getAccessorSite(), xClientId,
146             bucketName, decodedName, 0, null, ReplicatorConstants.Action.UNKNOWN);
147     // TODO Handle compression parameter, currently set to false
148     // TODO choose compression model
149     return readObject(request, closer, replicatorObject, false);

152   @HEAD
153   @Tag(name = ReplicatorConstants.Api.TAG_REPLICATOR + REMOTE)
154   @Path(ReplicatorConstants.Api.COLL_BUCKETS + "/{bucketName}/{pathDirectoryOrObject:.+}")
155   @APIResponse(responseCode = "204", description = "OK", headers = {
156       @Header(name = X_TYPE, description = "Type as StorageType", schema = @Schema(type = SchemaType.STRING))})
157   @APIResponse(responseCode = "401", description = "Unauthorized")
158   @APIResponse(responseCode = "404", description = "Object not found")
159   @APIResponse(responseCode = "500", description = "Internal Error")
160   @Operation(summary = "Check if object exists on a remote replicator", description = "Loops through the topology and" +
161       " search for a remote replicator owning the object")
162   @Blocking
163   public Uni<Response> checkObjectOrDirectory(@PathParam("bucketName") final String bucketName,
164                                               @PathParam("pathDirectoryOrObject") final String pathDirectoryOrObject,
165                                               @Parameter(name = FULL_CHECK, description = "If True implies Storage " +
166                                                   "checking", in = ParameterIn.QUERY, schema = @Schema(type =
167                                                   SchemaType.BOOLEAN), required = false) @DefaultValue("false") @QueryParam(AccessorConstants.Api.FULL_CHECK) final boolean fullCheck,
168                                               @Parameter(name = AccessorConstants.Api.X_CLIENT_ID, description =
169                                                   "Client ID", in = ParameterIn.HEADER, schema = @Schema(type =
170                                                   SchemaType.STRING), required = true) @HeaderParam(AccessorConstants.Api.X_CLIENT_ID) final String xClientId,
171                                               @HeaderParam(X_OP_ID) final String xOpId) {
172     return Uni.createFrom().emitter(em -> {
173       LOGGER.debugf("Check object exists : [bucket:%s][objectPath:%s][clientId:%s]", bucketName, pathDirectoryOrObject,
174           xClientId);
175       try (final var client = accessorObjectInternalApiFactory.newClient()) {
176         client.setOpId(xOpId);
177         final var storageType = client.checkObjectOrDirectory(bucketName, pathDirectoryOrObject, xClientId, fullCheck);
178         em.complete(Response.status(StorageType.NONE.equals(storageType) ? Status.NOT_FOUND : Status.OK)
179             .header(AccessorConstants.Api.X_TYPE, storageType).build());
180       } catch (final CcsWithStatusException e) {
181         LOGGER.errorf("Could not check object on any remote replicator: %s", e.getMessage());
182         if (e.getStatus() == Response.Status.NOT_FOUND.getStatusCode()) {
183           em.complete(Response.status(Response.Status.NOT_FOUND).header(AccessorConstants.Api.X_TYPE, StorageType.NONE)
184               .build());
185         } else {
186           em.complete(Response.status(Response.Status.INTERNAL_SERVER_ERROR)
187               .header(AccessorConstants.Api.X_TYPE, StorageType.NONE).build());
188         }
189       } catch (final RuntimeException e) {
190         ServerResponseFilter.handleException(em, e);
191       }
192     });
193   }

248   @Blocking
249   public Uni<AccessorBucket> getBucket(@PathParam("bucketName") final String bucketName,
250                                        @Parameter(name = X_CLIENT_ID, description = "Client ID", in =
251                                            ParameterIn.HEADER, schema = @Schema(type = SchemaType.STRING), required =
252                                            true) @HeaderParam(X_CLIENT_ID) String xClientId,
253                                        @Parameter(name = X_OP_ID, description = "Operation ID", in =
254                                            ParameterIn.HEADER, schema = @Schema(type = SchemaType.STRING), required =
255                                            false) @HeaderParam(X_OP_ID) final String xOpId) {
256     return Uni.createFrom().emitter(em -> {
257       LOGGER.debugf("Get bucket : [bucket:%s][clientId:%s]", bucketName, xClientId);
258       try (final var client = accessorBucketInternalApiFactory.newClient()) {
259         client.setOpId(xOpId);
260         final var bucket = client.getBucket(bucketName, xClientId);
261         em.complete(bucket);
262       } catch (final CcsWithStatusException e) {
263         LOGGER.errorf("Could not check bucket on any remote replicator: %s", e.getMessage());
264         if (e.getStatus() == Response.Status.NOT_FOUND.getStatusCode()) {
265           em.fail(e);
266         } else {
267           em.fail(new CcsOperationException(e));
268         }
269       } catch (final RuntimeException e) {
270         ServerResponseFilter.handleExceptionFail(em, e);
271       }
272     });
