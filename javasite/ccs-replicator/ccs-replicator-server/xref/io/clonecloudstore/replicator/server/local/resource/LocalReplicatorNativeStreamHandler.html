19  import java.io.InputStream;

33  import io.clonecloudstore.replicator.model.ReplicatorOrder;
34  import io.clonecloudstore.replicator.server.local.application.LocalReplicatorService;
35  import io.clonecloudstore.replicator.server.remote.client.RemoteReplicatorApiClient;
36  import io.clonecloudstore.replicator.server.remote.client.RemoteReplicatorApiClientFactory;

39  import jakarta.enterprise.context.RequestScoped;
40  import jakarta.enterprise.inject.spi.CDI;
41  import org.jboss.logging.Logger;
42  
43  @RequestScoped
44  public class LocalReplicatorNativeStreamHandler extends NativeStreamHandlerAbstract<ReplicatorOrder, AccessorObject> {
45    private static final Logger LOGGER = Logger.getLogger(LocalReplicatorNativeStreamHandler.class);
46    private static final String COULD_NOT_REMOTE_READ = "Could not remote read from any of remote replicators: %s";
47    private RemoteReplicatorApiClientFactory remoteReplicatorApiClientFactory;
48    private LocalReplicatorService localReplicatorService;
49    private AccessorObject fromInputStream = null;
50    private Topology topologyFound;
51  
52    @Override
53    protected void postSetup() {
54      super.postSetup();
55      // No native CDI
56      remoteReplicatorApiClientFactory = CDI.current().select(RemoteReplicatorApiClientFactory.class).get();
57      localReplicatorService = CDI.current().select(LocalReplicatorService.class).get();
58    }
59  
60    @Override
61    protected boolean checkDigestToCumpute(final ReplicatorOrder businessIn) {
62      return false;
63    }
64  
65    @Override
66    protected void checkPushAble(final ReplicatorOrder businessIn, final MultipleActionsInputStream inputStream)
67        throws CcsClientGenericException, CcsServerGenericException {
68      // NO PUSH
69    }
70  
71    @Override
72    protected AccessorObject getAnswerPushInputStream(final ReplicatorOrder replicatorOrder, final String finalHash,
73                                                      final long size) {
74      // NO PUSH
75      return null;
76    }
77  
78    @Override
79    protected Map<String, String> getHeaderPushInputStream(final ReplicatorOrder replicatorOrder, final String finalHash,
80                                                           final long size, final AccessorObject accessorObject) {
81      // NO PUSH
82      return replicatorOrder.getHeaders();
83    }
84  
85    @Override
86    protected boolean checkPullAble(final ReplicatorOrder replicatorOrder, final MultiMap headers) {
87      try {
88        final var topologyFoundOptional =
89            localReplicatorService.findValidTopologyObject(replicatorOrder.bucketName(), replicatorOrder.objectName(),
90                false, replicatorOrder.clientId(), replicatorOrder.toSite(), getOpId());
91        topologyFound = topologyFoundOptional.orElse(null);
92      } catch (CcsWithStatusException e) {
93        throw CcsServerGenericExceptionMapper.getCcsException(e.getStatus(), e.getMessage(), e);
94      }
95      return topologyFound != null;
96    }
97  
98    @Override
99    protected InputStream getPullInputStream(final ReplicatorOrder replicatorObject) {
100     RemoteReplicatorApiClient client = null;
101     try {
102       // Store output object for getHeaderPullInputStream
103       client = remoteReplicatorApiClientFactory.newClient(URI.create(topologyFound.uri()));
104       final var result = client.readRemoteObject(replicatorObject.bucketName(), replicatorObject.objectName(),
105           replicatorObject.clientId(), getOpId(), 0);
106       fromInputStream = result.dtoOut();
107       getCloser().add(client);
108       return result.inputStream();
109     } catch (final CcsServerGenericException | CcsClientGenericException e) {
110       LOGGER.errorf(COULD_NOT_REMOTE_READ, e.getMessage());
111       throw e;
112     } catch (final CcsWithStatusException e) {
113       LOGGER.errorf(COULD_NOT_REMOTE_READ, e.getMessage());
114       throw CcsServerGenericExceptionMapper.getCcsException(e.getStatus(), e.getMessage(), e);
115     } catch (final RuntimeException e) {
116       LOGGER.errorf(COULD_NOT_REMOTE_READ, e.getMessage());
117       throw new CcsOperationException(e.getMessage(), e);
118     }
119   }
120 
121   @Override
122   protected Map<String, String> getHeaderPullInputStream(final ReplicatorOrder replicatorOrder) {
123     if (fromInputStream != null) {
124       final Map<String, String> map = new HashMap<>();
125       AccessorHeaderDtoConverter.objectToMap(fromInputStream, map);
126       return map;
127     }
128     return replicatorOrder.getHeaders();
129   }
130 
131   @Override
132   protected Map<String, String> getHeaderError(final ReplicatorOrder replicatorOrder, final int status) {
133     return replicatorOrder.getHeaders();
134   }
135 }
