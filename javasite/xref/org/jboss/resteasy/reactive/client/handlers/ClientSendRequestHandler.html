100   @Override
101   public void handle(RestClientRequestContext requestContext) {
102     if (requestContext.isAborted()) {
103       return;
104     }
105     requestContext.suspend();
106     Uni<HttpClientRequest> future = createRequest(requestContext);
107 
108     // DNS failures happen before we send the request
109     future.subscribe().with(new Consumer<>() {
110       @Override
111       public void accept(HttpClientRequest httpClientRequest) {
112         // adapt headers to HTTP/2 depending on the underlying HTTP connection
113         ClientSendRequestHandler.this.adaptRequest(httpClientRequest);
114 
115         if (requestContext.isMultipart()) {
116           Promise<HttpClientRequest> requestPromise = Promise.promise();
117           QuarkusMultipartFormUpload actualEntity;
118           try {
119             actualEntity =
120                 ClientSendRequestHandler.this.setMultipartHeadersAndPrepareBody(httpClientRequest, requestContext);
121             if (loggingScope != LoggingScope.NONE) {
122               clientLogger.logRequest(httpClientRequest, null, true);
123             }
124 
125             Pipe<Buffer> pipe = actualEntity.pipe(); // Shouldn't this be called in an earlier phase ?
126             requestPromise.future().onComplete(ar -> {
127 
128               if (ar.succeeded()) {
129                 HttpClientRequest req = ar.result();
130                 if (httpClientRequest.headers() == null ||
131                     !httpClientRequest.headers().contains(HttpHeaders.CONTENT_LENGTH)) {
132                   req.setChunked(true);
133                 }
134                 pipe.endOnFailure(false);
135                 pipe.to(req, ar2 -> {
136                   if (ar2.failed()) {
137                     req.reset(0L, ar2.cause());
138                   }
139                 });
140                 actualEntity.run();
141               } else {
142                 pipe.close();
143               }
144             });
145             Future<HttpClientResponse> sent = httpClientRequest.response();
146             requestPromise.complete(httpClientRequest);
147             attachSentHandlers(sent, httpClientRequest, requestContext);
148           } catch (Throwable e) {
149             reportFinish(e, requestContext);
150             requestContext.resume(e);
151             return;
152           }
153         } else if (requestContext.isFileUpload()) {
154           Vertx vertx = Vertx.currentContext().owner();
155           Object entity = requestContext.getEntity().getEntity();
156           String filePathToUpload = null;
157           if (entity instanceof File) {
158             filePathToUpload = ((File) entity).getAbsolutePath();
159           } else if (entity instanceof Path) {
160             filePathToUpload = ((Path) entity).toAbsolutePath().toString();
161           }
162           vertx.fileSystem().open(filePathToUpload, new OpenOptions().setRead(true).setWrite(false), new Handler<>() {
163             @Override
164             public void handle(AsyncResult<AsyncFile> openedAsyncFile) {
165               if (openedAsyncFile.failed()) {
166                 requestContext.resume(openedAsyncFile.cause());
167                 return;
168               }
169 
170               MultivaluedMap<String, String> headerMap = requestContext.getRequestHeaders().asMap();
171               updateRequestHeadersFromConfig(requestContext, headerMap);
172 
173               // set the Vertx headers after we've run the interceptors because they can modify them
174               setVertxHeaders(httpClientRequest, headerMap);
175 
176               Future<HttpClientResponse> sent = httpClientRequest.send(openedAsyncFile.result());
177               attachSentHandlers(sent, httpClientRequest, requestContext);
178             }
179           });
180         } else if (requestContext.getEntity() != null &&
181             requestContext.getEntity().getEntity() instanceof InputStream) {
182           Future<HttpClientResponse> sent;
183           ReadStream<Buffer> actualEntity;
184           try {
185             actualEntity =
186                 ClientSendRequestHandler.this.setRequestHeadersForSendingInputStream(httpClientRequest, requestContext);
187             sent = httpClientRequest.send(actualEntity);
188             if (loggingScope != LoggingScope.NONE) {
189               clientLogger.logRequest(httpClientRequest, null, false);
190             }
191             attachSentHandlers(sent, httpClientRequest, requestContext);
192           } catch (Throwable e) {
193             requestContext.resume(e);
194           }
195         } else {
196           Future<HttpClientResponse> sent;
197           Buffer actualEntity;
198           try {
199             actualEntity =
200                 ClientSendRequestHandler.this.setRequestHeadersAndPrepareBody(httpClientRequest, requestContext);
201           } catch (Throwable e) {
202             requestContext.resume(e);
203             return;
204           }
205           if (actualEntity == AsyncInvokerImpl.EMPTY_BUFFER) {
206             sent = httpClientRequest.send();
207             if (loggingScope != LoggingScope.NONE) {
208               clientLogger.logRequest(httpClientRequest, null, false);
209             }
210           } else {
211             sent = httpClientRequest.send(actualEntity);
212             if (loggingScope != LoggingScope.NONE) {
213               clientLogger.logRequest(httpClientRequest, actualEntity, false);
214             }
215           }
216           attachSentHandlers(sent, httpClientRequest, requestContext);
217         }
218       }
219     }, new Consumer<>() {
220       @Override
221       public void accept(Throwable event) {
222         // set some properties to prevent NPEs down the chain
223         requestContext.setResponseHeaders(new MultivaluedTreeMap<>());
224         requestContext.setResponseReasonPhrase("unknown");
225 
226         if (event instanceof IOException) {
227           ProcessingException throwable = new ProcessingException(event);
228           reportFinish(throwable, requestContext);
229           requestContext.resume(throwable);
230         } else {
231           requestContext.resume(event);
232           reportFinish(event, requestContext);
233         }
234       }
235     });
236   }

238   private void attachSentHandlers(Future<HttpClientResponse> sent, HttpClientRequest httpClientRequest,
239                                   RestClientRequestContext requestContext) {
240     sent.onSuccess(new Handler<>() {
241       @Override
242       public void handle(HttpClientResponse clientResponse) {
243         try {
244           requestContext.initialiseResponse(clientResponse);
245           int status = clientResponse.statusCode();
246           if (requestContext.getCallStatsCollector() != null) {
247             if (status >= 500 && status < 600) {
248               reportFinish(new InternalServerErrorException(), requestContext);
249             } else {
250               reportFinish(null, requestContext);
251             }
252           }
253 
254           if (isResponseMultipart(requestContext)) {
255             QuarkusMultipartResponseDecoder multipartDecoder = new QuarkusMultipartResponseDecoder(clientResponse);
256 
257             clientResponse.handler(multipartDecoder::offer);
258 
259             clientResponse.endHandler(new Handler<>() {
260               @Override
261               public void handle(Void event) {
262                 multipartDecoder.offer(LastHttpContent.EMPTY_LAST_CONTENT);
263 
264                 List<InterfaceHttpData> datas = multipartDecoder.getBodyHttpDatas();
265                 requestContext.setResponseMultipartParts(datas);
266 
267                 if (loggingScope != LoggingScope.NONE) {
268                   clientLogger.logResponse(clientResponse, false);
269                 }
270 
271                 requestContext.resume();
272               }
273             });
274           } else if (!requestContext.isRegisterBodyHandler()) {
275             clientResponse.pause();
276             if (loggingScope != LoggingScope.NONE) {
277               clientLogger.logResponse(clientResponse, false);
278             }
279             requestContext.resume();
280           } else {
281             if (requestContext.isFileDownload()) {
282               // when downloading a file we copy the bytes to the file system and manually set the entity type
283               // this is needed because large files can cause OOM or exceed the InputStream limit (of 2GB)
284 
285               clientResponse.pause();
286               Vertx vertx = Vertx.currentContext().owner();
287               vertx.fileSystem().createTempFile("rest-client", "", new Handler<>() {
288                 @Override
289                 public void handle(AsyncResult<String> tempFileCreation) {
290                   if (tempFileCreation.failed()) {
291                     reportFinish(tempFileCreation.cause(), requestContext);
292                     requestContext.resume(tempFileCreation.cause());
293                     return;
294                   }
295                   String tmpFilePath = tempFileCreation.result();
296                   vertx.fileSystem().open(tmpFilePath, new OpenOptions().setWrite(true), new Handler<>() {
297                     @Override
298                     public void handle(AsyncResult<AsyncFile> asyncFileOpened) {
299                       if (asyncFileOpened.failed()) {
300                         reportFinish(asyncFileOpened.cause(), requestContext);
301                         requestContext.resume(asyncFileOpened.cause());
302                         return;
303                       }
304                       final AsyncFile tmpAsyncFile = asyncFileOpened.result();
305                       final Pump downloadPump = Pump.pump(clientResponse, tmpAsyncFile);
306                       downloadPump.start();
307 
308                       clientResponse.resume();
309                       clientResponse.endHandler(new Handler<>() {
310                         public void handle(Void event) {
311                           tmpAsyncFile.flush(new Handler<>() {
312                             public void handle(AsyncResult<Void> flushed) {
313                               if (flushed.failed()) {
314                                 reportFinish(flushed.cause(), requestContext);
315                                 requestContext.resume(flushed.cause());
316                                 return;
317                               }
318 
319                               if (loggingScope != LoggingScope.NONE) {
320                                 clientLogger.logRequest(httpClientRequest, null, false);
321                               }
322 
323                               requestContext.setTmpFilePath(tmpFilePath);
324                               requestContext.resume();
325                             }
326                           });
327                         }
328                       });
329                     }
330                   });
331                 }
332               });
333 
334             } else if (requestContext.isInputStreamDownload()) {
335               if (loggingScope != LoggingScope.NONE) {
336                 clientLogger.logResponse(clientResponse, false);
337               }
338               //TODO: make timeout configureable
339               requestContext.setResponseEntityStream(
340                   new VertxClientInputStream(clientResponse, 100000, requestContext));
341               requestContext.resume();
342             } else {
343               clientResponse.body(new Handler<>() {
344                 @Override
345                 public void handle(AsyncResult<Buffer> ar) {
346                   if (ar.succeeded()) {
347                     if (loggingScope != LoggingScope.NONE) {
348                       clientLogger.logResponse(clientResponse, false);
349                     }
350                     Buffer buffer = ar.result();
351                     try {
352                       if (buffer.length() > 0) {
353                         requestContext.setResponseEntityStream(new ByteArrayInputStream(buffer.getBytes()));
354                       } else {
355                         requestContext.setResponseEntityStream(null);
356                       }
357                       requestContext.resume();
358                     } catch (Throwable t) {
359                       requestContext.resume(t);
360                     }
361                   } else {
362                     requestContext.resume(ar.cause());
363                   }
364                 }
365               });
366             }
367           }
368         } catch (Throwable t) {
369           reportFinish(t, requestContext);
370           requestContext.resume(t);
371         }
372       }
373     }).onFailure(new Handler<>() {
374       @Override
375       public void handle(Throwable failure) {
376         if (failure instanceof HttpClosedException) {
377           // This is because of the Rest Client TCK
378           // HttpClosedException is a runtime exception. If we complete with that exception, it gets
379           // unwrapped by the rest client proxy and thus fails the TCK.
380           // By creating an IOException, we avoid that and provide a meaningful exception (because
381           // it's an I/O exception)
382           requestContext.resume(new ProcessingException(new IOException(failure.getMessage())));
383         } else if (failure instanceof IOException) {
384           requestContext.resume(new ProcessingException(failure));
385         } else {
386           requestContext.resume(failure);
387         }
388       }
389     });
390   }

392   private boolean isResponseMultipart(RestClientRequestContext requestContext) {
393     MultivaluedMap<String, String> responseHeaders = requestContext.getResponseHeaders();
394     List<String> contentTypes = responseHeaders.get(CONTENT_TYPE);
395     if (contentTypes != null) {
396       for (String contentType : contentTypes) {
397         if (contentType.toLowerCase(Locale.ROOT).startsWith(MediaType.MULTIPART_FORM_DATA)) {
398           return true;
399         }
400       }
401     }
402     return false;
403   }

405   private void reportFinish(Throwable throwable, RestClientRequestContext requestContext) {
406     ServiceInstance serviceInstance = requestContext.getCallStatsCollector();
407     if (serviceInstance != null) {
408       serviceInstance.recordReply();
409       serviceInstance.recordEnd(throwable);
410     }
411   }

413   public Uni<HttpClientRequest> createRequest(RestClientRequestContext state) {
414     HttpClient httpClient = state.getHttpClient();
415     URI uri = state.getUri();
416     Object readTimeout = state.getConfiguration().getProperty(QuarkusRestClientProperties.READ_TIMEOUT);
417     Uni<RequestOptions> requestOptions;
418     state.setMultipartResponsesData(multipartResponseDataMap);
419     if (uri.getScheme() == null) { // invalid URI
420       return Uni.createFrom().failure(new IllegalArgumentException("Invalid REST Client URL used: '" + uri + "'"));
421     }
422     try {
423       URL ignored = uri.toURL();
424     } catch (MalformedURLException mue) {
425       log.error("Invalid REST Client URL used: '" + uri + "'");
426       return Uni.createFrom().failure(new IllegalArgumentException("Invalid REST Client URL used: '" + uri + "'"));
427     }
428 
429     boolean isHttps = "https".equals(uri.getScheme());
430     int port = getPort(isHttps, uri.getPort());
431     requestOptions = Uni.createFrom().item(new RequestOptions().setHost(uri.getHost()).setPort(port).setSsl(isHttps));
432 
433     return requestOptions.onItem().transform(r -> r.setMethod(HttpMethod.valueOf(state.getHttpMethod()))
434         .setURI(uri.getRawPath() + (uri.getRawQuery() == null ? "" : "?" + uri.getRawQuery()))
435         .setFollowRedirects(followRedirects)).onItem().invoke(r -> {
436       if (readTimeout instanceof Long) {
437         r.setTimeout((Long) readTimeout);
438       }
439     }).onItem().transformToUni(new Function<RequestOptions, Uni<? extends HttpClientRequest>>() {
440       @Override
441       public Uni<? extends HttpClientRequest> apply(RequestOptions options) {
442         return AsyncResultUni.toUni(handler -> httpClient.request(options, handler));
443       }
444     });
445   }

459   private QuarkusMultipartFormUpload setMultipartHeadersAndPrepareBody(HttpClientRequest httpClientRequest,
460                                                                        RestClientRequestContext state)
461       throws Exception {
462     if (!(state.getEntity().getEntity() instanceof QuarkusMultipartForm)) {
463       throw new IllegalArgumentException(
464           "Multipart form upload expects an entity of type MultipartForm, got: " + state.getEntity().getEntity());
465     }
466 
467     MultivaluedMap<String, String> headerMap = state.getRequestHeaders().asMap();
468     updateRequestHeadersFromConfig(state, headerMap);
469     QuarkusMultipartForm multipartForm = (QuarkusMultipartForm) state.getEntity().getEntity();
470     multipartForm.preparePojos(state);
471 
472     Object property = state.getConfiguration().getProperty(QuarkusRestClientProperties.MULTIPART_ENCODER_MODE);
473     PausableHttpPostRequestEncoder.EncoderMode mode = PausableHttpPostRequestEncoder.EncoderMode.RFC1738;
474     if (property != null) {
475       mode = (PausableHttpPostRequestEncoder.EncoderMode) property;
476     }
477     QuarkusMultipartFormUpload multipartFormUpload =
478         new QuarkusMultipartFormUpload(Vertx.currentContext(), multipartForm, true, maxChunkSize, mode);
479     httpClientRequest.setChunked(multipartFormUpload.isChunked());
480     setEntityRelatedHeaders(headerMap, state.getEntity());
481 
482     // multipart has its own headers:
483     MultiMap multipartHeaders = multipartFormUpload.headers();
484     for (String multipartHeader : multipartHeaders.names()) {
485       headerMap.put(multipartHeader, multipartHeaders.getAll(multipartHeader));
486     }
487 
488     if (state.getEntity().getVariant() != null) {
489       Variant v = state.getEntity().getVariant();
490       String variantContentType = v.getMediaType().toString();
491       String multipartContentType = headerMap.getFirst(HttpHeaders.CONTENT_TYPE);
492       if (multipartContentType.startsWith(HttpHeaderValues.MULTIPART_FORM_DATA.toString()) &&
493           !variantContentType.startsWith(HttpHeaderValues.MULTIPART_FORM_DATA.toString())) {
494         // this is a total hack whose purpose is to allow the @Consumes annotation to override the media type
495         int semicolonIndex = multipartContentType.indexOf(';');
496         headerMap.put(HttpHeaders.CONTENT_TYPE,
497             List.of(variantContentType + multipartContentType.substring(semicolonIndex)));
498       }
499     }
500 
501     setVertxHeaders(httpClientRequest, headerMap);
502     return multipartFormUpload;
503   }

505   private Buffer setRequestHeadersAndPrepareBody(HttpClientRequest httpClientRequest, RestClientRequestContext state)
506       throws IOException {
507     MultivaluedMap<String, String> headerMap = state.getRequestHeaders().asMap();
508     updateRequestHeadersFromConfig(state, headerMap);
509 
510     Buffer actualEntity = AsyncInvokerImpl.EMPTY_BUFFER;
511     Entity<?> entity = state.getEntity();
512     if (entity != null) {
513       // no need to set the entity.getMediaType, it comes from the variant
514       setEntityRelatedHeaders(headerMap, entity);
515 
516       actualEntity = state.writeEntity(entity, headerMap,
517           state.getConfiguration().getWriterInterceptors().toArray(Serialisers.NO_WRITER_INTERCEPTOR));
518     } else {
519       // some servers don't like the fact that a POST or PUT does not have a method body if there is no
520       // content-length header associated
521       if (state.getHttpMethod().equals("POST") || state.getHttpMethod().equals("PUT")) {
522         headerMap.putSingle(HttpHeaders.CONTENT_LENGTH, "0");
523       }
524     }
525     // set the Vertx headers after we've run the interceptors because they can modify them
526     setVertxHeaders(httpClientRequest, headerMap);
527     return actualEntity;
528   }

530   private void adaptRequest(HttpClientRequest request) {
531     if (request.version() == HttpVersion.HTTP_2) {
532       // When using the protocol HTTP/2, Netty which is internally used by Vert.x will validate the headers and reject
533       // the requests with invalid metadata.
534       // When we start a new connection, the Vert.x client will automatically upgrade the first request we make to be
535       // valid in HTTP/2.
536       // The problem is that in next requests, the Vert.x client reuses the same connection within the same window time
537       // and hence does not upgrade the following requests. Therefore, even though the first request works fine, the
538       // next requests won't work.
539       // This has been reported in https://github.com/eclipse-vertx/vert.x/issues/4618.
540       // To workaround this issue, we need to "upgrade" the next requests by ourselves when the version is already set
541       // to HTTP/2:
542       if (request.path() == null || request.path().length() == 0) {
543         // HTTP/2 does not allow empty paths
544         request.setURI(request.getURI() + "/");
545       }
546     }
547   }

549   private void updateRequestHeadersFromConfig(RestClientRequestContext state,
550                                               MultivaluedMap<String, String> headerMap) {
551     Object staticHeaders = state.getConfiguration().getProperty(QuarkusRestClientProperties.STATIC_HEADERS);
552     if (staticHeaders instanceof Map) {
553       for (Map.Entry<String, String> entry : ((Map<String, String>) staticHeaders).entrySet()) {
554         headerMap.putSingle(entry.getKey(), entry.getValue());
555       }
556     }
557   }
558 
559   private void setVertxHeaders(HttpClientRequest httpClientRequest, MultivaluedMap<String, String> headerMap) {
560     MultiMap vertxHttpHeaders = httpClientRequest.headers();
561     for (Map.Entry<String, List<String>> entry : headerMap.entrySet()) {
562       vertxHttpHeaders.add(entry.getKey(), entry.getValue());
563     }
564   }

566   private void setEntityRelatedHeaders(MultivaluedMap<String, String> headerMap, Entity<?> entity) {
567     if (entity.getVariant() != null) {
568       Variant v = entity.getVariant();
569       if (!headerMap.containsKey(HttpHeaders.CONTENT_TYPE)) {
570         headerMap.putSingle(HttpHeaders.CONTENT_TYPE, v.getMediaType().toString());
571       }
572       if ((v.getLanguageString() != null) && !headerMap.containsKey(HttpHeaders.CONTENT_LANGUAGE)) {
573         headerMap.putSingle(HttpHeaders.CONTENT_LANGUAGE, v.getLanguageString());
574       }
575       if ((v.getEncoding() != null) && !headerMap.containsKey(HttpHeaders.CONTENT_ENCODING)) {
576         headerMap.putSingle(HttpHeaders.CONTENT_ENCODING, v.getEncoding());
577       }
578     }
579   }

581   private ReadStream<Buffer> setRequestHeadersForSendingInputStream(HttpClientRequest httpClientRequest,
582                                                                     RestClientRequestContext state) {
583     MultivaluedMap<String, String> headerMap = state.getRequestHeaders().asMap();
584     updateRequestHeadersFromConfig(state, headerMap);
585     headerMap.putSingle(HttpHeaders.CONTENT_TYPE, MediaType.APPLICATION_OCTET_STREAM);
586     Entity<?> entity = state.getEntity();
587     InputStream inputStream = (InputStream) entity.getEntity();
588     httpClientRequest.setChunked(true);
589     Vertx vertx = Vertx.currentContext().owner();
590     ReadStream<Buffer> readStream = new AsyncInputStream(vertx, inputStream, maxChunkSize);
591     // set the Vertx headers after we've run the interceptors because they can modify them
592     setVertxHeaders(httpClientRequest, headerMap);
593     return readStream;
594   }
