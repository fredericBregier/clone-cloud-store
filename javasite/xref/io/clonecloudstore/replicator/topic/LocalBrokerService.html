17  package io.clonecloudstore.replicator.topic;
18  
19  import io.clonecloudstore.common.quarkus.client.SimpleClientAbstract;
20  import io.clonecloudstore.common.quarkus.exception.CcsOperationException;
21  import io.clonecloudstore.common.quarkus.modules.ServiceProperties;
22  import io.clonecloudstore.replicator.config.ReplicatorConstants.Action;
23  import io.clonecloudstore.replicator.model.ReplicatorOrder;
24  import io.quarkus.arc.Unremovable;
25  import jakarta.enterprise.context.ApplicationScoped;
26  import org.eclipse.microprofile.reactive.messaging.Channel;
27  import org.eclipse.microprofile.reactive.messaging.Emitter;
28  import org.jboss.logging.Logger;
29  
30  import static io.clonecloudstore.replicator.config.ReplicatorConstants.Action.CREATE;
31  import static io.clonecloudstore.replicator.config.ReplicatorConstants.Action.DELETE;
32  import static io.clonecloudstore.replicator.config.ReplicatorConstants.Topic.REPLICATOR_REQUEST_OUT;
33  
34  @ApplicationScoped
35  @Unremovable
36  public class LocalBrokerService {
37    private static final Logger logger = Logger.getLogger(LocalBrokerService.class);
38  
39    private final Emitter<ReplicatorOrder> replicatorOrderEmitter;
40  
41    public LocalBrokerService(@Channel(REPLICATOR_REQUEST_OUT) Emitter<ReplicatorOrder> replicatorOrderEmitter) {
42      this.replicatorOrderEmitter = replicatorOrderEmitter;
43    }
44  
45    public void createBucket(final String bucket, final String clientId) {
46      logger.debugf("Send create bucket event to broadcast bucket replication topic : %s", bucket);
47      broadcastBucket(bucket, clientId, CREATE);
48    }
49  
50    public void deleteBucket(final String bucket, final String clientId) {
51      logger.debugf("Send delete bucket event to broadcast bucket replication topic : %s", bucket);
52      broadcastBucket(bucket, clientId, DELETE);
53    }
54  
55    public void createObject(final String bucket, final String object, final String clientId, final long size,
56                             final String hash) {
57      logger.debugf("Send create object event to broadcast object replication topic :%s / %s", bucket, object);
58      broadcastObject(bucket, object, clientId, size, hash, CREATE);
59    }
60  
61    public void deleteObject(final String bucket, final String object, final String clientId) {
62      logger.debugf("Send delete object event to broadcast object replication topic ::%s / %s", bucket, object);
63      broadcastObject(bucket, object, clientId, 0, null, DELETE);
64    }
65  
66    private void broadcastObject(final String bucket, final String object, final String clientId, final long size,
67                                 final String hash, final Action action) {
68      try {
69        final var replicatorOrder =
70            new ReplicatorOrder(SimpleClientAbstract.getMdcOpId(), ServiceProperties.getAccessorSite(), null, clientId,
71                bucket, object, size, hash, action);
72        logger.debugf("Produce object replication message to broadcast object replication topic : %s", replicatorOrder);
73        replicatorOrderEmitter.send(replicatorOrder);
74      } catch (final RuntimeException e) {
75        throw new CcsOperationException("Could not send replication broadcast event", e);
76      }
77    }
78  
79    private void broadcastBucket(final String bucket, final String clientId, final Action action) {
80      try {
81        final var replicatorOrder =
82            new ReplicatorOrder(SimpleClientAbstract.getMdcOpId(), ServiceProperties.getAccessorSite(), null, clientId,
83                bucket, action);
84        logger.debugf("Produce bucket replication message to broadcast bucket replication topic : %s", replicatorOrder);
85        replicatorOrderEmitter.send(replicatorOrder);
86      } catch (final RuntimeException e) {
87        throw new CcsOperationException("Could not send replication broadcast event", e);
88      }
89    }
90  }
