18  
19  import java.io.InputStream;
20  import java.util.HashMap;
21  import java.util.Map;

23  import io.clonecloudstore.accessor.client.internal.AccessorObjectInternalApiClient;
24  import io.clonecloudstore.accessor.client.internal.AccessorObjectInternalApiFactory;
25  import io.clonecloudstore.accessor.client.model.AccessorHeaderDtoConverter;
26  import io.clonecloudstore.accessor.model.AccessorObject;
27  import io.clonecloudstore.common.quarkus.exception.CcsClientGenericException;
28  import io.clonecloudstore.common.quarkus.exception.CcsOperationException;
29  import io.clonecloudstore.common.quarkus.exception.CcsServerGenericException;
30  import io.clonecloudstore.common.quarkus.exception.CcsServerGenericExceptionMapper;
31  import io.clonecloudstore.common.quarkus.server.service.NativeStreamHandlerAbstract;
32  import io.clonecloudstore.common.standard.exception.CcsWithStatusException;
33  import io.clonecloudstore.common.standard.inputstream.MultipleActionsInputStream;
34  import io.clonecloudstore.replicator.model.ReplicatorOrder;
35  import io.vertx.core.MultiMap;
36  import jakarta.enterprise.context.RequestScoped;
37  import jakarta.enterprise.inject.spi.CDI;
38  import org.jboss.logging.Logger;
39  
40  @RequestScoped
41  public class RemoteReplicatorNativeStreamHandler extends NativeStreamHandlerAbstract<ReplicatorOrder, AccessorObject> {
42    private static final Logger LOGGER = Logger.getLogger(RemoteReplicatorNativeStreamHandler.class);
43    private static final String COULD_NOT_REMOTE_READ = "Could not remote read from any of remote replicators: %s";
44    AccessorObjectInternalApiFactory accessorObjectInternalApiFactory;
45    private AccessorObject fromInputStream = null;
46  
47    @Override
48    protected void postSetup() {
49      super.postSetup();
50      accessorObjectInternalApiFactory = CDI.current().select(AccessorObjectInternalApiFactory.class).get();
51    }
52  
53    @Override
54    protected boolean checkDigestToCumpute(final ReplicatorOrder businessIn) {
55      return false;
56    }
57  
58    @Override
59    protected AccessorObject getAnswerPushInputStream(final ReplicatorOrder replicatorOrder, final String finalHash,
60                                                      final long size) {
61      // NO PUSH
62      return null;
63    }
64  
65    @Override
66    protected Map<String, String> getHeaderPushInputStream(final ReplicatorOrder replicatorOrder, final String finalHash,
67                                                           final long size, final AccessorObject accessorObject) {
68      // NO PUSH
69      return replicatorOrder.getHeaders();
70    }
71  
72    @Override
73    protected void checkPushAble(final ReplicatorOrder payload, final MultipleActionsInputStream inputStream) {
74      // NO PUSH
75    }
76  
77    @Override
78    protected boolean checkPullAble(final ReplicatorOrder replicatorOrder, final MultiMap headers) {
79      // Check shall be done before and will be done implicitly during access
80      return true;
81    }
82  
83    @Override
84    protected InputStream getPullInputStream(final ReplicatorOrder replicatorOrder) {
85      AccessorObjectInternalApiClient client = null;
86      try {
87        client = accessorObjectInternalApiFactory.newClient();
88        getCloser().add(client);
89        // Store output object for getHeaderPullInputStream
90        client.setOpId(getOpId());
91        final var result =
92            client.getObject(replicatorOrder.bucketName(), replicatorOrder.objectName(), replicatorOrder.clientId());
93        fromInputStream = result.dtoOut();
94        return result.inputStream();
95      } catch (final CcsClientGenericException | CcsServerGenericException e) {
96        LOGGER.errorf(COULD_NOT_REMOTE_READ, e.getMessage());
97        throw e;
98      } catch (final CcsWithStatusException e) {
99        LOGGER.errorf(COULD_NOT_REMOTE_READ, e.getMessage());
100       throw CcsServerGenericExceptionMapper.getCcsException(e.getStatus(), e.getMessage(), e);
101     } catch (final RuntimeException e) {
102       LOGGER.errorf(COULD_NOT_REMOTE_READ, e.getMessage());
103       throw new CcsOperationException(e.getMessage(), e);
104     }
105   }
106 
107   @Override
108   protected Map<String, String> getHeaderPullInputStream(final ReplicatorOrder replicatorOrder) {
109     if (fromInputStream != null) {
110       final Map<String, String> map = new HashMap<>();
111       AccessorHeaderDtoConverter.objectToMap(fromInputStream, map);
112       return map;
113     }
114     return replicatorOrder.getHeaders();
115   }
116 
117   @Override
118   protected Map<String, String> getHeaderError(final ReplicatorOrder replicatorOrder, final int status) {
119     return replicatorOrder.getHeaders();
120   }
121 }
