37  @ApplicationScoped
38  public class LocalReplicatorService {
39    private static final Logger LOGGER = Logger.getLogger(LocalReplicatorService.class);
40    private final TopologyApiClientFactory topologyApiClientFactory;
41    private final RemoteReplicatorApiClientFactory remoteReplicatorApiClientFactory;
42  
43    public LocalReplicatorService(final TopologyApiClientFactory topologyApiClientFactory,
44                                  final RemoteReplicatorApiClientFactory remoteReplicatorApiClientFactory) {
45      this.topologyApiClientFactory = topologyApiClientFactory;
46      this.remoteReplicatorApiClientFactory = remoteReplicatorApiClientFactory;
47    }
48  
49    public Collection<Topology> getTopologies(final String topologyId) throws CcsWithStatusException {
50      final Collection<Topology> topologies;
51      try (final var topologyClient = topologyApiClientFactory.newClient()) {
52        if (ParametersChecker.isNotEmpty(topologyId)) {
53          topologies = new ArrayList<>();
54          final var topology = topologyClient.findBySite(topologyId);
55          if (topology != null) {
56            topologies.add(topology);
57          }
58        } else {
59          topologies = topologyClient.listWithStatus(TopologyStatus.UP);
60        }
61      }
62      return topologies;
63    }
64  
65    public Optional<Topology> findValidTopologyBucket(final String bucket, final boolean fullCheck, final String clientId,
66                                                      final String topologyId, final String opId)
67        throws CcsWithStatusException {
68      return findValidTopology(topologyId, bucket, null, fullCheck, clientId, opId);
69    }
70  
71    public AccessorBucket getBucket(final String bucket, final String clientId, final Topology topology,
72                                    final String opId) throws CcsWithStatusException {
73      final var remoteReplicatorUri = URI.create(topology.uri());
74      try (final RemoteReplicatorApiClient client = remoteReplicatorApiClientFactory.newClient(remoteReplicatorUri)) {
75        return client.getBucket(bucket, clientId, opId);
76      }
77    }
78  
79    public Optional<Topology> findValidTopologyObject(final String bucket, final String objectName,
80                                                      final boolean fullCheck, final String clientId,
81                                                      final String topologyId, final String opId)
82        throws CcsWithStatusException {
83      return findValidTopology(topologyId, bucket, objectName, fullCheck, clientId, opId);
84    }
85  
86    private Optional<Topology> checkOneTopology(final Topology topology, final String bucket, final String objectName,
87                                                final boolean fullCheck, final String clientId, final String opId)
88        throws CcsWithStatusException {
89      final var remoteReplicatorUri = URI.create(topology.uri());
90      try (final RemoteReplicatorApiClient client = remoteReplicatorApiClientFactory.newClient(remoteReplicatorUri)) {
91        if (ParametersChecker.isNotEmpty(objectName)) {
92          LOGGER.debugf("Check object with remote replicator : %s", topology.uri());
93          final var storageType = client.checkObjectOrDirectoryCache(bucket, objectName, fullCheck, clientId, opId);
94          if (StorageType.OBJECT.equals(storageType)) {
95            // Store Topology Candidate
96            return Optional.of(topology);
97          }
98        } else {
99          LOGGER.debugf("Check bucket with remote replicator : %s", topology.uri());
100         final var storageType = client.checkBucketCache(bucket, fullCheck, clientId, opId);
101         if (StorageType.BUCKET.equals(storageType)) {
102           // Store Topology Candidate
103           return Optional.of(topology);
104         }
105       }
106     }
107     throw new CcsWithStatusException(topology, 404);
108   }
109 
110   private Optional<Topology> findValidTopology(final String topologyId, final String bucket, final String objectName,
111                                                final boolean fullCheck, final String clientId, final String opId)
112       throws CcsWithStatusException {
113     final Collection<Topology> topologies = getTopologies(topologyId);
114     for (final var topology : topologies) {
115       // ignores local replicator by checking configuration
116       if (!ServiceProperties.getAccessorSite().equals(topology.id())) {
117         try {
118           return checkOneTopology(topology, bucket, objectName, fullCheck, clientId, opId);
119         } catch (final CcsWithStatusException e) {
120           if (e.getStatus() != 404) {
121             throw e;
122           }
123         }
124       }
125     }
126     return Optional.empty();
127   }
