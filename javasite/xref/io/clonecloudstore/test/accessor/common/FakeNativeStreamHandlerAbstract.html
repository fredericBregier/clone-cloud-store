50  @Dependent
51  public abstract class FakeNativeStreamHandlerAbstract
52      extends NativeStreamHandlerAbstract<AccessorObject, AccessorObject> {
53    private static final Logger LOGGER = Logger.getLogger(FakeNativeStreamHandlerAbstract.class);
54    private DriverApi driverApi;
55    private AccessorObject checked;
56    private String clientId;
57  
58    @Override
59    protected void postSetup() {
60      internalInitBeforeAction();
61    }
62  
63    protected abstract boolean isPublic();
64  
65    /**
66     * Initialize information from Headers
67     */
68    protected void internalInitBeforeAction() {
69      setDriverApi();
70      getCloser().add(driverApi);
71      final var headers = getRequest().headers();
72      clientId = headers.get(X_CLIENT_ID);
73      final var currentBucketName = getBusinessIn().getBucket();
74      final var currentObjectName = getBusinessIn().getName();
75      AccessorHeaderDtoConverter.objectFromMap(getBusinessIn(), headers);
76      // Force for Replicator already having the right bucket name and object name, but not others while already computed
77      LOGGER.debugf("Previous bucket %s => %s", getBusinessIn().getBucket(), currentBucketName);
78      getBusinessIn().setBucket(currentBucketName);
79      if (ParametersChecker.isNotEmpty(currentObjectName)) {
80        getBusinessIn().setName(currentObjectName);
81      }
82    }
83  
84    private void setDriverApi() {
85      if (driverApi == null) {
86        driverApi = DriverApiRegistry.getDriverApiFactory().getInstance();
87      }
88    }

90    @Override
91    protected void checkPushAble(final AccessorObject object, final MultipleActionsInputStream nettyToInputStream) {
92      if (FakeObjectServiceAbstract.errorCode > 0) {
93        throw CcsServerGenericExceptionMapper.getCcsException(FakeObjectServiceAbstract.errorCode);
94      }
95      try {
96        final var objectStorage =
97            new StorageObject(object.getBucket(), object.getName(), object.getHash(), object.getSize(),
98                object.getCreation());
99        driverApi.objectPrepareCreateInBucket(objectStorage, nettyToInputStream);
100     } catch (final DriverNotFoundException e) {
101       throw new CcsNotExistException(e.getMessage(), e);
102     } catch (final DriverAlreadyExistException e) {
103       throw new CcsAlreadyExistException(e.getMessage(), e);
104     } catch (final DriverException e) {
105       throw new CcsOperationException(e.getMessage(), e);
106     }
107   }

116   @Override
117   protected AccessorObject getAnswerPushInputStream(final AccessorObject object, final String finalHash,
118                                                     final long size) {
119     // Hash from request
120     var hash = object.getHash();
121     if (finalHash != null) {
122       // Hash from NettyToInputStream (on the fly)
123       hash = finalHash;
124     }
125     try {
126       final var storageObject =
127           driverApi.objectFinalizeCreateInBucket(object.getBucket(), object.getName(), size, hash);
128       final var accessorObject = FakeObjectServiceAbstract.fromStorageObject(storageObject);
129       remoteCreation(accessorObject, clientId);
130       return accessorObject;
131     } catch (final DriverNotFoundException e) {
132       throw new CcsNotExistException(e.getMessage(), e);
133     } catch (final DriverAlreadyExistException e) {
134       throw new CcsAlreadyExistException(e.getMessage(), e);
135     } catch (final DriverException e) {
136       throw new CcsOperationException(e.getMessage(), e);
137     }
138   }

140   @Override
141   protected Map<String, String> getHeaderPushInputStream(final AccessorObject objectIn, final String finalHash,
142                                                          final long size, final AccessorObject objectOut) {
143     final Map<String, String> map = new HashMap<>();
144     AccessorHeaderDtoConverter.objectToMap(objectOut, map);
145     // Hash from request
146     if (ParametersChecker.isNotEmpty(finalHash)) {
147       // Hash from NettyToInputStream (on the fly)
148       map.put(X_OBJECT_HASH, finalHash);
149     }
150     if (size > 0) {
151       map.put(X_OBJECT_SIZE, Long.toString(size));
152     }
153     return map;
154   }

168   protected boolean internalCheckPullable(final AccessorObject object) {
169     LOGGER.infof("Check FakeApi %s/%s", object.getBucket(), object.getName());
170     if (FakeObjectServiceAbstract.errorCode > 0) {
171       if (FakeObjectServiceAbstract.errorCode == 404) {
172         return false;
173       }
174       throw CcsServerGenericExceptionMapper.getCcsException(FakeObjectServiceAbstract.errorCode);
175     }
176     try {
177       final var storageObject = driverApi.objectGetMetadataInBucket(object.getBucket(), object.getName());
178       if (storageObject != null) {
179         checked = FakeObjectServiceAbstract.fromStorageObject(storageObject);
180         LOGGER.infof("Checked FakeApi %s/%s", object.getBucket(), object.getName());
181         return true;
182       }
183       if (isPublic() && checkRemotePullable(object, getRequest().headers(), clientId)) {
184         checked = object.setStatus(AccessorStatus.READY).cloneInstance().setHash("hash").setCreation(Instant.now())
185             .setId(GuidLike.getGuid()).setSize(FakeObjectServiceAbstract.length);
186       } else {
187         checked = null;
188       }
189       LOGGER.infof("Checked failed FakeApi %s/%s", object.getBucket(), object.getName());
190       return false;
191     } catch (final DriverNotFoundException e) {
192       if (isPublic()) {
193         return checkRemotePullable(object, getRequest().headers(), clientId);
194       }
195       LOGGER.infof("Checked failed FakeApi %s/%s", object.getBucket(), object.getName());
196       return false;
197     } catch (final DriverException e) {
198       throw new CcsOperationException(e.getMessage(), e);
199     }
200   }

206     throw new CcsNotExistException(object.getName());
207   }

209   @Override
210   protected InputStream getPullInputStream(final AccessorObject object) {
211     try {
212       if (FakeObjectServiceAbstract.errorCode > 0) {
213         if (FakeObjectServiceAbstract.errorCode >= 400) {
214           throw CcsServerGenericExceptionMapper.getCcsException(FakeObjectServiceAbstract.errorCode);
215         } else {
216           return new FakeInputStream(100);
217         }
218       }
219       if (!internalCheckPullable(object)) {
220         LOGGER.infof("Pull FakeApi %s/%s", object.getBucket(), object.getName());
221       }
222       return driverApi.objectGetInputStreamInBucket(object.getBucket(), object.getName());
223     } catch (final DriverNotFoundException e) {
224       LOGGER.infof("Pull failed but try remote FakeApi %s/%s", object.getBucket(), object.getName());
225       if (isPublic()) {
226         // Else use remote read
227         return getRemotePullInputStream(object, clientId);
228       }
229       throw new CcsNotExistException(e.getMessage(), e);
230     } catch (final DriverException e) {
231       throw new CcsOperationException(e.getMessage(), e);
232     }
233   }
