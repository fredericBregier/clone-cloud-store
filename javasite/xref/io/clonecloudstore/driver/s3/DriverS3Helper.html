99    S3Client getClient() throws DriverRuntimeException {
100     try {
101       LOGGER.debugf("Charge configuration S3: %s %s", getDriverS3Host(), getDriverS3Region());
102       return S3Client.builder().endpointOverride(new URI(getDriverS3Host())).credentialsProvider(
103               StaticCredentialsProvider.create(AwsBasicCredentials.create(getDriverS3KeyId(), getDriverS3Key())))
104           .region(Region.of(getDriverS3Region())).build();

110   StorageBucket createS3Bucket(final S3Client s3Client, final StorageBucket bucket)
111       throws DriverAlreadyExistException, DriverNotAcceptableException, DriverException { // NOSONAR Exception details
112     try {
113       // Check if Bucket already exists
114       if (!existS3Bucket(s3Client, bucket.bucket())) {
115         final var bucketRequest = CreateBucketRequest.builder().bucket(bucket.bucket()).build();
116         internalCreateBucket(s3Client, bucket, bucketRequest);
117         final var bucketRequestWait = HeadBucketRequest.builder().bucket(bucket.bucket()).build();
118         // Wait until the bucket is created
119         final var s3Waiter = s3Client.waiter();
120         final var waiterResponse = s3Waiter.waitUntilBucketExists(bucketRequestWait);
121         final var responsed = waiterResponse.matched().response();
122         if (responsed.isEmpty() || !responsed.get().sdkHttpResponse().isSuccessful()) {
123           // KO
124           throw new DriverNotAcceptableException("Cannot create Bucket: " + bucket);
125         }
126         final var response = getS3Buckets(s3Client);
127         final var buckets = response.buckets();
128         for (final var bucket1 : buckets) {
129           if (bucket1.name().equals(bucket.bucket())) {
130             return fromBucket(bucket1);
131           }
132         }
133         throw new DriverException("Cannot find created Bucket: " + bucket);
134       } else {
135         throw new DriverAlreadyExistException(BUCKET_ALREADY_EXISTS + bucket);
136       }
137     } catch (final RuntimeException e) {
138       throw new DriverException(e);
139     }
140   }

142   private static void internalCreateBucket(final S3Client s3Client, final StorageBucket bucket,
143                                            final CreateBucketRequest bucketRequest)
144       throws DriverNotAcceptableException, DriverAlreadyExistException {
145     try {
146       final var response = s3Client.createBucket(bucketRequest);
147       if (!response.sdkHttpResponse().isSuccessful()) {
148         throw new DriverNotAcceptableException(
149             "Bucket cannot be created, code: " + response.sdkHttpResponse().statusCode() + FOR + bucket);
150       }
151     } catch (final BucketAlreadyExistsException ignored) {
152       throw new DriverAlreadyExistException(BUCKET_ALREADY_EXISTS + bucket);
153     }
154   }

157     final var bucketRequestExist = HeadBucketRequest.builder().bucket(bucket).build();
158     try {
159       return s3Client.headBucket(bucketRequestExist).sdkHttpResponse().isSuccessful();
160     } catch (final NoSuchBucketException ignored) {
161       return false;
162     } catch (final RuntimeException e) {
163       if (e instanceof S3Exception se && (se.statusCode() == 400)) {
164         return false;
165       }
166       throw new DriverException(e);
167     }

185   void deleteS3Bucket(final S3Client s3Client, final String bucket)
186       throws DriverException, DriverNotFoundException, DriverNotAcceptableException { // NOSONAR Exception details
187     if (!existS3Bucket(s3Client, bucket)) {
188       // Not found
189       throw new DriverNotFoundException(BUCKET_DOES_NOT_EXIST + bucket);
190     }
191     var status = Response.Status.NOT_ACCEPTABLE.getStatusCode();
192     try {
193       final var response = s3Client.deleteBucket(DeleteBucketRequest.builder().bucket(bucket).build());
194       status = response.sdkHttpResponse().statusCode();
195       if (!response.sdkHttpResponse().isSuccessful()) {
196         // KO
197         throw new DriverNotAcceptableException(
198             "Cannot delete Bucket, code: " + response.sdkHttpResponse().statusCode() + " for: " + bucket);
199       }
200     } catch (final NoSuchBucketException e) {
201       throw new DriverNotFoundException(BUCKET_DOES_NOT_EXIST + bucket, e);
202     } catch (final SdkException e) {
203       throw new DriverNotAcceptableException("Cannot delete Bucket, code: " + status + " for: " + bucket);
204     } catch (final RuntimeException e) {
205       throw new DriverException(e);
206     }
207   }

215   /**
216    * Gets S3 objects that reside in a specific bucket and whose keys conform to the
217    * specified prefix using v2 of the AWS Java SDK.
218    * <br><br>
219    * The objects returned will have a last-modified date between {@code start} and
220    * {@code end}.
221    * <br><br>
222    * Any objects that have been modified outside of the specified date-time range will
223    * not be returned.
224    *
225    * @param s3Client The v2 AWS S3 client used to make the request to S3.
226    * @param bucket   The bucket where the S3 objects are located.
227    * @param prefix   The common prefix that the keys of the S3 objects must conform to.
228    * @param start    The objects returned will have been modified after this instant.
229    * @param end      The objects returned will have been modified before this instant.
230    * @return A {@link Stream} of {@link S3Object} objects.
231    */
232   Stream<S3Object> getS3ObjectsStreamFilteredInBucket(final S3Client s3Client, final String bucket, final String prefix,
233                                                       final Instant start, final Instant end)
234       throws DriverException, DriverNotFoundException { // NOSONAR Exception details
235     final ListObjectsV2Request request = getListObjectsV2Request(s3Client, bucket, prefix);
236     try {
237       if (start != null || end != null) {
238         return s3Client.listObjectsV2Paginator(request).contents().stream().filter(s3Object -> {
239           final var lastModified = s3Object.lastModified();
240           if (start != null && start.isAfter(lastModified)) {
241             return false;
242           }
243           if (end != null) {
244             return !end.isBefore(lastModified);
245           }
246           return true;
247         });
248       }
249       return s3Client.listObjectsV2Paginator(request).contents().stream();
250     } catch (final NoSuchBucketException e) {
251       throw new DriverNotFoundException(BUCKET_DOES_NOT_EXIST + bucket, e);
252     } catch (final RuntimeException e) {
253       throw new DriverException(e);
254     }
255   }

257   private ListObjectsV2Request getListObjectsV2Request(final S3Client s3Client, final String bucket,
258                                                        final String prefix) throws DriverException {
259     if (!existS3Bucket(s3Client, bucket)) {
260       // Not found
261       throw new DriverNotFoundException(BUCKET_DOES_NOT_EXIST + bucket);
262     }
263     final ListObjectsV2Request request;
264     // If start or end is set, LIMIT must not be set
265     if (ParametersChecker.isNotEmpty(prefix)) {
266       request = ListObjectsV2Request.builder().bucket(bucket).prefix(prefix).maxKeys(MAX_ITEMS).build();
267     } else {
268       request = ListObjectsV2Request.builder().bucket(bucket).maxKeys(MAX_ITEMS).build();
269     }
270     return request;
271   }

312   StorageObject getS3ObjectInBucket(final S3Client s3Client, final String bucket, final String s3name)
313       throws DriverException, DriverNotFoundException { // NOSONAR Exception details
314     final var objectRequestExist =
315         HeadObjectRequest.builder().bucket(bucket).key(s3name).checksumMode(ChecksumMode.ENABLED).build();
316     var status = 500;
317     try {
318       final var response = s3Client.headObject(objectRequestExist);
319       if (response.sdkHttpResponse().isSuccessful()) {
320         return fromS3Head(s3Client, bucket, s3name, response);
321       }
322       status = response.sdkHttpResponse().statusCode();
323     } catch (final NoSuchBucketException e) {
324       throw new DriverNotFoundException(BUCKET_DOES_NOT_EXIST + bucket, e);
325     } catch (final NoSuchKeyException e) {
326       throw new DriverNotFoundException(OBJECT_DOES_NOT_EXIST + bucket + ":" + s3name, e);
327     } catch (final RuntimeException e) {
328       throw new DriverException(e);
329     }
330     // Not found
331     throw new DriverNotFoundException(OBJECT_DOES_NOT_EXIST + ", code: " + status + FOR + bucket + ":" + s3name);
332   }

334   StorageObject fromS3Head(final S3Client s3Client, final String bucket, final String name,
335                            final HeadObjectResponse response) throws DriverException {
336     LOGGER.debugf("DEBUG %s/s %s %d %s %s", bucket, name, response.checksumSHA256(), response.contentLength(),
337         response.lastModified(), response.metadata());
338     final String sha;
339     try {
340       final var map = new HashMap<>(response.metadata());
341       if (map.containsKey(SHA_256)) {
342         sha = map.remove(SHA_256);
343       } else {
344         final var getObjectTaggingResponse =
345             s3Client.getObjectTagging(GetObjectTaggingRequest.builder().bucket(bucket).key(name).build());
346         if (!getObjectTaggingResponse.sdkHttpResponse().isSuccessful()) {
347           sha = null;
348         } else {
349           final var tag = getObjectTaggingResponse.tagSet().stream().filter(t -> t.key().equals(SHA_256)).findFirst();
350           sha = tag.map(Tag::value).orElse(null);
351         }
352       }
353       return new StorageObject(bucket, name, sha, response.contentLength(), response.lastModified(), response.expires(),
354           map);
355     } catch (final NoSuchBucketException | NoSuchKeyException e) {
356       throw new DriverNotFoundException(e);
357     } catch (final RuntimeException e) {
358       throw new DriverException(e);
359     }
360   }

362   InputStream getS3ObjectBodyInBucket(final S3Client s3Client, final String bucket, final String s3name,
363                                       final boolean checkExistence)
364       throws DriverNotFoundException, DriverException { // NOSONAR Exception details
365     if (checkExistence && !existS3DirectoryOrObjectInBucket(s3Client, bucket, s3name).equals(StorageType.OBJECT)) {
366       // Not found
367       if (!existS3Bucket(s3Client, bucket)) {
368         throw new DriverNotFoundException(BUCKET_DOES_NOT_EXIST + bucket);
369       }
370       throw new DriverNotFoundException(OBJECT_DOES_NOT_EXIST + bucket + ":" + s3name);
371     }
372     try {
373       return s3Client.getObject(GetObjectRequest.builder().bucket(bucket).key(s3name).build());
374     } catch (final NoSuchKeyException | NoSuchBucketException e) {
375       throw new DriverNotFoundException(e);
376     } catch (final RuntimeException e) {
377       throw new DriverException(e);
378     }
379   }

381   StorageType existS3DirectoryOrObjectInBucket(final S3Client s3Client, final String bucket,
382                                                final String directoryOrObject) throws DriverException {
383     try {
384       // First try for an Object then for Directory
385       if (existS3ObjectInBucket(s3Client, bucket, directoryOrObject)) {
386         return StorageType.OBJECT;
387       }
388       final var iterator = getS3ObjectsIteratorFilteredInBucket(s3Client, bucket, directoryOrObject, null, null);
389       var storageType = iterator.hasNext() ? StorageType.DIRECTORY : StorageType.NONE;
390       SystemTools.consumeAll(iterator);
391       return storageType;
392     } catch (final DriverNotFoundException | NoSuchBucketException | NoSuchKeyException e) {
393       return StorageType.NONE;
394     } catch (final RuntimeException e) {
395       LOGGER.error(e.getMessage());
396       throw new DriverException(e.getMessage(), e);
397     }
398   }

431   StorageObject waitUntilObjectExist(final S3Client s3Client, final String bucket, final String s3name,
432                                      final String sha256ForTag)
433       throws DriverNotAcceptableException, DriverException { // NOSONAR Exception details
434     final var objectRequestWait =
435         HeadObjectRequest.builder().bucket(bucket).key(s3name).checksumMode(ChecksumMode.ENABLED).build();
436     WaiterResponse<HeadObjectResponse> waiter = null;
437     var status = 0;
438     try {
439       // Wait until the object is created
440       waiter = s3Client.waiter().waitUntilObjectExists(objectRequestWait);
441       var responsed = waiter.matched().response();
442       if (responsed.isPresent()) {
443         final var response = responsed.get();
444         status = response.sdkHttpResponse().statusCode();
445         if (response.sdkHttpResponse().isSuccessful()) {
446           if (ParametersChecker.isNotEmpty(sha256ForTag)) {
447             // Set the new Sha256 as Tag since object already created
448             // Maybe use the copy to rewrite metadata instead of Tag
449             setShaAsTagForObject(s3Client, bucket, s3name, sha256ForTag);
450           }
451           return fromS3Head(s3Client, bucket, s3name, response);
452         }
453       }
454     } catch (final NoSuchKeyException | NoSuchBucketException e) {
455       throw new DriverNotFoundException(e);
456     } catch (final RuntimeException e) {
457       throw new DriverException(e);
458     }
459     var exceptioned = waiter.matched().exception();
460     exceptioned.ifPresent(LOGGER::error);
461     throw new DriverNotAcceptableException("Object cannot be created, code: " + status + FOR + bucket + ":" + s3name);
462   }

464   private static void setShaAsTagForObject(final S3Client s3Client, final String bucket, final String s3name,
465                                            final String sha256ForTag) {
466     try {
467       final var putObjectTaggingResponse = s3Client.putObjectTagging(
468           PutObjectTaggingRequest.builder().bucket(bucket).key(s3name)
469               .tagging(t -> t.tagSet(b -> b.key(SHA_256).value(sha256ForTag))).build());
470       if (!putObjectTaggingResponse.sdkHttpResponse().isSuccessful()) {
471         LOGGER.warn("Cannot set tag");
472       }
473     } catch (final RuntimeException e) {
474       LOGGER.warn("Cannot tag object: " + s3name, e);
475     }
476   }

478   void createS3ObjectInBucketNoCheck(final S3Client s3Client, final StorageObject object, final InputStream inputStream)
479       throws DriverException, DriverNotAcceptableException { // NOSONAR Exception details
480     final RequestBody requestBody;
481     try {
482       if (object.size() > 0) {
483         requestBody = RequestBody.fromInputStream(inputStream, object.size());
484       } else {
485         LOGGER.warn("Unknown length shall not be OK with upload");
486         final ContentStreamProvider provider = () -> ReleasableInputStream.wrap(inputStream).disableClose();
487         requestBody = RequestBody.fromContentProvider(provider, MediaType.APPLICATION_OCTET_STREAM);
488       }
489       final var builder = PutObjectRequest.builder().bucket(object.bucket()).key(object.name());
490       if (object.expiresDate() != null) {
491         builder.expires(object.expiresDate());
492       }
493       final Map<String, String> map = HashMap.newHashMap(1);
494       if (object.metadata() != null) {
495         map.putAll(object.metadata());
496       }
497       if (ParametersChecker.isNotEmpty(object.hash())) {
498         map.put(SHA_256, object.hash());
499       }
500       LOGGER.debugf("Metadata %s", map);
501       if (!map.isEmpty()) {
502         builder.metadata(map);
503       }
504       final var response = s3Client.putObject(builder.build(), requestBody);
505       if (!response.sdkHttpResponse().isSuccessful()) {
506         throw new DriverNotAcceptableException(
507             "Object cannot be created, code: " + response.sdkHttpResponse().statusCode() + FOR + object.bucket() + ":" +
508                 object.name());
509       }
510       LOGGER.debugf("MAI %s", inputStream);
511     } catch (final NoSuchKeyException | NoSuchBucketException e) {
512       throw new DriverNotFoundException(e);
513     } catch (final RuntimeException e) {
514       throw new DriverException(e);
515     }

518   private static class S3ObjectIterator implements Iterator<S3Object> {
519     private final Iterator<S3Object> iterator;
520     private final Instant start;
521     private final Instant end;
522     private S3Object s3Object;
523 
524     public S3ObjectIterator(final Iterator<S3Object> iterator, final Instant start, final Instant end) {
525       this.iterator = iterator;
526       this.start = start;
527       this.end = end;
528       s3Object = null;
529     }
530 
531     private S3Object nextInternal() {
532       while (iterator.hasNext()) {
533         final var item = iterator.next();
534         final var lastModified = item.lastModified();
535         if ((start != null && start.isAfter(lastModified)) || (end != null && end.isBefore(lastModified))) {
536           continue;
537         }
538         return item;
539       }
540       return null;
541     }
542 
543     @Override
544     public boolean hasNext() {
545       if (s3Object == null) {
546         s3Object = nextInternal();
547       }
