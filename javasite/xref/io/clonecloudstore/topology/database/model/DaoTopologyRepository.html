17  package io.clonecloudstore.topology.database.model;
18  
19  import java.util.List;
20  
21  import io.clonecloudstore.common.database.utils.DbQuery;
22  import io.clonecloudstore.common.database.utils.DbUpdate;
23  import io.clonecloudstore.common.database.utils.RepositoryBaseInterface;
24  import io.clonecloudstore.common.database.utils.RestQuery;
25  import io.clonecloudstore.common.database.utils.exception.CcsDbException;
26  import io.clonecloudstore.common.quarkus.exception.CcsOperationException;
27  import io.clonecloudstore.topology.model.Topology;
28  import io.clonecloudstore.topology.model.TopologyStatus;
29  
30  public interface DaoTopologyRepository extends RepositoryBaseInterface<DaoTopology> {
31    String TABLE_NAME = "topology";
32    String URI = "uri";
33    String NAME = "name";
34    String STATUS = "status";
35  
36    default Topology insertTopology(final Topology topology) throws CcsDbException {
37      try {
38        final var daoTopology = this.createEmptyItem().fromDto(topology);
39        this.insert(daoTopology);
40        return daoTopology.getDto();
41      } catch (final RuntimeException e) {
42        throw new CcsDbException(e);
43      }
44    }
45  
46    default Topology findBySite(final String site) throws CcsDbException {
47      try {
48        final var dao = this.findWithPk(site);
49        return (dao != null ? dao.getDto() : null);
50      } catch (final RuntimeException e) {
51        throw new CcsDbException(e);
52      }
53    }
54  
55    default List<Topology> findAllTopologies() throws CcsDbException {
56      try {
57        return findStream(new DbQuery()).map(DaoTopology::getDto).toList();
58      } catch (final RuntimeException e) {
59        throw new CcsDbException(e);
60      }
61    }
62  
63    default List<Topology> findTopologies(final TopologyStatus status) throws CcsDbException {
64      try {
65        return findStream(new DbQuery(RestQuery.QUERY.EQ, STATUS, status)).map(DaoTopology::getDto).toList();
66      } catch (final RuntimeException e) {
67        throw new CcsDbException(e);
68      }
69    }
70  
71    default Topology updateTopologyStatus(final Topology topology, final TopologyStatus status) throws CcsDbException {
72      try {
73        final var query = new DbQuery(RestQuery.QUERY.EQ, getPkName(), topology.id());
74        final var dbUpdate = new DbUpdate().set(STATUS, status);
75  
76        if (this.update(query, dbUpdate) != 1) {
77          throw new CcsDbException(new CcsOperationException("Could not update topology"));
78        }
79        return new Topology(topology, status);
80      } catch (final RuntimeException e) {
81        throw new CcsDbException(e);
82      }
83    }
84  
85    default Topology updateTopology(final Topology topology) throws CcsDbException {
86      try {
87        final var query = new DbQuery(RestQuery.QUERY.EQ, getPkName(), topology.id());
88        final var dbUpdate =
89            new DbUpdate().set(STATUS, topology.status()).set(NAME, topology.name()).set(URI, topology.uri());
90  
91        if (this.update(query, dbUpdate) != 1) {
92          throw new CcsDbException(new CcsOperationException("Could not update topology"));
93        }
94        return topology;
95      } catch (final RuntimeException e) {
96        throw new CcsDbException(e);
97      }
98    }
99  
100   default boolean deleteTopology(final String site) throws CcsDbException {
101     try {
102       final var query = new DbQuery(RestQuery.QUERY.EQ, getPkName(), site);
103       if (this.delete(query) != 1) {
104         throw new CcsDbException(new CcsOperationException("Could not delete topology"));
105       } else {
106         return true;
107       }
108     } catch (final RuntimeException e) {
109       throw new CcsDbException(e);
110     }
111   }
112 }
