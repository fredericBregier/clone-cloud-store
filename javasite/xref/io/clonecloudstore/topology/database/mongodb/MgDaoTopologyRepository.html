17  package io.clonecloudstore.topology.database.mongodb;
18  
19  import com.mongodb.MongoException;
20  import com.mongodb.client.model.IndexOptions;
21  import com.mongodb.client.model.Indexes;
22  import io.clonecloudstore.common.database.mongo.ExtendedPanacheMongoRepositoryBase;
23  import io.clonecloudstore.common.database.utils.exception.CcsDbException;
24  import io.clonecloudstore.topology.database.model.DaoTopology;
25  import io.clonecloudstore.topology.database.model.DaoTopologyRepository;
26  import io.quarkus.arc.lookup.LookupIfProperty;
27  import jakarta.enterprise.context.ApplicationScoped;
28  
29  import static io.clonecloudstore.common.database.utils.DbType.CCS_DB_TYPE;
30  import static io.clonecloudstore.common.database.utils.DbType.MONGO;
31  
32  @LookupIfProperty(name = CCS_DB_TYPE, stringValue = MONGO)
33  @ApplicationScoped
34  public class MgDaoTopologyRepository extends ExtendedPanacheMongoRepositoryBase<DaoTopology, MgDaoTopology>
35      implements DaoTopologyRepository {
36  
37    @Override
38    public String getTable() {
39      return TABLE_NAME;
40    }
41  
42  
43    @Override
44    public DaoTopology createEmptyItem() {
45      return new MgDaoTopology();
46    }
47  
48    public void createIndex() throws CcsDbException {
49      try {
50        mongoCollection().createIndex(Indexes.ascending(ID, NAME, STATUS),
51            new IndexOptions().name(TABLE_NAME + "_filter_idx"));
52      } catch (final MongoException e) {
53        throw new CcsDbException("Cannot Create Index", e);
54      }
55    }
56  }
