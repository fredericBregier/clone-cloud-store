17  package io.clonecloudstore.common.standard.properties;
18  
19  import java.util.concurrent.ExecutorService;
20  import java.util.concurrent.Executors;
21  import java.util.concurrent.ThreadFactory;
22  
23  import com.fasterxml.jackson.core.JsonParser;
24  import com.fasterxml.jackson.databind.DeserializationFeature;
25  import com.fasterxml.jackson.databind.ObjectMapper;
26  import com.fasterxml.jackson.databind.SerializationFeature;
27  import com.fasterxml.jackson.datatype.jsr310.JavaTimeModule;
28  import io.clonecloudstore.common.standard.exception.CcsInvalidArgumentRuntimeException;
29  import io.clonecloudstore.common.standard.system.DaemonThreadFactory;
30  import io.clonecloudstore.common.standard.system.SystemPropertyUtil;
31  import io.vertx.core.Vertx;
32  
33  public class StandardProperties {
34    /**
35     * Compatible with 64KB, 96 KB and 128 KB buffer sizes (Default 384 KB)
36     */
37    public static final int DEFAULT_PIPED_BUFFER_SIZE = 384 * 1024;
38    /**
39     * Internal Machine Id used by JvmProcessMacIds for MacAddress (null or empty meaning using default value)
40     */
41    public static final String CCS_MACHINE_ID = "ccs.machineId";
42    /**
43     * Property to define Buffer Size
44     */
45    public static final String CCS_BUFFER_SIZE = "ccs.bufferSize";
46    /**
47     * Property to define Max waiting time in milliseconds before Time Out within packets (in particular unknown
48     * size)
49     */
50    public static final String CCS_MAX_WAIT_MS = "ccs.maxWaitMs";
51    private static final long DEFAULT_MAX_WAIT_MS = 1000;
52    /**
53     * Optimal is between 64KB, 96KB and 128KB.
54     * Note: Quarkus seems to limit to 64KB but setting the same value gives smaller chunk size
55     */
56    private static final int DEFAULT_BUFFER_SIZE = 131072;
57    private static int bufSize = SystemPropertyUtil.get(CCS_BUFFER_SIZE, DEFAULT_BUFFER_SIZE);
58    private static long maxWaitMs = SystemPropertyUtil.get(CCS_MAX_WAIT_MS, DEFAULT_MAX_WAIT_MS);
59    /**
60     * Global Thread pool for Internal actions such as in TeeInputStream, ZstdCompressedInputStream, S3
61     */
62    public static final ExecutorService VIRTUAL_EXECUTOR_SERVICE = Executors.newVirtualThreadPerTaskExecutor();
63    public static final ThreadFactory DAEMON_THREAD_FACTORY = new DaemonThreadFactory();
64    public static final ExecutorService STANDARD_EXECUTOR_SERVICE = Executors.newCachedThreadPool(DAEMON_THREAD_FACTORY);
65    /**
66     * Default global Vertx
67     */
68    private static Vertx vertx = null;
69    /**
70     * Default global ObjectMapper
71     */
72    private static ObjectMapper objectMapper = null;
73  
74    /**
75     * @return the MachineId if specified as ccs.machineId using 6 bytes in Hexadecimal format
76     */
77    public static String getCcsMachineId() {
78      return SystemPropertyUtil.get(CCS_MACHINE_ID);
79    }
80  
81    /**
82     * @return the maximum Buffer Size (default 128 KB)
83     */
84    public static int getBufSize() {
85      return bufSize;
86    }
87  
88    public static void setBufSize(final int bufSize) {
89      SystemPropertyUtil.set(CCS_BUFFER_SIZE, bufSize);
90      StandardProperties.bufSize = bufSize;
91    }
92  
93    /**
94     * @return Max Wait in milliseconds before TimeOut occurs within transmission (end of unknown size) (Default 1000 ms)
95     */
96    public static long getMaxWaitMs() {
97      return maxWaitMs;
98    }
99  
100   public static void setMaxWaitMs(final long maxWaitMs) {
101     SystemPropertyUtil.set(CCS_MAX_WAIT_MS, maxWaitMs);
102     StandardProperties.maxWaitMs = maxWaitMs;
103   }
104 
105   /**
106    * @return the global Vertx core
107    */
108   public static Vertx getVertx() {
109     if (vertx == null) {
110       vertx = Vertx.vertx();
111     }
112     return vertx;
113   }
114 
115   /**
116    * To set up from CDI as Quarkus
117    */
118   public static void setCdiVertx(final Vertx vertx) {
119     if (vertx != null && StandardProperties.vertx != null) {
120       StandardProperties.vertx.close();
121     }
122     StandardProperties.vertx = vertx;
123   }
124 
125   /**
126    * @return the global ObjectMapper
127    */
128   public static ObjectMapper getObjectMapper() {
129     if (objectMapper == null) {
130       objectMapper = new ObjectMapper().disable(SerializationFeature.FAIL_ON_EMPTY_BEANS)
131           .configure(JsonParser.Feature.ALLOW_SINGLE_QUOTES, true)
132           .configure(JsonParser.Feature.ALLOW_UNQUOTED_FIELD_NAMES, true)
133           .configure(DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES, false).registerModule(new JavaTimeModule())
134           .configure(SerializationFeature.WRITE_DATES_AS_TIMESTAMPS, false);
135     }
136     return objectMapper;
137   }
138 
139   /**
140    * To set up from CDI as Quarkus
141    */
142   public static void setCdiObjectMapper(final ObjectMapper objectMapper) {
143     if (objectMapper == null) {
144       throw new CcsInvalidArgumentRuntimeException("ObjectMapper cannot be null");
145     }
146     StandardProperties.objectMapper = objectMapper;
147   }
148 
149   protected StandardProperties() {
150     // Empty
151   }
152 
153   public static String confugrationToString() {
154     return String.format("\"%s\":\"%s\", \"%s\":%d, \"%s\":%d", CCS_MACHINE_ID, getCcsMachineId(), CCS_BUFFER_SIZE,
155         getBufSize(), CCS_MAX_WAIT_MS, getMaxWaitMs());
156   }
157 }
