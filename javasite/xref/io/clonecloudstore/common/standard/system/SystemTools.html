40    /**
41     * @param instant instant to truncate
42     * @return an Instant truncated to Millisecond (since most of the DB truncated it)

44    public static Instant toMillis(final Instant instant) {
45      if (instant != null) {
46        return instant.truncatedTo(ChronoUnit.MILLIS);

93    /**
94     * @param iterator to consume fully
95     * @return the number of items consumed
96     */
97    public static long consumeAll(final Iterator<?> iterator) {
98      long cpt = 0;
99      if (iterator != null) {
100       while (iterator.hasNext()) {
101         iterator.next();
102         cpt++;
103       }
104     }
105     return cpt;
106   }
107 
108   /**
109    * @param stream to consume fully
110    * @return the number of items consumed
111    */
112   public static long consumeAll(final Stream<?> stream) {
113     long cpt = 0;
114     if (stream != null) {
115       cpt = stream.count();
116     }
117     return cpt;
118   }

120   /**
121    * Get the field from the object named fieldName from Class
122    *
123    * @throws NoSuchFieldException   field does not exist
124    * @throws IllegalAccessException issue on access
125    */
126   public static Object getField(final Class<?> clasz, final String fieldName, final Object object)
127       throws NoSuchFieldException, IllegalAccessException {
128     try {
129       final var field = clasz.getDeclaredField(fieldName);
130       field.setAccessible(true);
131       return field.get(object);
132     } catch (final NoSuchFieldException e) {
133       final var superClass = clasz.getSuperclass();
134       if (superClass == null) {
135         throw e;
136       } else {
137         return getField(superClass, fieldName, object);
138       }
139     }
140   }

153   /**
154    * Consume all InputStream in error case with no error
155    */
156   public static void consumeWhileErrorInputStream(final InputStream inputStream) {
157     if (inputStream == null) {
158       return;
159     }
160     MultipleActionsInputStream mi = null;
161     if (inputStream instanceof MultipleActionsInputStream mi1) {
162       mi = mi1;
163       mi.changeExceptionDuringCount(Boolean.FALSE);
164     }
165     try {
166       long currentTime = System.currentTimeMillis();
167       long len;
168       final var bytes = new byte[StandardProperties.getBufSize()];
169       do {
170         long newTime = System.currentTimeMillis();
171         if (newTime - currentTime > StandardProperties.getMaxWaitMs()) {
172           break;
173         }
174         currentTime = newTime;
175         try {
176           len = inputStream.read(bytes, 0, bytes.length);
177         } catch (final IOException ignore) {
178           return;
179         }
180         Thread.yield();
181       } while (len >= 0);
182     } finally {
183       silentlyCloseNoException(inputStream);
184       if (mi != null) {
185         mi.changeExceptionDuringCount(Boolean.TRUE);
186       }
187     }
188   }
189 
190   /**
191    * Consume all InputStream in error case with no error with a timeout
192    */
193   public static void consumeWhileErrorInputStream(final InputStream inputStream, long timeout) {
194     try {
195       final var wait = STANDARD_EXECUTOR_SERVICE.submit(() -> consumeWhileErrorInputStream(inputStream));
196       Thread.yield();
197       wait.get(timeout, TimeUnit.MILLISECONDS);
198     } catch (final InterruptedException e) {
199       Thread.currentThread().interrupt();
200     } catch (final ExecutionException | TimeoutException ignore) {
201       // ignore
202     }
203   }

214   /**
215    * Transfer all inputStream to outputStream
216    *
217    * @return length of transferred bytes
218    */
219   public static long transferTo(final InputStream inputStream, final OutputStream outputStream,
220                                 final boolean flushOnChunk) throws IOException {
221     ParametersChecker.checkParameter("OutputStream and InputStream cannot be null", outputStream, inputStream);
222     var transferred = 0L;
223     var read = 0;
224     final var buf = new byte[StandardProperties.getBufSize()];
225     while ((read = inputStream.read(buf, 0, buf.length)) >= 0) {
226       outputStream.write(buf, 0, read);
227       transferred += read;
228       if (flushOnChunk) {
229         outputStream.flush();
230       }
231       Thread.yield();
232     }
233     outputStream.flush();
234     inputStream.close();
235     return transferred;
236   }

238   /**
239    * Skip bytes from InputStream
240    *
241    * @return the number of really skipped bytes
242    */
243   public static long skip(final InputStream inputStream, final long skip) throws IOException {
244     final var bytes = new byte[StandardProperties.getBufSize()];
245     var still = skip;
246     var total = 0L;
247     var max = (int) Math.min(StandardProperties.getBufSize(), still);
248     var read = 0;
249     while (max > 0 && (read = inputStream.read(bytes, 0, max)) >= 0) {
250       still -= read;
251       total += read;
252       max = (int) Math.min(StandardProperties.getBufSize(), still);
253       if (max <= 0) {
254         return total;
255       }
256     }
257     if (read < 0) {
258       inputStream.close();
259     }
260     return total;
261   }
