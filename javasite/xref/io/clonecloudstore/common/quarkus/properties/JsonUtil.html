17  package io.clonecloudstore.common.quarkus.properties;
18  
19  import com.fasterxml.jackson.annotation.JsonInclude;
20  import com.fasterxml.jackson.databind.ObjectMapper;
21  import io.clonecloudstore.common.standard.properties.StandardProperties;
22  import io.quarkus.arc.Arc;
23  import io.quarkus.arc.Unremovable;
24  import io.quarkus.jackson.ObjectMapperCustomizer;
25  import io.quarkus.runtime.Startup;
26  import io.quarkus.runtime.StartupEvent;
27  import jakarta.enterprise.context.ApplicationScoped;
28  import jakarta.enterprise.event.Observes;
29  import jakarta.inject.Inject;
30  import jakarta.inject.Singleton;
31  import org.jboss.logging.Logger;
32  
33  /**
34   * Utility class for Json to get the Jackson ObjectMapper, either from CDI of Quarkus, or from
35   * scratch if not in CDI (but should, in that case a Warning will be raised)
36   */
37  @ApplicationScoped
38  @Startup
39  @Unremovable
40  public final class JsonUtil {
41    private static final Logger LOGGER = Logger.getLogger(JsonUtil.class);
42    @Unremovable
43    private static final JsonUtil jsonHandler = new JsonUtil();
44    @Unremovable
45    private ObjectMapper mapper;
46    @Unremovable
47    private ObjectMapper mapperNoNull;
48  
49    JsonUtil() {
50      startup(null);
51    }
52  
53    @Inject
54    JsonUtil(final ObjectMapper objectMapper) {
55      this.mapper = objectMapper;
56      setupCcsProperties();
57    }
58  
59    void startup(@Observes final StartupEvent event) { // NOSONAR intentional
60      ObjectMapper objectMapper = null;
61      if (Arc.container() != null) {
62        objectMapper = Arc.container().instance(ObjectMapper.class).get();
63      }
64      if (objectMapper == null) {
65        LOGGER.warn("Cannot initialize ObjectMapper using Default Arc instance");
66        objectMapper = StandardProperties.getObjectMapper();
67      }
68      mapper = objectMapper;
69      setupCcsProperties();
70    }
71  
72    private void setupCcsProperties() {
73      mapperNoNull = mapper.copy().setSerializationInclusion(JsonInclude.Include.NON_NULL);
74      StandardProperties.setCdiObjectMapper(mapper);
75    }
76  
77    /**
78     * @return an ObjectMapper, either the one in the Quarkus CDI or a from scratch if none exists yet
79     */
80    @Unremovable
81    public static ObjectMapper getInstance() {
82      return jsonHandler.mapper;
83    }
84  
85    /**
86     * @return an ObjectMapper with No Null field serialization
87     */
88    @Unremovable
89    public static ObjectMapper getInstanceNoNull() {
90      return jsonHandler.mapperNoNull;
91    }
92  
93    @Singleton
94    public static class RegisterEmptyModuleCustomizer implements ObjectMapperCustomizer {
95  
96      @Override
97      public void customize(final ObjectMapper objectMapper) {
98        // nothing
99      }
100   }
101 }
