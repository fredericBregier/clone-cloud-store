17  package io.clonecloudstore.common.quarkus.properties;
18  
19  import java.time.Duration;
20  
21  import io.clonecloudstore.common.standard.properties.Module;
22  import io.clonecloudstore.common.standard.properties.StandardProperties;
23  import io.clonecloudstore.common.standard.system.SystemPropertyUtil;
24  import io.quarkus.arc.Unremovable;
25  import io.quarkus.runtime.Startup;
26  import io.vertx.mutiny.core.Vertx;
27  import jakarta.enterprise.context.ApplicationScoped;
28  import jakarta.enterprise.inject.spi.CDI;
29  import org.eclipse.microprofile.config.ConfigProvider;
30  import org.jboss.logging.Logger;
31  import org.jboss.logmanager.MDC;
32  
33  /**
34   * Configurable values
35   */
36  @ApplicationScoped
37  @Unremovable
38  public class QuarkusProperties extends StandardProperties {
39    private static final Logger LOGGER = Logger.getLogger(QuarkusProperties.class);
40    /**
41     * Property to define Buffer Size for a Driver Chunk
42     */
43    public static final String CCS_DRIVER_MAX_CHUNK_SIZE = "ccs.driverMaxChunkSize";
44    /**
45     * Property to define if Server will compute SHA 256 on the fly
46     */
47    public static final String CCS_SERVER_COMPUTE_SHA_256 = "ccs.server.computeSha256";
48    /**
49     * Property to define Max transferring time in milliseconds before Time Out (must take into account large file and
50     * bandwidth)
51     */
52    public static final String CCS_CLIENT_RESPONSE_TIMEOUT = "ccs.client.response.timeout";
53    static final int DEFAULT_DRIVER_MAX_CHUNK_SIZE = 512 * 1024 * 1024;
54    private static final int DEFAULT_RESPONSE_TIMEOUT_MS = 300000;
55    public static final String MODULE_MDC = "module";
56    private static long clientResponseTimeOut =
57        SystemPropertyUtil.get(CCS_CLIENT_RESPONSE_TIMEOUT, DEFAULT_RESPONSE_TIMEOUT_MS);
58    private static Duration durationResponseTimeout = Duration.ofMillis(clientResponseTimeOut);
59    private static int driverMaxChunkSize =
60        QuarkusSystemPropertyUtil.getIntegerConfig(CCS_DRIVER_MAX_CHUNK_SIZE, DEFAULT_DRIVER_MAX_CHUNK_SIZE);
61    private static Module module = Module.UNKNOWN;
62    private static boolean computeSha256 = QuarkusSystemPropertyUtil.getBooleanConfig(CCS_SERVER_COMPUTE_SHA_256, false);
63  
64    static {
65      // Get the MACHINE_ID if specified and set it too in System Properties
66      var optional = ConfigProvider.getConfig().getOptionalValue(StandardProperties.CCS_MACHINE_ID, String.class);
67      optional.ifPresent(s -> SystemPropertyUtil.set(StandardProperties.CCS_MACHINE_ID, s));
68      // Get the CCS_BUFFER_SIZE if specified and set it too in System Properties
69      optional = ConfigProvider.getConfig().getOptionalValue(CCS_BUFFER_SIZE, String.class);
70      optional.ifPresent(s -> setBufSize(Integer.parseInt(s)));
71      // Get the CCS_MAX_WAIT_MS if specified and set it too in System Properties
72      optional = ConfigProvider.getConfig().getOptionalValue(CCS_MAX_WAIT_MS, String.class);
73      optional.ifPresent(s -> setMaxWaitMs(Integer.parseInt(s)));
74      // Get the CCS_MAX_TRANSFER_MS if specified and compute dependent values
75      optional = ConfigProvider.getConfig().getOptionalValue(CCS_CLIENT_RESPONSE_TIMEOUT, String.class);
76      optional.ifPresent(s -> setClientResponseTimeOut(Integer.parseInt(s)));
77    }
78  
79    /**
80     * @return The duration before TimeOut occurs except InputStream
81     */
82    public static Duration getDurationResponseTimeout() {
83      return durationResponseTimeout;
84    }
85  
86    public static long clientResponseTimeOut() {
87      return QuarkusProperties.clientResponseTimeOut;
88    }
89  
90    public static void setClientResponseTimeOut(final long clientResponseTimeOut) {
91      SystemPropertyUtil.set(CCS_CLIENT_RESPONSE_TIMEOUT, clientResponseTimeOut);
92      QuarkusProperties.clientResponseTimeOut = clientResponseTimeOut;
93      durationResponseTimeout = Duration.ofMillis(clientResponseTimeOut);
94    }
95  
96    @Startup
97    void initVertx() {
98      setCdiVertx(CDI.current().select(Vertx.class).get().getDelegate());
99      MDC.put(MODULE_MDC, module.name());
100   }
101 
102   /**
103    * @return the maximum Chunk Size (Default 512 MB)
104    */
105   public static int getDriverMaxChunkSize() {
106     return driverMaxChunkSize;
107   }
108 
109   public static void setDriverMaxChunkSize(final int driverMaxChunkSize) {
110     QuarkusProperties.driverMaxChunkSize = driverMaxChunkSize;
111   }
112 
113   public static Module getCcsModule() {
114     return module;
115   }
116 
117   public static void setCcsModule(final Module module) {
118     if (module.ordinal() < QuarkusProperties.module.ordinal()) {
119       QuarkusProperties.module = module;
120       refreshModuleMdc();
121       LOGGER.infof("Set Module = %s", module);
122     }
123   }
124 
125   public static void refreshModuleMdc() {
126     MDC.put(MODULE_MDC, module.name());
127   }
128 
129   protected QuarkusProperties() {
130     super();
131     // Nothing
132   }
133 
134   /**
135    * @return True if the server shall compute Sha256 on the fly
136    */
137   public static boolean serverComputeSha256() {
138     return computeSha256;
139   }
140 
141   /***
142    * To change the behavior dynamically
143    * @param computeSha256 True if the client shall compute Sha256 on the fly
144    */
145   public static void setServerComputeSha256(final boolean computeSha256) {
146     QuarkusProperties.computeSha256 = computeSha256;
147   }
148 
149   public static String confugrationToString() {
150     return String.format("%s, \"%s\":%d, \"%s\":%d, \"%s\":%d, \"%s\":\"%s\"",
151         StandardProperties.confugrationToString(), CCS_DRIVER_MAX_CHUNK_SIZE, getDriverMaxChunkSize(),
152         CCS_SERVER_COMPUTE_SHA_256, getDriverMaxChunkSize(), CCS_CLIENT_RESPONSE_TIMEOUT, clientResponseTimeOut(),
153         MODULE_MDC, getCcsModule());
154   }
155 }
