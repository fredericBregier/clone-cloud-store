17  package io.clonecloudstore.common.quarkus.server.service;
18  
19  import java.io.IOException;
20  
21  import io.clonecloudstore.common.quarkus.client.SimpleClientAbstract;
22  import io.clonecloudstore.common.quarkus.exception.CcsClientGenericException;
23  import io.clonecloudstore.common.quarkus.exception.CcsOperationException;
24  import io.clonecloudstore.common.quarkus.exception.CcsServerGenericException;
25  import io.clonecloudstore.common.quarkus.properties.QuarkusProperties;
26  import io.smallrye.mutiny.subscription.UniEmitter;
27  import jakarta.ws.rs.container.ContainerRequestContext;
28  import jakarta.ws.rs.container.ContainerResponseContext;
29  import jakarta.ws.rs.container.ContainerResponseFilter;
30  import jakarta.ws.rs.core.Response;
31  import jakarta.ws.rs.ext.Provider;
32  import org.jboss.resteasy.reactive.client.api.WebClientApplicationException;
33  
34  import static io.clonecloudstore.common.quarkus.client.SimpleClientAbstract.X_ERROR;
35  import static io.clonecloudstore.common.quarkus.client.SimpleClientAbstract.X_MODULE;
36  import static io.clonecloudstore.common.quarkus.client.SimpleClientAbstract.X_OP_ID;
37  
38  @Provider
39  public class ServerResponseFilter implements ContainerResponseFilter {
40    @Override
41    public void filter(final ContainerRequestContext containerRequestContext,
42                       final ContainerResponseContext responseContext) throws IOException {
43      if (!responseContext.getHeaders().containsKey(X_OP_ID)) {
44        responseContext.getHeaders().putSingle(X_OP_ID, SimpleClientAbstract.getMdcOpId());
45      }
46      if (!responseContext.getHeaders().containsKey(X_MODULE)) {
47        responseContext.getHeaders().putSingle(X_MODULE, QuarkusProperties.getCcsModule().name());
48      }
49    }
50  
51    public static void handleException(final UniEmitter<? super Response> em, final Exception e) {
52      switch (e) {
53        case final CcsClientGenericException cc:
54          em.complete(Response.status(cc.getStatus()).header(X_ERROR, cc.getMessage()).build());
55          break;
56        case final CcsServerGenericException cs:
57          em.complete(Response.status(cs.getStatus()).header(X_ERROR, cs.getMessage()).build());
58          break;
59        case final WebClientApplicationException wc:
60          em.complete(wc.getResponse());
61          break;
62        default:
63          em.complete(Response.status(Response.Status.INTERNAL_SERVER_ERROR).header(X_ERROR, e.getMessage()).build());
64      }
65    }
66  
67    public static void handleExceptionFail(final UniEmitter<?> em, final Exception e) {
68      switch (e) {
69        case final CcsClientGenericException cc:
70          em.fail(cc);
71          break;
72        case final CcsServerGenericException cs:
73          em.fail(cs);
74          break;
75        case final WebClientApplicationException wc:
76          em.fail(wc);
77          break;
78        default:
79          em.fail(new CcsOperationException(e.getMessage()));
80      }
81    }
82  }
