100   /**
101    * Check if object or directory exists
102    */
103   public StorageType objectOrDirectoryExists(final String bucketName, final String objectOrDirectoryName,
104                                              final boolean fullCheck, final String clientId, final String opId,
105                                              final boolean external) throws CcsOperationException {
106     try {
107       // Check in DB and associated status
108       final var daoAccessorObjectIterator =
109           objectRepository.getObjectPrefix(bucketName, objectOrDirectoryName, AccessorStatus.READY);
110       var found = false;
111       if (daoAccessorObjectIterator != null && (daoAccessorObjectIterator.hasNext())) {
112         found = true;
113         final var daoAccessorObject = daoAccessorObjectIterator.next();
114         if (!daoAccessorObject.getName().equals(objectOrDirectoryName)) {
115           return StorageType.DIRECTORY;
116         }
117       }
118       if (!found) {
119         final var response = remoteCheckObject(external, bucketName, objectOrDirectoryName, clientId, opId);
120         if (response.targetId() != null && AccessorProperties.isFixOnAbsent()) {
121           generateReplicationOrderForObject(bucketName, objectOrDirectoryName, clientId, opId, response.targetId(), 0,
122               null);
123         }
124         return response.response();
125       }
126       SystemTools.consumeAll(daoAccessorObjectIterator);
127       if (fullCheck) {
128         // Check in S3
129         return getStorageType(bucketName, objectOrDirectoryName);
130       } else {
131         return StorageType.OBJECT;
132       }
133     } catch (final CcsDbException e) {
134       throw new CcsOperationException(
135           "Database error on check object or directory : " + bucketName + " - " + objectOrDirectoryName, e);
136     }
137   }

139   public void generateReplicationOrderForObject(final String bucketName, final String objectOrDirectoryName,
140                                                 final String clientId, final String opId, final String targetId,
141                                                 long len, String hash) {
142     localReplicatorService.generateLocalReplicationOrder(
143         new ReplicatorOrder(opId, targetId, ServiceProperties.getAccessorSite(), clientId, bucketName,
144             objectOrDirectoryName, len, hash, ReplicatorConstants.Action.CREATE));
145   }

147   private ReplicatorResponse<StorageType> remoteCheckObject(final boolean remoteCheck, final String bucketName,
148                                                             final String objectOrDirectoryName, final String clientId,
149                                                             final String opId) {
150     try {
151       if (remoteCheck && AccessorProperties.isRemoteRead()) {
152         return localReplicatorService.remoteCheckObject(bucketName, objectOrDirectoryName, clientId, opId);
153       }
154     } catch (final CcsOperationException ignore) {
155       // Ignore
156     }
157     return new ReplicatorResponse<>(StorageType.NONE, null);
158   }

160   private StorageType getStorageType(final String bucketName, final String objectOrDirectoryName) {
161     try (final var driver = storageDriverFactory.getInstance()) {
162       return driver.directoryOrObjectExistsInBucket(bucketName, objectOrDirectoryName);
163     } catch (final DriverException e) {
164       throw new CcsOperationException(mesg(bucketName, objectOrDirectoryName) + ISSUE_STRING + e.getMessage(), e);
165     }
166   }

168   /**
169    * @param bucketName technical name
170    * @param filter     the filter to apply on Objects
171    * @return a stream (InputStream) of AccessorObject line by line (newline separated)
172    */
173   public InputStream filterObjects(final String bucketName, final AccessorFilter filter) throws CcsOperationException {
174     try {
175       DbQuery query = new DbQuery(RestQuery.QUERY.EQ, BUCKET, bucketName);
176       if (filter != null) {
177         final var subQuery = DbQueryAccessorHelper.getDbQuery(filter);
178         if (subQuery != null) {
179           query = new DbQuery(RestQuery.CONJUNCTION.AND, query, subQuery);
180         }
181       }
182       final var iterator = objectRepository.findIterator(query);
183       return StreamIteratorUtils.getInputStreamFromIterator(iterator, source -> ((DaoAccessorObject) source).getDto(),
184           AccessorObject.class);
185     } catch (final CcsDbException | IOException e) {
186       throw new CcsOperationException("Database error on filter object : " + bucketName + " - " +
187           (filter != null ? filter.toString() : "No Filter"), e);
188     }
189   }

191   /**
192    * Check from DB if Object is pullable
193    */
194   public ReplicatorResponse<AccessorObject> checkPullable(final String bucketName, final String objectName,
195                                                           final boolean external, final String clientId,
196                                                           final String opId)
197       throws CcsNotExistException, CcsOperationException {
198     // TODO shall check if bucket is also available
199     try {
200       // Search objectName in database and check status
201       final var daoAccessorObject = objectRepository.getObject(bucketName, objectName, AccessorStatus.READY);
202       // TODO check if driver says it exists too
203       // If not found or not available or not in driver
204       if (daoAccessorObject == null || !AccessorStatus.READY.equals(daoAccessorObject.getStatus())) {
205         // Else use remote check.
206         final var storageType = remoteCheckObject(external, bucketName, objectName, clientId, opId);
207         if (StorageType.OBJECT.equals(storageType.response())) {
208           return new ReplicatorResponse<>(new AccessorObject().setBucket(bucketName).setName(objectName),
209               storageType.targetId());
210         }
211         throw new CcsNotExistException("Object not found");
212       }
213       return new ReplicatorResponse<>(daoAccessorObject.getDto(), null);
214     } catch (final CcsDbException e) {
215       throw new CcsOperationException("Database error on check pullable : " + bucketName + " - " + objectName, e);
216     }
217   }

265   /**
266    * Before really creating Object, creates it in DB if possible
267    *
268    * @throws CcsNotAcceptableException if already in creation step
269    */
270   public DaoAccessorObject createObject(final AccessorObject accessorObject, final String hash, final long len)
271       throws CcsOperationException, CcsAlreadyExistException, CcsNotExistException, CcsNotAcceptableException {
272     try {
273       // Check non-existence in DB/S3
274       // First Bucket check
275       final var bucket = bucketRepository.findBucketById(accessorObject.getBucket());
276       if (bucket == null || !AccessorStatus.READY.equals(bucket.getStatus())) {
277         throw new CcsNotExistException("Bucket does not exists");
278       }
279       // Now Object check
280       var daoAccessorObject = objectRepository.getObject(accessorObject.getBucket(), accessorObject.getName());
281       if (daoAccessorObject != null) {
282         if (daoAccessorObject.getStatus() != AccessorStatus.DELETED) {
283           if (daoAccessorObject.getStatus() == AccessorStatus.UPLOAD) {
284             throw new CcsNotAcceptableException(
285                 mesg(accessorObject.getBucket(), accessorObject.getName()) + " already in creation, Status: " +
286                     daoAccessorObject.getStatus());
287           }
288           throw new CcsAlreadyExistException(
289               mesg(accessorObject.getBucket(), accessorObject.getName()) + STATUS_STRING +
290                   daoAccessorObject.getStatus());
291         }
292         daoAccessorObject.setStatus(AccessorStatus.UPLOAD).setHash(hash).setSize(len).setCreation(Instant.now());
293         if (ParametersChecker.isNotEmpty(hash) || len > 0) {
294           objectRepository.updateFull(daoAccessorObject);
295         } else {
296           objectRepository.updateObjectStatus(accessorObject.getBucket(), accessorObject.getName(),
297               AccessorStatus.UPLOAD, daoAccessorObject.getCreation());
298         }
299       } else {
300         // Create object in database with status "In progress"
301         accessorObject.setStatus(AccessorStatus.UPLOAD);
302         daoAccessorObject = objectRepository.createEmptyItem().fromDto(accessorObject);
303         daoAccessorObject.setId(GuidLike.getGuid()).setSite(ServiceProperties.getAccessorSite())
304             .setCreation(Instant.now()).setStatus(AccessorStatus.UPLOAD);
305         objectRepository.insert(daoAccessorObject);
306       }
307       return daoAccessorObject;
308     } catch (final CcsDbException e) {
309       throw new CcsOperationException(
310           "Database error on create object : " + accessorObject.getBucket() + " - " + accessorObject.getName(), e);
311     }
312   }

314   /**
315    * Once Object really created in Driver Storage, finalize the Object in DB and Replicator if needed
316    */
317   public DaoAccessorObject createObjectFinalize(final String bucketName, final String objectName, final String hash,
318                                                 final long len, final String clientId, final boolean external)
319       throws CcsOperationException {
320     try {
321       // Update Database with status available and metadata from ObjectStorage
322       objectRepository.updateObjectStatusHashLen(bucketName, objectName, AccessorStatus.READY, hash, len);
323       if (external) {
324         // Send message to replicator topic.
325         localReplicatorService.create(bucketName, objectName, clientId, len, hash);
326       }
327       return objectRepository.getObject(bucketName, objectName);
328     } catch (final CcsDbException e) {
329       throw new CcsOperationException("Database error on create object finalize : " + bucketName + " - " + objectName,
330           e);
331     }
332   }

334   /**
335    * Delete object in DB and through Replicator if needed
336    */
337   public void deleteObject(final String bucketName, final String objectName, final String clientId,
338                            final boolean external)
339       throws CcsDeletedException, CcsNotExistException, CcsOperationException {
340     try {
341       // Check existence first in DB
342       final var daoAccessorObject = objectRepository.getObject(bucketName, objectName);
343       if (daoAccessorObject != null) {
344         LOGGER.infof("Dao: %s", daoAccessorObject);
345         if (daoAccessorObject.getStatus() != AccessorStatus.READY) {
346           if (daoAccessorObject.getStatus().equals(AccessorStatus.DELETED)) {
347             throw new CcsDeletedException(mesg(bucketName, objectName) + STATUS_STRING + daoAccessorObject.getStatus());
348           }
349           throw new CcsOperationException(mesg(bucketName, objectName) + STATUS_STRING + daoAccessorObject.getStatus());
350         }
351         objectRepository.updateObjectStatus(bucketName, objectName, AccessorStatus.DELETING, null);
352       } else {
353         throw new CcsNotExistException(mesg(bucketName, objectName) + STATUS_STRING + AccessorStatus.UNKNOWN);
354       }
355       // Delete in S3
356       deleteObjectOnStorage(bucketName, objectName, daoAccessorObject);
357       // Update status in DB to Deleted
358       objectRepository.updateObjectStatus(bucketName, objectName, AccessorStatus.DELETED, null);
359       if (external) {
360         // Send message to replicator topic
361         localReplicatorService.delete(bucketName, objectName, clientId);
362       }
363     } catch (final CcsDbException e) {
364       throw new CcsOperationException("Database error on delete object : " + bucketName + " - " + objectName, e);
365     }
366   }

368   private void deleteObjectOnStorage(final String bucketName, final String objectName,
369                                      final DaoAccessorObject daoAccessorObject) throws CcsDbException {
370     try (final var driver = storageDriverFactory.getInstance()) {
371       driver.objectDeleteInBucket(bucketName, objectName);
372     } catch (final DriverNotFoundException e) {
373       // Ignore
374       LOGGER.infof("Try to delete but not found: %s", daoAccessorObject);
375     } catch (final DriverException e) {
376       objectRepository.updateObjectStatus(bucketName, objectName, AccessorStatus.ERR_DEL, null);
377       throw new CcsOperationException(mesg(bucketName, objectName) + STATUS_STRING + daoAccessorObject.getStatus(), e);
378     }
379   }
