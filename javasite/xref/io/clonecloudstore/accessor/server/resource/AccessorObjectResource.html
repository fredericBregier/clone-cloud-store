391   /**
392    * Listing of Objects
393    *
394    * @param alreadyCompressed If True, and if the InputStream is to be compressed, will be kept as is; else will
395    *                          compress the InputStream if it has to be
396    */
397   protected Uni<Response> readObjectList(final HttpServerRequest request, final Closer closer,
398                                          final AccessorObject businessIn, final boolean alreadyCompressed) {
399     LOGGER.debugf("GET start");
400     getNativeStream().setup(request, closer, false, businessIn, 0, null, alreadyCompressed);
401     return Uni.createFrom().emitter(em -> {
402       try {
403         em.complete(getNativeStream().pullList());
404       } catch (NativeServerResponseException e) {
405         em.complete(e.getResponse());
406       } catch (final Exception e) {
407         em.complete(createErrorResponse(e));
408       }
409     });
410   }
