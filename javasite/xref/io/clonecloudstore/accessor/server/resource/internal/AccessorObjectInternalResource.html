169   @Path("{bucketName}/{pathDirectoryOrObject:.+}")
170   @HEAD
171   @Operation(summary = "Check if object or directory exist", description = "Check if object or directory exist " +
172       "(fullcheck true implies check down to Storage) ")
173   @APIResponse(responseCode = "204", description = "OK", headers = {
174       @Header(name = X_TYPE, description = "Type as StorageType", schema = @Schema(type = SchemaType.STRING,
175           enumeration = {
176           "NONE", "BUCKET", "DIRECTORY", "OBJECT"}))})
177   @APIResponse(responseCode = "400", description = "Bad Request")
178   @APIResponse(responseCode = "401", description = "Unauthorized")
179   @APIResponse(responseCode = "500", description = "Internal Error")
180   @Blocking
181   public Uni<Response> checkObjectOrDirectory(@PathParam("bucketName") String bucketName,
182                                               @PathParam("pathDirectoryOrObject") String pathDirectoryOrObject,
183                                               @Parameter(name = FULL_CHECK, description = "If True implies Storage " +
184                                                   "checking", in = ParameterIn.QUERY, schema = @Schema(type =
185                                                   SchemaType.BOOLEAN), required = false) @DefaultValue("false") @QueryParam(FULL_CHECK) boolean fullCheck,
186                                               @Parameter(name = AccessorConstants.Api.X_CLIENT_ID, description =
187                                                   "Client ID", in = ParameterIn.HEADER, schema = @Schema(type =
188                                                   SchemaType.STRING), required = true) @HeaderParam(X_CLIENT_ID) String clientId,
189                                               @Parameter(name = X_OP_ID, description = "Operation ID", in =
190                                                   ParameterIn.HEADER, schema = @Schema(type = SchemaType.STRING),
191                                                   required = false) @HeaderParam(X_OP_ID) final String opId) {
192     return Uni.createFrom().emitter(em -> {
193       final var finalBucketName = DaoAccessorBucketRepository.getRealBucketName(clientId, bucketName, false);
194       final var finalObjectName = ParametersChecker.getSanitizedName(pathDirectoryOrObject);
195       LOGGER.infof(BUCKETNAME_OBJECT, finalBucketName, finalObjectName);
196       try {
197         final var storageType = service.objectOrDirectoryExists(finalBucketName, finalObjectName, fullCheck);
198         if (storageType.equals(StorageType.NONE)) {
199           em.complete(Response.status(Response.Status.NOT_FOUND).header(X_TYPE, StorageType.NONE).build());
200         } else {
201           em.complete(Response.ok().header(X_TYPE, storageType).build());
202         }
203       } catch (final RuntimeException e) {
204         ServerResponseFilter.handleException(em, e);
205       }
206     });
207   }

295   /**
296    * Listing of Objects
297    *
298    * @param alreadyCompressed If True, and if the InputStream is to be compressed, will be kept as is; else will
299    *                          compress the InputStream if it has to be
300    */
301   protected Uni<Response> readObjectList(final HttpServerRequest request, final Closer closer,
302                                          final AccessorObject businessIn, final boolean alreadyCompressed) {
303     LOGGER.debugf("GET start");
304     getNativeStream().setup(request, closer, false, businessIn, 0, null, alreadyCompressed);
305     return Uni.createFrom().emitter(em -> {
306       try {
307         em.complete(getNativeStream().pullList());
308       } catch (NativeServerResponseException e) {
309         em.complete(e.getResponse());
310       } catch (final Exception e) {
311         em.complete(createErrorResponse(e));
312       }
313     });
