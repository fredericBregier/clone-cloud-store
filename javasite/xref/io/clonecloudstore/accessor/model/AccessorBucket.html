17  package io.clonecloudstore.accessor.model;
18  
19  import java.time.Instant;
20  import java.util.Objects;
21  
22  import com.fasterxml.jackson.core.JsonProcessingException;
23  import io.clonecloudstore.common.standard.properties.StandardProperties;
24  import io.clonecloudstore.common.standard.system.ParametersChecker;
25  import io.clonecloudstore.common.standard.system.SystemTools;
26  import io.quarkus.runtime.annotations.RegisterForReflection;
27  
28  /**
29   * Accessor Bucket DTO
30   */
31  @RegisterForReflection
32  public class AccessorBucket {
33    private String id;
34    private String site;
35    private String name;
36    private Instant creation;
37    private Instant expires;
38  
39    private AccessorStatus status = AccessorStatus.UNKNOWN;
40  
41    public AccessorBucket() {
42      // Empty
43    }
44  
45    public String getId() {
46      return id;
47    }
48  
49    public AccessorBucket setId(final String id) {
50      ParametersChecker.checkSanityBucketName(id);
51      this.id = id;
52      return this;
53    }
54  
55    public String getSite() {
56      return site;
57    }
58  
59    public AccessorBucket setSite(final String site) {
60      this.site = site;
61      return this;
62    }
63  
64    public String getName() {
65      return name;
66    }
67  
68    public AccessorBucket setName(final String name) {
69      ParametersChecker.checkSanityBucketName(name);
70      this.name = name;
71      return this;
72    }
73  
74    public Instant getCreation() {
75      return creation;
76    }
77  
78    public AccessorBucket setCreation(final Instant creation) {
79      this.creation = SystemTools.toMillis(creation);
80      return this;
81    }
82  
83    public Instant getExpires() {
84      return expires;
85    }
86  
87    public AccessorBucket setExpires(final Instant expires) {
88      this.expires = SystemTools.toMillis(expires);
89      return this;
90    }
91  
92    public AccessorStatus getStatus() {
93      return status;
94    }
95  
96    public AccessorBucket setStatus(final AccessorStatus status) {
97      this.status = status;
98      return this;
99    }
100 
101   @Override
102   public boolean equals(final Object obj) {
103     if (this == obj) {
104       return true;
105     }
106     if (obj instanceof AccessorBucket that) {
107       return Objects.equals(id, that.id) && Objects.equals(site, that.site) && Objects.equals(name, that.name) &&
108           Objects.equals(creation, that.creation) && Objects.equals(expires, that.expires) &&
109           Objects.equals(status, that.status);
110     }
111     return false;
112   }
113 
114   @Override
115   public int hashCode() {
116     return Objects.hash(id, site, name, creation, expires, status);
117   }
118 
119   @Override
120   public String toString() {
121     try {
122       return StandardProperties.getObjectMapper().writeValueAsString(this);
123     } catch (final JsonProcessingException e) {
124       return "{" + "\"id\":\"" + id + "\"" + ", \"site\":\"" + site + "\"" + ", \"name\":\"" + name + "\"" +
125           ", \"status\":\"" + status + "\"" + ", \"creation\":" + creation + ", \"expires\":" + expires + "}";
126     }
127   }
128 
129   public AccessorBucket cloneInstance() {
130     final var accessorBucket = new AccessorBucket();
131     final var finalStatus = getStatus() == null ? AccessorStatus.UNKNOWN : getStatus();
132     accessorBucket.setId(getId()).setSite(getSite()).setName(getName()).setStatus(finalStatus)
133         .setCreation(getCreation()).setExpires(getExpires());
134     return accessorBucket;
135   }
136 }
