133   private DaoAccessorObject getObject(final String bucket, final String name)
134       throws InterruptedException, CcsDbException {
135     for (int i = 0; i < 100; i++) {
136       var dao = objectRepository.getObject(bucket, name);
137       if (dao != null) {
138         return dao;
139       }
140       Thread.sleep(10);
141     }
142     throw new CcsDbException(new CcsNotExistException(bucket + " / " + name));
143   }

145   private void checkObject(final String bucket, final String name, final AccessorStatus status)
146       throws InterruptedException, CcsDbException {
147     DaoAccessorObject dao = null;
148     for (int i = 0; i < 100; i++) {
149       dao = getObject(bucket, name);
150       if (status.equals(dao.getStatus())) {
151         return;
152       }
153       Thread.sleep(10);
154     }
155     fail("Object has not the desired status: " + status + " = " + dao);
156   }

222       // Try recreate

225       Thread.yield();
226       temp = MetricsCheck.waitForValueTest(createObject, creObject, 300);

232     }
233     LOG.info("Check Delete of Object");
234     // Check delete for Object
235     try (final var driver = storageDriverFactory.getInstance()) {

237           objectRepository.getObject(CLIENTID_BUCKET, OBJECT_NAME).getStatus());

297       emitter.send(orderBucket2);
298       Thread.yield();
299       temp = MetricsCheck.waitForValueTest(deleteBucket, delBucket, 300);
300       assertEquals(delBucket, temp);
301       delBucket = temp;
302       assertEquals(StorageType.NONE, driver.directoryOrObjectExistsInBucket(CLIENTID_BUCKET, OBJECT_NAME));
303       Assertions.assertEquals(AccessorStatus.DELETED,
304           objectRepository.getObject(CLIENTID_BUCKET, OBJECT_NAME).getStatus());
305       assertFalse(driver.bucketExists(CLIENTID_BUCKET));
306       Assertions.assertEquals(AccessorStatus.DELETED, bucketRepository.findBucketById(CLIENTID_BUCKET).getStatus());
307       // Recreate Bucket then Delete it
308       final var orderBucket3 = new ReplicatorOrder(orderBucket, ReplicatorConstants.Action.CREATE);
309       emitter.send(orderBucket3);
310       Thread.yield();
311       temp = MetricsCheck.waitForValueTest(createBucket, creBucket + 1, 300);

352     }

366       assertFalse(driver.bucketExists("clientid2-bucket"));

368     }
369     LOG.info("Check Wrong Name");
370     final var orderBucket2 =
371         new ReplicatorOrder(OP_ID, TO, FROM, "clientid2", "WrongBucketName", ReplicatorConstants.Action.CREATE);
372     try (final var driver = storageDriverFactory.getInstance()) {
373       // Wrong name
374       assertFalse(driver.bucketExists("WrongBucketName"));
375       emitter.send(orderBucket2);
376       Thread.sleep(WAIT_FOR_CONSUME);
377       assertFalse(driver.bucketExists("WrongBucketName"));
378       assertNull(bucketRepository.findBucketById("WrongBucketName"));
379     }
380     final var orderBucket3 =
381         new ReplicatorOrder(OP_ID, TO, FROM, "clientid2", "clientid3-bucket", ReplicatorConstants.Action.DELETE);
382     try (final var driver = storageDriverFactory.getInstance()) {
383       LOG.info("Check Delete non existing");
384       // Delete non existing
385       emitter.send(orderBucket3);
386       Thread.sleep(WAIT_FOR_CONSUME);
387       assertFalse(driver.bucketExists("clientid3-bucket"));
388       assertNull(bucketRepository.findBucketById("clientid3-bucket"));

393       Thread.sleep(WAIT_FOR_CONSUME);
394       driver.bucketDelete("clientid3-bucket");

396       emitter.send(orderBucket3);
397       Thread.sleep(WAIT_FOR_CONSUME);
398       assertFalse(driver.bucketExists("clientid3-bucket"));
399       Assertions.assertEquals(AccessorStatus.ERR_DEL, bucketRepository.findBucketById("clientid3-bucket").getStatus());
400       LOG.info("Check Retry Delete");
401       // Retry to delete already in Error Delete
402       emitter.send(orderBucket3);
403       Thread.sleep(WAIT_FOR_CONSUME);
404       assertFalse(driver.bucketExists("clientid3-bucket"));
405       Assertions.assertEquals(AccessorStatus.ERR_DEL, bucketRepository.findBucketById("clientid3-bucket").getStatus());
406     }
407     final var orderBucket4 =
408         new ReplicatorOrder(OP_ID, TO, FROM, "clientid2", "clientid3-bucket2", ReplicatorConstants.Action.DELETE);
409     try (final var driver = storageDriverFactory.getInstance()) {
410       LOG.info("Create a bucket that already exists");
411       // Create first
412       final var orderBucket5 = new ReplicatorOrder(orderBucket4, ReplicatorConstants.Action.CREATE);
413       emitter.send(orderBucket5);
414       Thread.sleep(WAIT_FOR_CONSUME);
415       assertTrue(driver.bucketExists("clientid3-bucket2"));
416       Assertions.assertEquals(AccessorStatus.READY, bucketRepository.findBucketById("clientid3-bucket2").getStatus());
417       // Now delete DB entry
418       bucketRepository.deleteWithPk("clientid3-bucket2");
419       assertTrue(driver.bucketExists("clientid3-bucket2"));
420       assertNull(bucketRepository.findBucketById("clientid3-bucket2"));
421       // And try to recreate it
422       emitter.send(orderBucket5);
423       Thread.sleep(WAIT_FOR_CONSUME);
424       assertTrue(driver.bucketExists("clientid3-bucket2"));
425       Assertions.assertEquals(AccessorStatus.ERR_UPL, bucketRepository.findBucketById("clientid3-bucket2").getStatus());
426     }
