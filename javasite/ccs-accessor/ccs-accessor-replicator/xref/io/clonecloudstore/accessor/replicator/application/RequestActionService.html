81    /**
82     * Create bucket from client Id and technicalBucketName
83     *
84     * @param clientId            Format client ID use to identify client
85     * @param technicalBucketName Technical Bucket Name
86     * @return AccessorBucket add on Database and in object storage
87     */
88    public AccessorBucket createBucket(final String clientId, final String technicalBucketName)
89        throws CcsAlreadyExistException, CcsOperationException {
90      final var bucketName = DaoAccessorBucketRepository.getBucketName(clientId, technicalBucketName);
91      try {
92        //Check format (special char, uppercase...) and size (min, max) for bucket name and technicalBucketName
93        ParametersChecker.checkSanityBucketName(bucketName);
94        ParametersChecker.checkSanityBucketName(technicalBucketName);
95      } catch (final CcsInvalidArgumentRuntimeException e) {
96        final var message = String.format("Bucket Name %s is invalid", technicalBucketName);
97        throw new CcsOperationException(message);
98      }
99      //Check if bucket already exist in database.
100     AccessorBucket result;
101     try {
102       result = bucketRepository.findBucketById(technicalBucketName);
103     } catch (final CcsDbException e) {
104       throw new CcsOperationException("Error on find repository with technical ID", e);
105     }
106     if (result != null && !AccessorStatus.DELETED.equals(result.getStatus())) {
107       final var message = String.format("Bucket %s already exist", technicalBucketName);
108       LOGGER.errorf(message);
109       throw new CcsAlreadyExistException(message);
110     } else if (result == null) {
111       //Create Bucket in Database with Status In Progress.
112       //Only name and technical name (over value are erased during creation).
113       result = new AccessorBucket();
114       result.setName(bucketName);
115       result.setId(technicalBucketName);
116       result.setSite(ServiceProperties.getAccessorSite());
117     }
118     return createBucketFromDto(result);
119   }

121   /**
122    * Create Bucket from DTO Accessor
123    *
124    * @param bucket Accessor Bucket with functional data
125    * @return AccessorBucket add on Database and in object storage
126    */
127   private AccessorBucket createBucketFromDto(final AccessorBucket bucket) throws CcsOperationException {
128     LOGGER.debugf("Create bucket %s in database and in object storage", bucket.getId());
129     try {
130       //Insert in Database
131       // Since it can be in DELETED status, it might be updated (creation + status)
132       bucket.setCreation(Instant.now());
133       AccessorBucket newBucket = bucket;
134       if (AccessorStatus.DELETED.equals(newBucket.getStatus())) {
135         newBucket.setStatus(AccessorStatus.UPLOAD);
136         bucketRepository.updateBucketStatus(newBucket, AccessorStatus.UPLOAD, newBucket.getCreation());
137       } else {
138         newBucket.setStatus(AccessorStatus.UPLOAD);
139         newBucket = bucketRepository.insertBucket(newBucket);
140       }
141       //Create Bucket in Object Storage
142       final var storageBucket = new StorageBucket(newBucket.getId(), newBucket.getCreation());
143       return createBucketOnStorage(storageBucket, newBucket);
144     } catch (final CcsDbException e) {
145       throw new CcsOperationException("Database error on bucket " + bucket.getName() + " creation", e);
146     } finally {
147       flushBucket();
148     }
149   }
150 
151   private AccessorBucket createBucketOnStorage(final StorageBucket storageBucket, final AccessorBucket bucket)
152       throws CcsDbException {
153     try (final var storageDriver = storageDriverFactory.getInstance()) {
154       // retrieving metadata from storage, would data be duplicated in database
155       storageDriver.bucketCreate(storageBucket);
156       //Update Database in Database with status Available.
157       var newBucket = bucketRepository.updateBucketStatus(bucket, AccessorStatus.READY, null);
158       LOGGER.debugf("Bucket %s created in database and in object storage", bucket.getId());
159       return newBucket;
160     } catch (final CcsOperationException e) {
161       LOGGER.error(e.getMessage());
162       bucketRepository.updateBucketStatus(bucket, AccessorStatus.ERR_UPL, null);
163       throw e;
164     } catch (final CcsInvalidArgumentRuntimeException | DriverException | CcsServerGenericException e) {
165       LOGGER.error(e.getMessage());
166       bucketRepository.updateBucketStatus(bucket, AccessorStatus.ERR_UPL, null);
167       throw new CcsOperationException(e.getMessage(), e);
168     }
169   }

179   /**
180    * Delete bucket from technical bucket name
181    *
182    * @param technicalBucketName Bucket technical name
183    * @return the associated DTO- deleted
184    */
185   public AccessorBucket deleteBucket(final String technicalBucketName)
186       throws CcsNotExistException, CcsDeletedException, CcsOperationException, CcsNotAcceptableException {
187     try {
188       final var bucket = bucketRepository.findBucketById(technicalBucketName);
189       if (bucket == null) {
190         throw new CcsNotExistException("Bucket " + technicalBucketName + " doesn't exist");
191       } else if (bucket.getStatus() == AccessorStatus.READY) {
192         final var count = objectRepository.count(new DbQuery(RestQuery.CONJUNCTION.AND,
193             new DbQuery(RestQuery.QUERY.EQ, DaoAccessorObjectRepository.BUCKET, bucket.getId()),
194             new DbQuery(RestQuery.QUERY.EQ, DaoAccessorObjectRepository.STATUS, AccessorStatus.READY)));
195         if (count > 0) {
196           throw new CcsNotAcceptableException("Bucket is not empty");
197         }
198         bucketRepository.updateBucketStatus(bucket, AccessorStatus.DELETING, null);
199         return deleteBucketOnStorage(technicalBucketName, bucket);
200       } else if (bucket.getStatus() == AccessorStatus.DELETED) {
201         throw new CcsDeletedException("Bucket " + technicalBucketName + " is already deleted");
202       } else {
203         throw new CcsOperationException("Bucket Status " + bucket.getStatus());
204       }
205     } catch (final CcsDbException e) {
206       throw new CcsOperationException("Database error on delete bucket " + technicalBucketName, e);
207     } finally {
208       flushBucket();
209     }
210   }

212   private AccessorBucket deleteBucketOnStorage(final String technicalBucketName, final AccessorBucket bucket)
213       throws CcsDbException {
214     try {
215       try (final var storageDriver = storageDriverFactory.getInstance()) {
216         // retrieving metadata from storage, would data be duplicated in database
217         storageDriver.bucketDelete(technicalBucketName);
218       }
219       return bucketRepository.updateBucketStatus(bucket, AccessorStatus.DELETED, null);
220     } catch (final DriverException | RuntimeException e) {
221       LOGGER.error(e.getMessage());
222       bucketRepository.updateBucketStatus(bucket, AccessorStatus.ERR_DEL, null);
223       throw new CcsOperationException(e.getMessage(), e);
224     }
225   }

227   public void createObject(final ReplicatorOrder replicatorOrder)
228       throws CcsNotExistException, CcsDeletedException, CcsOperationException, CcsNotAcceptableException {
229     final var daoObject = createObjectInternal(replicatorOrder);
230     LOGGER.debugf("Debug Log Creation: %s from %s", daoObject.getBucket(), daoObject);
231     try (final var driverApi = storageDriverFactory.getInstance();
232          final var client = localReplicatorApiClientFactory.newClient()) {
233       final var result = client.readRemoteObject(replicatorOrder.bucketName(), replicatorOrder.objectName(),
234           replicatorOrder.clientId(), replicatorOrder.fromSite(), GuidLike.getGuid());
235       final var dto = result.dtoOut();
236       updateObjectFromRemote(daoObject, dto);
237       final var objectStorage =
238           new StorageObject(daoObject.getBucket(), daoObject.getName(), daoObject.getHash(), daoObject.getSize(),
239               daoObject.getCreation(), null, daoObject.getMetadata());
240       driverApi.objectPrepareCreateInBucket(objectStorage, result.inputStream());
241       driverApi.objectFinalizeCreateInBucket(daoObject.getBucket(), daoObject.getName(), daoObject.getSize(),
242           daoObject.getHash());
243       createObjectFinalize(daoObject);
244     } catch (final DriverAlreadyExistException e) {
245       objectInError(daoObject, replicatorOrder);
246       throw new CcsAlreadyExistException(e.getMessage(), e);
247     } catch (final CcsWithStatusException e) {
248       objectInError(daoObject, replicatorOrder);
249       throw CcsServerGenericExceptionMapper.getCcsException(e.getStatus(), e.getMessage(), e);
250     } catch (final DriverNotFoundException e) {
251       objectInError(daoObject, replicatorOrder);
252       throw new CcsNotExistException(e.getMessage(), e);
253     } catch (final CcsNotExistException | CcsServerGenericException e) {
254       objectInError(daoObject, replicatorOrder);
255       throw e;
256     } catch (final DriverException e) {
257       objectInError(daoObject, replicatorOrder);
258       throw new CcsOperationException(e.getMessage(), e);
259       // CcsAlreadyExistException ignored
260     } finally {
261       flushObject();
262     }
263   }

273   private void objectInError(final DaoAccessorObject daoObject, final ReplicatorOrder replicatorOrder)
274       throws CcsOperationException {
275     try {
276       LOGGER.debugf("Update to Error %s", replicatorOrder);
277       objectRepository.updateObjectStatus(daoObject.getBucket(), daoObject.getName(), AccessorStatus.ERR_UPL,
278           Instant.now());
279     } catch (final CcsDbException ex) {
280       throw new CcsOperationException(ex.getMessage(), ex);
281     }

284   /**
285    * Before really creating Object, creates it in DB if possible
286    *
287    * @throws CcsNotAcceptableException if already in creation step
288    */
289   private DaoAccessorObject createObjectInternal(final ReplicatorOrder replicatorOrder)
290       throws CcsAlreadyExistException, CcsNotExistException, CcsNotAcceptableException, CcsOperationException {
291     try {
292       // Check non-existence in DB/S3
293       // First Bucket check
294       final var bucket = bucketRepository.findBucketById(replicatorOrder.bucketName());
295       if (bucket == null || !AccessorStatus.READY.equals(bucket.getStatus())) {
296         throw new CcsNotExistException("Bucket does not exists");
297       }
298       // Now Object check
299       var daoAccessorObject = objectRepository.getObject(replicatorOrder.bucketName(), replicatorOrder.objectName());
300       if (daoAccessorObject != null) {
301         if (daoAccessorObject.getStatus() != AccessorStatus.DELETED) {
302           if (daoAccessorObject.getStatus() == AccessorStatus.UPLOAD) {
303             throw new CcsNotAcceptableException(
304                 mesg(replicatorOrder.bucketName(), replicatorOrder.objectName()) + " already in creation, Status: " +
305                     daoAccessorObject.getStatus());
306           }
307           throw new CcsAlreadyExistException(
308               mesg(replicatorOrder.bucketName(), replicatorOrder.objectName()) + STATUS_STRING +
309                   daoAccessorObject.getStatus());
310         }
311         daoAccessorObject.setStatus(AccessorStatus.UPLOAD).setHash(replicatorOrder.hash())
312             .setSize(replicatorOrder.size()).setCreation(Instant.now());
313         if (ParametersChecker.isNotEmpty(replicatorOrder.hash()) || replicatorOrder.size() > 0) {
314           objectRepository.updateFull(daoAccessorObject);
315         } else {
316           objectRepository.updateObjectStatus(replicatorOrder.bucketName(), replicatorOrder.objectName(),
317               AccessorStatus.UPLOAD, daoAccessorObject.getCreation());
318         }
319       } else {
320         //Create object in database with status "In progress"
321         daoAccessorObject = objectRepository.createEmptyItem();
322         daoAccessorObject.setBucket(replicatorOrder.bucketName()).setName(replicatorOrder.objectName())
323             .setHash(replicatorOrder.hash()).setSize(replicatorOrder.size());
324         daoAccessorObject.setId(GuidLike.getGuid()).setSite(ServiceProperties.getAccessorSite())
325             .setCreation(Instant.now()).setStatus(AccessorStatus.UPLOAD);
326         objectRepository.insert(daoAccessorObject);
327       }
328       return daoAccessorObject;
329     } catch (final CcsDbException e) {
330       throw new CcsOperationException(
331           "Database error on create object : " + replicatorOrder.bucketName() + " - " + replicatorOrder.objectName(),
332           e);
333     }
334   }

336   private void updateObjectFromRemote(final DaoAccessorObject daoObject, final AccessorObject dto) {
337     try {
338       objectRepository.updateFromDto(daoObject, dto);
339     } catch (CcsDbException e) {
340       throw new CcsOperationException(e);
341     }

344   /**
345    * Once Object really created in Cloud Cloud Store, finalize the Object in DB
346    */
347   private DaoAccessorObject createObjectFinalize(final DaoAccessorObject daoObject) throws CcsOperationException {
348     try {
349       // Update Database with status available and metadata from ObjectStorage
350       objectRepository.updateObjectStatusHashLen(daoObject.getBucket(), daoObject.getName(), AccessorStatus.READY,
351           daoObject.getHash(), daoObject.getSize());
352       return objectRepository.getObject(daoObject.getBucket(), daoObject.getName());
353     } catch (final CcsDbException e) {
354       throw new CcsOperationException(
355           "Database error on create object finalize : " + daoObject.getBucket() + " - " + daoObject.getName(), e);
356     }
357   }

359   /**
360    * Delete object in DB and through Replicator if needed
361    */
362   public void deleteObject(final String bucketName, final String objectName)
363       throws CcsDeletedException, CcsNotExistException, CcsOperationException {
364     try {
365       // Check existence first in DB
366       final var daoAccessorObject = objectRepository.getObject(bucketName, objectName);
367       if (daoAccessorObject != null) {
368         LOGGER.debugf("Dao: %s", daoAccessorObject);
369         if (daoAccessorObject.getStatus() != AccessorStatus.READY) {
370           if (daoAccessorObject.getStatus().equals(AccessorStatus.DELETED)) {
371             throw new CcsDeletedException(mesg(bucketName, objectName) + " is already Deleted");
372           }
373           throw new CcsOperationException(mesg(bucketName, objectName) + STATUS_STRING + daoAccessorObject.getStatus());
374         }
375         objectRepository.updateObjectStatus(bucketName, objectName, AccessorStatus.DELETING, null);
376       } else {
377         throw new CcsNotExistException(mesg(bucketName, objectName) + STATUS_STRING + AccessorStatus.UNKNOWN);
378       }
379       // Delete in S3
380       deleteObjectOnStorage(bucketName, objectName, daoAccessorObject);
381       // Update status in DB to Deleted
382       objectRepository.updateObjectStatus(bucketName, objectName, AccessorStatus.DELETED, null);
383     } catch (final CcsDbException e) {
384       throw new CcsOperationException("Database error on delete object : " + bucketName + " - " + objectName, e);
385     } finally {
386       flushObject();
387     }
388   }
389 
390   private void deleteObjectOnStorage(final String bucketName, final String objectName,
391                                      final DaoAccessorObject daoAccessorObject) throws CcsDbException {
392     try (final var driver = storageDriverFactory.getInstance()) {
393       driver.objectDeleteInBucket(bucketName, objectName);
394     } catch (final DriverNotFoundException e) {
395       // Ignore
396       LOGGER.debugf("Try to delete but not found: %s", daoAccessorObject);
397     } catch (final DriverException e) {
398       objectRepository.updateObjectStatus(bucketName, objectName, AccessorStatus.ERR_DEL, null);
399       throw new CcsOperationException(mesg(bucketName, objectName) + STATUS_STRING + daoAccessorObject.getStatus(), e);
400     }
401   }
