45  /**
46   * Accessor Bucket Service
47   */
48  @ApplicationScoped
49  public class AccessorObjectService {
50    private static final String STATUS_STRING = " Status: ";
51    private static final String ISSUE_STRING = " issue: ";
52    private final DriverApiFactory storageDriverFactory;
53  
54    public AccessorObjectService() {
55      this.storageDriverFactory = DriverApiRegistry.getDriverApiFactory();
56    }
57  
58    private String mesg(final String bucketName, final String objectName) {
59      return "Bucket: " + bucketName + " Object: " + objectName;
60    }
61  
62    private static AccessorObject fromStorageObject(final StorageObject storageObject) {
63      return new AccessorObject().setBucket(storageObject.bucket()).setName(storageObject.name())
64          .setSite(ServiceProperties.getAccessorSite()).setCreation(storageObject.creationDate())
65          .setSize(storageObject.size()).setHash(storageObject.hash()).setStatus(AccessorStatus.READY)
66          .setMetadata(storageObject.metadata()).setExpires(storageObject.expiresDate());
67    }
68  
69    /**
70     * Check if object or directory exists
71     */
72    public StorageType objectOrDirectoryExists(final String bucketName, final String objectOrDirectoryName)
73        throws CcsOperationException {
74      try (final var client = storageDriverFactory.getInstance()) {
75        return client.directoryOrObjectExistsInBucket(bucketName, objectOrDirectoryName);
76      } catch (final DriverException e) {
77        throw new CcsOperationException(mesg(bucketName, objectOrDirectoryName) + ISSUE_STRING + e.getMessage(), e);
78      }
79    }
80  
81    /**
82     * Utility to get Object Metadata from Driver Storage
83     */
84    private StorageObject getObjectMetadata(final String bucketName, final String objectName)
85        throws CcsNotExistException, CcsOperationException {
86      try (final var driver = storageDriverFactory.getInstance()) {
87        return driver.objectGetMetadataInBucket(bucketName, objectName);
88      } catch (final DriverNotFoundException e) {
89        throw new CcsNotExistException(mesg(bucketName, objectName) + ISSUE_STRING + e.getMessage(), e);
90      } catch (final DriverException e) {
91        throw new CcsOperationException(mesg(bucketName, objectName) + ISSUE_STRING + e.getMessage(), e);
92      }
93    }
94  
95    /**
96     * Get DB Object DTO
97     */
98    public AccessorObject getObjectInfo(final String bucketName, final String objectName)
99        throws CcsNotExistException, CcsOperationException {
100     final var storageObject = getObjectMetadata(bucketName, objectName);
101     return fromStorageObject(storageObject).setStatus(AccessorStatus.READY);
102   }

104   /**
105    * Before really creating Object, creates it in DB if possible
106    *
107    * @throws CcsNotAcceptableException if already in creation step
108    */
109   public AccessorObject createObject(final AccessorObject accessorObject, final String hash, final long len)
110       throws CcsOperationException, CcsAlreadyExistException, CcsNotExistException, CcsNotAcceptableException {
111     try {
112       // Check non-existence in S3
113       try (final var client = storageDriverFactory.getInstance()) {
114         if (!client.bucketExists(accessorObject.getBucket())) {
115           throw new CcsNotExistException(mesg(accessorObject.getBucket(), accessorObject.getName()));
116         }
117         final var type = client.directoryOrObjectExistsInBucket(accessorObject.getBucket(), accessorObject.getName());
118         if (type.equals(StorageType.NONE)) {
119           final var result = new AccessorObject();
120           result.setBucket(accessorObject.getBucket()).setName(accessorObject.getName())
121               .setExpires(accessorObject.getExpires()).setMetadata(accessorObject.getMetadata())
122               .setSite(ServiceProperties.getAccessorSite());
123           result.setStatus(AccessorStatus.UPLOAD).setHash(hash).setSize(len).setCreation(Instant.now());
124           return result;
125         }
126         throw new CcsAlreadyExistException(
127             mesg(accessorObject.getBucket(), accessorObject.getName()) + STATUS_STRING + AccessorStatus.READY);
128       }
129       // First Bucket check
130     } catch (final DriverException e) {
131       throw new CcsOperationException(
132           "Error on create object : " + accessorObject.getBucket() + " - " + accessorObject.getName(), e);
133     }
134   }
135 
136   /**
137    * Once Object really created in Driver Storage, finalize the Object in DB and Replicator if needed
138    */
139   public AccessorObject createObjectFinalize(final StorageObject storageObject, final String hash, final long len)
140       throws CcsOperationException {
141     return fromStorageObject(storageObject).setStatus(AccessorStatus.READY).setSite(ServiceProperties.getAccessorSite())
142         .setSize(len).setHash(hash);
143   }
144 
145   /**
146    * Delete object in DB and through Replicator if needed
147    */
148   public void deleteObject(final String bucketName, final String objectName)
149       throws CcsDeletedException, CcsNotExistException, CcsOperationException {
150     // Delete in S3
151     deleteObjectOnStorage(bucketName, objectName);
152   }
153 
154   private void deleteObjectOnStorage(final String bucketName, final String objectName) {
155     try (final var driver = storageDriverFactory.getInstance()) {
156       driver.objectDeleteInBucket(bucketName, objectName);
157     } catch (final DriverNotFoundException e) {
158       throw new CcsNotExistException(mesg(bucketName, objectName), e);
159     } catch (final DriverNotAcceptableException e) {
160       throw new CcsNotAcceptableException(mesg(bucketName, objectName), e);
161     } catch (final DriverException e) {
162       throw new CcsOperationException(mesg(bucketName, objectName), e);
163     }
164   }

166   private static AccessorObject filter(final AccessorObject accessorObject, final AccessorFilter filter) {
167     if (accessorObject == null || filter == null) {
168       // Status not checked
169       return accessorObject != null ? accessorObject.setStatus(AccessorStatus.READY) : null;
170     }
171     if (ParametersChecker.isNotEmpty(filter.getNamePrefix()) &&
172         !accessorObject.getName().startsWith(filter.getNamePrefix())) {
173       return null;
174     }
175     if (filter.getSizeLessThan() > 0 && accessorObject.getSize() > filter.getSizeLessThan()) {
176       return null;
177     }
178     if (filter.getSizeGreaterThan() > 0 && accessorObject.getSize() < filter.getSizeGreaterThan()) {
179       return null;
180     }
181     if (ParametersChecker.isNotEmpty(accessorObject.getCreation())) {
182       if (ParametersChecker.isNotEmpty(filter.getCreationBefore()) &&
183           filter.getCreationBefore().isBefore(accessorObject.getCreation())) {
184         return null;
185       }
186       if (ParametersChecker.isNotEmpty(filter.getCreationAfter()) &&
187           filter.getCreationAfter().isAfter(accessorObject.getCreation())) {
188         return null;
189       }
190     }
191     if (ParametersChecker.isNotEmpty(accessorObject.getExpires())) {
192       if (ParametersChecker.isNotEmpty(filter.getExpiresBefore()) &&
193           filter.getExpiresBefore().isBefore(accessorObject.getExpires())) {
194         return null;
195       }
196       if (ParametersChecker.isNotEmpty(filter.getExpiresAfter()) &&
197           filter.getExpiresAfter().isAfter(accessorObject.getExpires())) {
198         return null;
199       }
200     }
201     if (filter.getMetadataFilter() != null && accessorObject.getMetadata() != null &&
202         (!filter.getMetadataFilter().isEmpty())) {
203       final var map = accessorObject.getMetadata();
204       for (final var entry : filter.getMetadataFilter().entrySet()) {
205         if (!map.containsKey(entry.getKey())) {
206           return null;
207         }
208         if (!Objects.equals(map.get(entry.getKey()), entry.getValue())) {
209           return null;
210         }
211       }
212     }
213     // Status not checked
214     return accessorObject.setStatus(AccessorStatus.READY);
215   }

217   public InputStream filterObjects(final String bucket, final AccessorFilter filter, final DriverApi driver) {
218     try {
219       if (filter == null) {
220         final var iterator = driver.objectsIteratorInBucket(bucket);
221         return StreamIteratorUtils.getInputStreamFromIterator(iterator,
222             source -> filter(fromStorageObject((StorageObject) source), filter), AccessorObject.class);
223       }
224       final var iterator = driver.objectsIteratorInBucket(bucket, filter.getNamePrefix(), filter.getCreationAfter(),
225           filter.getCreationBefore());
226       return StreamIteratorUtils.getInputStreamFromIterator(iterator,
227           source -> filter(fromStorageObject((StorageObject) source), filter), AccessorObject.class);
228     } catch (final DriverNotFoundException e) {
229       throw new CcsNotExistException(e.getMessage(), e);
230     } catch (final DriverNotAcceptableException e) {
231       throw new CcsNotAcceptableException(e.getMessage(), e);
232     } catch (final DriverException | IOException e) {
233       throw new CcsOperationException(e);
234     }
235   }
