98    public static AccessorObject fromStorageObject(final StorageObject storageObject) {
99      return new AccessorObject().setBucket(storageObject.bucket()).setName(storageObject.name())
100         .setSite(FakeBucketServiceAbstract.site).setId(GuidLike.getGuid()).setCreation(storageObject.creationDate())
101         .setSize(storageObject.size()).setHash(storageObject.hash()).setStatus(AccessorStatus.READY)
102         .setMetadata(storageObject.metadata());
103   }

105   private static StorageObject pseudoList(final AtomicInteger cpt, final String techName, final String namePrefix) {
106     return new StorageObject(techName, namePrefix + cpt.incrementAndGet(), "hash", length, Instant.now());
107   }

109   private void computeList(final Iterator<StorageObject> stream, final UniEmitter<? super InputStream> em)
110       throws IOException {
111     final var inputStream =
112         StreamIteratorUtils.getInputStreamFromIterator(stream, source -> fromStorageObject((StorageObject) source),
113             AccessorObject.class);
114     em.complete(inputStream);
115   }

117   protected void listObjects(final UniEmitter<? super InputStream> em, final String bucketName,
118                              final String xNamePrefix, final String xCreationAfter, final String xCreationBefore,
119                              final String clientId, final boolean isPublic, final HttpServerRequest request,
120                              final Closer closer) {
121     var prefixHeader = xNamePrefix;
122     if (ParametersChecker.isEmpty(prefixHeader)) {
123       prefixHeader = "prefix";
124     }
125     final var namePrefix = ParametersChecker.getSanitizedName(prefixHeader);
126     final var techName = FakeBucketServiceAbstract.getBucketTechnicalName(clientId, bucketName, isPublic);
127     if (nbList > 0) {
128       final var cpt = new AtomicInteger(0);
129       final var iterator = Stream.generate(() -> pseudoList(cpt, techName, namePrefix)).limit(nbList).iterator();
130       try {
131         computeList(iterator, em);
132       } catch (final IOException e) {
133         em.fail(new CcsOperationException(e.getMessage()));
134       }
135     } else {
136       try (final var fakeDriver = DriverApiRegistry.getDriverApiFactory().getInstance()) {
137         Instant from = null;
138         Instant to = null;
139         if (ParametersChecker.isNotEmpty(xCreationAfter)) {
140           from = Instant.parse(xCreationAfter);
141         }
142         if (ParametersChecker.isNotEmpty(xCreationBefore)) {
143           to = Instant.parse(xCreationBefore);
144         }
145         final var iterator = fakeDriver.objectsIteratorInBucket(techName, namePrefix, from, to);
146         computeList(iterator, em);
147       } catch (final DriverNotFoundException e) {
148         em.fail(new CcsNotExistException(e.getMessage()));
149       } catch (final DriverException | IOException e) {
150         em.fail(new CcsOperationException(e.getMessage()));
151       }
152     }
153   }

183   protected void checkObjectOrDirectory(final UniEmitter<? super Response> em, final String bucketName,
184                                         final String pathDirectoryOrObject, final boolean fullCheck,
185                                         final String clientId, final boolean isPublic) {
186     final var techName = FakeBucketServiceAbstract.getBucketTechnicalName(clientId, bucketName, isPublic);
187     try (final var fakeDriver = DriverApiRegistry.getDriverApiFactory().getInstance()) {
188       final var storageType = fakeDriver.directoryOrObjectExistsInBucket(techName,
189           ParametersChecker.getSanitizedName(pathDirectoryOrObject));
190       if (storageType.equals(StorageType.NONE)) {
191         em.complete(Response.status(Response.Status.NOT_FOUND).header(X_TYPE, StorageType.NONE).build());
192       } else {
193         em.complete(Response.ok().header(X_TYPE, storageType).build());
194       }
195     } catch (final DriverException e) {
196       em.complete(Response.status(Response.Status.INTERNAL_SERVER_ERROR).build());
197     }
198   }

200   protected Uni<Response> checkObjectOrDirectory0(final String bucketName, final String pathDirectoryOrObject,
201                                                   final boolean fullCheck, final String clientId,
202                                                   final boolean isPublic) {
203     return Uni.createFrom().emitter(em -> {
204       if (errorCode > 0) {
205         if (errorCode >= 400 && errorCode != 404) {
206           em.fail(CcsServerGenericExceptionMapper.getCcsException(errorCode));
207         } else if (errorCode == 404) {
208           em.complete((Response.status(Response.Status.NOT_FOUND).header(X_TYPE, StorageType.NONE).build()));
209         } else {
210           em.complete((Response.status(Response.Status.NO_CONTENT).header(X_TYPE, StorageType.OBJECT).build()));
211         }
212       } else {
213         checkObjectOrDirectory(em, bucketName, pathDirectoryOrObject, fullCheck, clientId, isPublic);
214       }
215     });
216   }

254   protected void deleteObject(final UniEmitter<? super Response> em, final String bucketName, final String objectName,
255                               final String clientId, final boolean isPublic) {
256     final var techName = FakeBucketServiceAbstract.getBucketTechnicalName(clientId, bucketName, isPublic);
257     try (final var fakeDriver = DriverApiRegistry.getDriverApiFactory().getInstance()) {
258       fakeDriver.objectDeleteInBucket(techName, ParametersChecker.getSanitizedName(objectName));
259       final var accessorObject =
260           new AccessorObject().setBucket(techName).setName(objectName).setSite(FakeBucketServiceAbstract.site)
261               .setId(GuidLike.getGuid()).setStatus(AccessorStatus.DELETING);
262       remoteDeleteObject(accessorObject, clientId);
263       em.complete(Response.noContent().build());
264     } catch (final DriverNotFoundException e) {
265       em.complete(Response.status(Response.Status.NOT_FOUND).build());
266     } catch (final DriverNotAcceptableException e) {
267       em.complete(Response.status(Response.Status.NOT_ACCEPTABLE).build());
268     } catch (final DriverException e) {
269       em.complete(Response.status(Response.Status.INTERNAL_SERVER_ERROR).build());
270     }
271   }

297                                             ParameterIn.HEADER, schema = @Schema(type = SchemaType.STRING), required
298                                             = false) @DefaultValue("") @HeaderParam(X_OP_ID) final String opId,
299                                         @DefaultValue("") @HeaderParam(X_OBJECT_ID) final String xObjectId,
300                                         @DefaultValue("") @HeaderParam(X_OBJECT_SITE) final String xObjectSite,
301                                         @DefaultValue("") @HeaderParam(X_OBJECT_BUCKET) final String xObjectBucket,
302                                         @DefaultValue("") @HeaderParam(X_OBJECT_NAME) final String xObjectName,
303                                         @DefaultValue("0") @HeaderParam(X_OBJECT_SIZE) final long xObjectSize,
304                                         @DefaultValue("") @HeaderParam(X_OBJECT_HASH) final String xObjectHash,
305                                         @DefaultValue("") @HeaderParam(X_OBJECT_METADATA) final String xObjectMetadata,
306                                         @DefaultValue("") @HeaderParam(X_OBJECT_EXPIRES) final String xObjectExpires,
307                                         final boolean isPublic, final InputStream inputStream) {
308     final var techName = FakeBucketServiceAbstract.getBucketTechnicalName(clientId, bucketName, isPublic);
309     var storeLength = xObjectSize;
310     if (storeLength <= 0) {
311       var headerLength = httpHeaders.getHeaderString(CONTENT_LENGTH);
312       if (ParametersChecker.isNotEmpty(headerLength)) {
313         storeLength = Long.parseLong(headerLength);
314       }
315     }
316     final var accessorObject = new AccessorObject();
317     AccessorHeaderDtoConverter.objectFromMap(accessorObject, request.headers());
318     accessorObject.setName(ParametersChecker.getSanitizedName(objectName)).setSite(FakeBucketServiceAbstract.site)
319         .setBucket(techName).setSize(storeLength);
320     // TODO choose compression model
321     return createObject(request, closer, accessorObject, accessorObject.getSize(), accessorObject.getHash(), false,
322         inputStream);
323   }
