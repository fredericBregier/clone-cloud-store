17  package io.clonecloudstore.accessor.client.internal.resource.fakeserver;
18  
19  import java.time.Instant;
20  import java.util.Collection;
21  
22  import io.clonecloudstore.accessor.client.internal.api.AccessorBucketInternalApi;
23  import io.clonecloudstore.accessor.config.AccessorConstants;
24  import io.clonecloudstore.accessor.model.AccessorBucket;
25  import io.clonecloudstore.accessor.model.AccessorStatus;
26  import io.clonecloudstore.common.quarkus.exception.CcsServerGenericExceptionMapper;
27  import io.clonecloudstore.common.standard.system.SingletonUtils;
28  import io.clonecloudstore.driver.api.StorageType;
29  import io.smallrye.mutiny.Uni;
30  import jakarta.ws.rs.Path;
31  import jakarta.ws.rs.core.Response;
32  
33  @Path(AccessorConstants.Api.INTERNAL_ROOT)
34  public class FakeBucketInternalService implements AccessorBucketInternalApi {
35    public static int errorCode = 0;
36  
37    @Override
38    public Uni<Collection<AccessorBucket>> getBuckets(final String clientId, final String opId) {
39      return Uni.createFrom().emitter(em -> {
40        if (errorCode > 0) {
41          throw CcsServerGenericExceptionMapper.getCcsException(errorCode);
42        }
43        em.complete(SingletonUtils.singletonList());
44      });
45    }
46  
47    @Override
48    public Uni<Response> checkBucket(final String bucketName, final boolean fullCheck, final String clientId,
49                                     final String opId) {
50      return Uni.createFrom().emitter(em -> {
51        if (errorCode > 0) {
52          if (errorCode != 404) {
53            throw CcsServerGenericExceptionMapper.getCcsException(errorCode);
54          }
55          em.complete((Response.status(Response.Status.NOT_FOUND).header(AccessorConstants.Api.X_TYPE, StorageType.NONE)
56              .build()));
57          return;
58        }
59        em.complete(Response.ok().header(AccessorConstants.Api.X_TYPE, StorageType.BUCKET).build());
60      });
61    }
62  
63    @Override
64    public Uni<AccessorBucket> getBucket(final String bucketName, final String clientId, final String opId) {
65      return Uni.createFrom().emitter(em -> {
66        if (errorCode > 0) {
67          throw CcsServerGenericExceptionMapper.getCcsException(errorCode);
68        }
69        final var accessorBucket =
70            new AccessorBucket().setName(bucketName).setSite("site").setStatus(AccessorStatus.READY)
71                .setCreation(Instant.now()).setId(clientId + '-' + bucketName);
72        em.complete(accessorBucket);
73      });
74    }
75  
76    @Override
77    public void close() {
78  
79    }
80  }
