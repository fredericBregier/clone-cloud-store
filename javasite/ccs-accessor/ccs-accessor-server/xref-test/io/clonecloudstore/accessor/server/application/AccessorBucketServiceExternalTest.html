134   private long getBucketCreateFromTopic(final long desired) throws InterruptedException {
135     for (int i = 0; i < 200; i++) {
136       final var value = FakeActionTopicConsumer.getBucketCreate();
137       if (value >= desired) {
138         Log.infof("Found %d", value);
139         return value;
140       }
141       Thread.sleep(10);
142     }
143     return FakeActionTopicConsumer.getBucketCreate();
144   }

146   @Test
147   void getBucket() throws CcsDbException {
148 
149     final var unknownBucketTechnicalName =
150         DaoAccessorBucketRepository.getBucketTechnicalName(clientId, "unknownbucket");
151     //Get Bucket not exist
152     final Exception exceptionMin = assertThrows(CcsNotExistException.class,
153         () -> service.getBucket(unknownBucketTechnicalName, true, clientId, GuidLike.getGuid()));
154     assertNotNull(exceptionMin);
155     assertFalse(service.checkBucket(unknownBucketTechnicalName, false, true, clientId, GuidLike.getGuid()));
156     // Check remote
157     assertTrue(AccessorProperties.isRemoteRead());
158     assertTrue(AccessorProperties.isFixOnAbsent());
159     try {
160       FakeLocalReplicatorBucketServiceImpl.errorCode = 204;
161       assertTrue(service.checkBucket(unknownBucketTechnicalName, false, true, clientId, GuidLike.getGuid()));
162       assertEquals(0, FakeActionTopicConsumer.getBucketCreate());
163       FakeLocalReplicatorBucketServiceImpl.errorCode = 200;
164       final var bucket = service.getBucket(unknownBucketTechnicalName, true, clientId, GuidLike.getGuid());
165       assertEquals(unknownBucketTechnicalName, bucket.getId());
166       assertEquals(1, getBucketCreateFromTopic(1));
167       // Only getBucket try to fix from the remote, so increment value
168       service.getBucket(unknownBucketTechnicalName, true, clientId, GuidLike.getGuid());
169       assertEquals(2, getBucketCreateFromTopic(2));
170     } catch (InterruptedException e) {
171       throw new RuntimeException(e);
172     } finally {
173       FakeLocalReplicatorBucketServiceImpl.errorCode = 0;
174       FakeLocalReplicatorObjectServiceImpl.errorCode = 0;
175     }
176     // Same but fix if absent is off
177     assertTrue(AccessorProperties.isRemoteRead());
178     AccessorProperties.setFixOnAbsent(false);
179     assertFalse(AccessorProperties.isFixOnAbsent());
180     try {
181       FakeLocalReplicatorBucketServiceImpl.errorCode = 204;
182       assertTrue(service.checkBucket(unknownBucketTechnicalName, false, true, clientId, GuidLike.getGuid()));
183       assertEquals(2, FakeActionTopicConsumer.getBucketCreate());
184       FakeLocalReplicatorBucketServiceImpl.errorCode = 200;
185       assertThrows(CcsNotExistException.class,
186           () -> service.getBucket(unknownBucketTechnicalName, true, clientId, GuidLike.getGuid()));
187       Thread.sleep(100);
188       assertEquals(2, getBucketCreateFromTopic(2));
189     } catch (InterruptedException e) {
190       throw new RuntimeException(e);
191     } finally {
192       FakeLocalReplicatorBucketServiceImpl.errorCode = 0;
193       FakeLocalReplicatorObjectServiceImpl.errorCode = 0;
194       AccessorProperties.setFixOnAbsent(true);
195     }
196     // Same but remote is off
197     AccessorProperties.setRemoteRead(false);
198     assertFalse(AccessorProperties.isRemoteRead());
199     assertTrue(AccessorProperties.isFixOnAbsent());
200     try {
201       FakeLocalReplicatorBucketServiceImpl.errorCode = 204;
202       assertFalse(service.checkBucket(unknownBucketTechnicalName, false, true, clientId, GuidLike.getGuid()));
203       assertEquals(2, FakeActionTopicConsumer.getBucketCreate());
204       FakeLocalReplicatorBucketServiceImpl.errorCode = 200;
205       assertThrows(CcsNotExistException.class,
206           () -> service.getBucket(unknownBucketTechnicalName, true, clientId, GuidLike.getGuid()));
207       Thread.sleep(100);
208       assertEquals(2, getBucketCreateFromTopic(2));
209     } catch (InterruptedException e) {
210       throw new RuntimeException(e);
211     } finally {
212       FakeLocalReplicatorBucketServiceImpl.errorCode = 0;
213       FakeLocalReplicatorObjectServiceImpl.errorCode = 0;
214       AccessorProperties.setRemoteRead(true);
215     }
216 
217     //Create a bucket and get Bucket
218     final var bucketName = "testgetbucket";
219     final var opId = GuidLike.getGuid();
220     final var bucketTechnicalName = DaoAccessorBucketRepository.getBucketTechnicalName(clientId, bucketName);
221     final AccessorBucket bucket;
222     service.createBucket(clientId, bucketTechnicalName, true);
223     bucket = service.getBucket(bucketTechnicalName, true, clientId, opId);
224     assertEquals(bucketName, bucket.getName());
225     assertEquals(AccessorStatus.READY, bucket.getStatus());
226     assertTrue(service.checkBucket(bucketTechnicalName, false, true, clientId, GuidLike.getGuid()));
227 
228     //Check read bucket deleted
229     service.deleteBucket(clientId, bucketTechnicalName, true);
230 
231     assertThrows(CcsDeletedException.class, () -> service.getBucket(bucketTechnicalName, true, clientId, opId));
232     assertFalse(service.checkBucket(bucketTechnicalName, false, true, clientId, GuidLike.getGuid()));
233 
234     final var bucketNameStatusNotA = "bucketnotavailable";
235     final var technicalBucketNameNotA = DaoAccessorBucketRepository.getPrefix(clientId) + bucketNameStatusNotA;
236     final AccessorBucket bucketStatusNotAvailable;
237     bucketStatusNotAvailable = service.createBucket(clientId, technicalBucketNameNotA, true);
238     assertEquals(bucketNameStatusNotA, bucketStatusNotAvailable.getName());
239     assertEquals(AccessorStatus.READY, bucketStatusNotAvailable.getStatus());
240     assertTrue(service.checkBucket(technicalBucketNameNotA, false, true, clientId, GuidLike.getGuid()));
241 
242     //Change status to simulate an operation
243     final var bucketTmp = bucketRepository.findBucketById(technicalBucketNameNotA);
244     bucketRepository.updateBucketStatus(bucketTmp, AccessorStatus.DELETING, null);
245 
246     assertThrows(CcsOperationException.class, () -> service.getBucket(technicalBucketNameNotA, true, clientId, opId));
247     assertFalse(service.checkBucket(technicalBucketNameNotA, false, true, clientId, GuidLike.getGuid()));
248   }

267   @Test
268   void deleteBucket() throws CcsDbException {
269     final var bucketName = "bucketdeletedtest";
270     final AccessorBucket bucket;
271     {
272       final var bucketTechnicalName = DaoAccessorBucketRepository.getBucketTechnicalName(clientId, bucketName);
273       bucket = service.createBucket(clientId, bucketTechnicalName, true);
274       assertEquals(bucketName, bucket.getName());
275       assertEquals(bucketTechnicalName, bucket.getId());
276       assertEquals(AccessorStatus.READY, bucket.getStatus());
277     }
278     //Delete bucket already
279     {
280       final var bucketTechnicalName = DaoAccessorBucketRepository.getBucketTechnicalName(clientId, bucketName);
281       final var bucketDeleted = service.deleteBucket(clientId, bucketTechnicalName, true);
282       Assertions.assertEquals(AccessorStatus.DELETED, bucketDeleted.getStatus());
283       final Exception exceptionDeleted =
284           assertThrows(CcsDeletedException.class, () -> service.deleteBucket(clientId, bucketTechnicalName, true));
285       assertNotNull(exceptionDeleted);
286       //Delete bucket already deleted
287       try (final var storageDriver = storageDriverFactory.getInstance()) {
288         // retrieving metadata from storage, would data be duplicated in database
289         final Exception exceptionDeletedStorage =
290             assertThrows(DriverNotFoundException.class, () -> storageDriver.bucketDelete(bucketDeleted.getId()));
291         assertNotNull(exceptionDeletedStorage);
292       }
293     }
294     //Delete unknown Bucket
295     final var unknownBucketTechnicalName =
296         DaoAccessorBucketRepository.getBucketTechnicalName(clientId, "unknownBucket");
297     final Exception exceptionMin = assertThrows(CcsNotExistException.class,
298         () -> service.deleteBucket(clientId, unknownBucketTechnicalName, true));
299     assertNotNull(exceptionMin);
300 
301     //Delete bucket not AVAILABLE or DELETED status
302     final var bucketNameStatusNotAD = "bucketnotavailablerdeleted";
303     final var bucketTechnicalNameNotAD =
304         DaoAccessorBucketRepository.getBucketTechnicalName(clientId, bucketNameStatusNotAD);
305 
306     final AccessorBucket bucketStatusNotAD;
307     bucketStatusNotAD = service.createBucket(clientId, bucketTechnicalNameNotAD, true);
308     assertEquals(bucketNameStatusNotAD, bucketStatusNotAD.getName());
309     assertEquals(AccessorStatus.READY, bucketStatusNotAD.getStatus());
310 
311     final var technicalBucketName = DaoAccessorBucketRepository.getBucketTechnicalName(clientId, bucketName);
312     //Change status to simulate an operation
313     final var bucketTmp = bucketRepository.findBucketById(technicalBucketName);
314     bucketRepository.updateBucketStatus(bucketTmp, AccessorStatus.DELETING, null);
315     final var bucketTechnicalName = DaoAccessorBucketRepository.getBucketTechnicalName(clientId, bucketName);
316     final Exception exceptionStatusAD =
317         assertThrows(CcsOperationException.class, () -> service.deleteBucket(clientId, bucketTechnicalName, true));
318     assertNotNull(exceptionStatusAD);
319   }
