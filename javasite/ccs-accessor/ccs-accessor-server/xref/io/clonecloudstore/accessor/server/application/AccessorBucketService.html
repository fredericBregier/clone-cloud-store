64    public AccessorBucketService(final LocalReplicatorService localReplicatorService,
65                                 final Instance<DaoAccessorBucketRepository> bucketRepositoryInstance,
66                                 final Instance<DaoAccessorObjectRepository> objectRepositoryInstance) {
67      this.localReplicatorService = localReplicatorService;
68      // Normal injection does not work, probably due to test only dependency
69      this.bucketRepository = bucketRepositoryInstance.get();
70      this.objectRepository = objectRepositoryInstance.get();
71      this.storageDriverFactory = DriverApiRegistry.getDriverApiFactory();

74    /**
75     * Create bucket from client Id and technicalBucketName
76     *
77     * @param clientId            Format client ID use to identify client
78     * @param technicalBucketName Technical Bucket Name
79     * @param isPublic            True means replicate with replicator module
80     * @return AccessorBucket add on Database and in object storage
81     */
82    public AccessorBucket createBucket(final String clientId, final String technicalBucketName, final boolean isPublic)
83        throws CcsAlreadyExistException, CcsOperationException {
84      final var bucketName = DaoAccessorBucketRepository.getBucketName(clientId, technicalBucketName);
85      try {
86        //Check format (special char, uppercase...) and size (min, max) for bucket name and technicalBucketName
87        ParametersChecker.checkSanityBucketName(bucketName);
88        ParametersChecker.checkSanityBucketName(technicalBucketName);
89      } catch (final CcsInvalidArgumentRuntimeException e) {
90        final var message = String.format("Bucket Name %s is invalid", technicalBucketName);
91        throw new CcsOperationException(message);
92      }
93  
94      //Check if bucket already exist in database.
95      AccessorBucket result;
96      try {
97        result = bucketRepository.findBucketById(technicalBucketName);
98      } catch (final CcsDbException e) {
99        throw new CcsOperationException("Error on find repository with technical ID", e);
100     }
101     if (result != null && !AccessorStatus.DELETED.equals(result.getStatus())) {
102       final var message = String.format("Bucket %s already exist", technicalBucketName);
103       LOGGER.errorf(message);
104       throw new CcsAlreadyExistException(message);
105     } else if (result == null) {
106       //Create Bucket in Database with Status In Progress.
107       //Only name and technical name (over value are erased during creation).
108       result = new AccessorBucket();
109       result.setName(bucketName);
110       result.setId(technicalBucketName);
111       result.setSite(ServiceProperties.getAccessorSite());
112     }
113     final AccessorBucket newBucket = createBucketFromDto(result);
114     if (isPublic) {
115       //Replicate message with replicator (send message to broker).
116       LOGGER.infof("Replicate Create Order for %s", newBucket.getId());
117       localReplicatorService.create(newBucket.getId(), clientId);
118     }
119     return newBucket;
120   }

122   /**
123    * Create Bucket from DTO Accessor
124    *
125    * @param bucket Accessor Bucket with functional data
126    * @return AccessorBucket add on Database and in object storage
127    */
128   private AccessorBucket createBucketFromDto(final AccessorBucket bucket) throws CcsOperationException {
129 
130     LOGGER.infof("Create bucket %s in database and in object storage", bucket.getId());
131     try {
132       //Insert in Database
133       // Since it can be in DELETED status, it might be updated (creation + status)
134       bucket.setCreation(Instant.now());
135       AccessorBucket newBucket = bucket;
136       if (AccessorStatus.DELETED.equals(newBucket.getStatus())) {
137         newBucket.setStatus(AccessorStatus.UPLOAD);
138         bucketRepository.updateBucketStatus(newBucket, AccessorStatus.UPLOAD, newBucket.getCreation());
139       } else {
140         newBucket.setStatus(AccessorStatus.UPLOAD);
141         newBucket = bucketRepository.insertBucket(newBucket);
142       }
143 
144       //Create Bucket in Object Storage
145       final var storageBucket = new StorageBucket(newBucket.getId(), newBucket.getCreation());
146       return createBucketOnStorage(storageBucket, newBucket);
147     } catch (final CcsDbException e) {
148       throw new CcsOperationException("Database error on bucket " + bucket.getName() + " creation", e);
149     }
150   }
151 
152   private AccessorBucket createBucketOnStorage(final StorageBucket storageBucket, final AccessorBucket bucket)
153       throws CcsDbException {
154     try (final var storageDriver = storageDriverFactory.getInstance()) {
155       // retrieving metadata from storage, would data be duplicated in database
156       storageDriver.bucketCreate(storageBucket);
157 
158       //Update Database in Database with status Available.
159       return bucketRepository.updateBucketStatus(bucket, AccessorStatus.READY, null);
160     } catch (final CcsInvalidArgumentRuntimeException | DriverException | CcsServerGenericException e) {
161       LOGGER.error(e.getMessage());
162       bucketRepository.updateBucketStatus(bucket, AccessorStatus.ERR_UPL, null);
163       throw new CcsOperationException(e.getMessage(), e);
164     }
165   }

167   /**
168    * Get Bucket information from bucket technical name
169    *
170    * @param technicalBucketName Bucket technical name
171    * @return AccessorBucket found with technical name
172    */
173   public AccessorBucket getBucket(final String technicalBucketName, final boolean external, final String clientId,
174                                   final String opId)
175       throws CcsNotExistException, CcsDeletedException, CcsOperationException {
176     try {
177       //Search bucket in database with ID.
178       final var bucket = bucketRepository.findBucketById(technicalBucketName);
179       if (bucket != null) {
180         if (bucket.getStatus() == AccessorStatus.READY) {
181           return bucket;
182         } else if (bucket.getStatus() == AccessorStatus.DELETED) {
183           throw new CcsDeletedException(BUCKET_STRING + technicalBucketName + "is deleted");
184         } else {
185           throw new CcsOperationException(BUCKET_STRING + technicalBucketName + " status " + bucket.getStatus());
186         }
187       } else {
188         if (external && AccessorProperties.isRemoteRead()) {
189           // Use remote check
190           AccessorBucket response = getRemoteAccessorBucket(technicalBucketName, clientId, opId);
191           if (response != null) {
192             return response;
193           }
194         }
195         throw new CcsNotExistException(BUCKET_STRING + technicalBucketName + " doesn't exist");
196       }
197     } catch (final CcsDbException e) {
198       throw new CcsOperationException("Database error on get bucket : " + technicalBucketName, e);
199     }
200   }
201 
202   private AccessorBucket getRemoteAccessorBucket(final String technicalBucketName, final String clientId,
203                                                  final String opId) {
204     try {
205       final var response = localReplicatorService.remoteGetBucket(technicalBucketName, clientId, opId);
206       if (response.response() != null && response.targetId() != null && AccessorProperties.isFixOnAbsent()) {
207         localReplicatorService.generateLocalReplicationOrder(
208             new ReplicatorOrder(opId, response.targetId(), ServiceProperties.getAccessorSite(), clientId,
209                 technicalBucketName, ReplicatorConstants.Action.CREATE));
210         return response.response();
211       }
212     } catch (final CcsOperationException ignore) {
213       // Ignored
214     }
215     return null;
216   }

232   /**
233    * Check if Bucket exists
234    *
235    * @param technicalBucketName Bucket technical name
236    * @param fullCheck           True to check Object Storage
237    * @param external            True to check remotely if not found locally
238    * @param clientId            the clientId
239    * @param opId                the OperationId
240    * @return True if it exists
241    */
242   public boolean checkBucket(final String technicalBucketName, final boolean fullCheck, final boolean external,
243                              final String clientId, final String opId) throws CcsOperationException {
244     try {
245       // Could use simply COUNT
246       final var bucket = bucketRepository.findBucketById(technicalBucketName);
247       if (bucket == null && external && AccessorProperties.isRemoteRead()) {
248         // Use remote check
249         return remoteCheckBucket(technicalBucketName, clientId, opId);
250       }
251       if (bucket == null || bucket.getStatus() != AccessorStatus.READY) {
252         return false;
253       }
254       if (fullCheck) {
255         return checkBucketOnStorage(technicalBucketName);
256       }
257       return true;
258     } catch (final CcsDbException e) {
259       throw new CcsOperationException("Database error on check bucket : " + technicalBucketName, e);
260     }
261   }
262 
263   private boolean remoteCheckBucket(final String technicalBucketName, final String clientId, final String opId) {
264     try {
265       return localReplicatorService.remoteCheckBucket(technicalBucketName, clientId, opId);
266     } catch (final CcsOperationException ignore) {
267       return false;
268     }
269   }

271   private boolean checkBucketOnStorage(final String technicalBucketName) {
272     try (final var storageDriver = storageDriverFactory.getInstance()) {
273       return storageDriver.bucketExists(technicalBucketName);
274     } catch (final DriverException e) {
275       throw new CcsOperationException("Issue while checking bucket : " + technicalBucketName, e);
276     }
277   }

279   /**
280    * Delete bucket from technical bucket name
281    *
282    * @param clientId            Client ID used to identify client
283    * @param technicalBucketName Bucket technical name
284    * @param external            true to send replication message on replicator
285    * @return the associated DTO- deleted
286    */
287   public AccessorBucket deleteBucket(final String clientId, final String technicalBucketName, final boolean external)
288       throws CcsNotExistException, CcsDeletedException, CcsOperationException, CcsNotAcceptableException {
289     try {
290       var bucket = bucketRepository.findBucketById(technicalBucketName);
291       if (bucket == null) {
292         throw new CcsNotExistException(BUCKET_STRING + technicalBucketName + " doesn't exist");
293       } else if (bucket.getStatus() == AccessorStatus.READY) {
294         final var count = objectRepository.count(
295             new DbQuery(RestQuery.CONJUNCTION.AND, new DbQuery(RestQuery.QUERY.EQ, BUCKET, bucket.getId()),
296                 new DbQuery(RestQuery.QUERY.EQ, STATUS, AccessorStatus.READY)));
297         if (count > 0) {
298           throw new CcsNotAcceptableException("Bucket is not empty");
299         }
300         bucketRepository.updateBucketStatus(bucket, AccessorStatus.DELETING, null);
301         final var deleted = updateAccessorBucketDelete(technicalBucketName, bucket);
302         if (external) {
303           LOGGER.infof("Replicate Delete Order for %s ", bucket.getId());
304           localReplicatorService.delete(bucket.getId(), clientId);
305         }
306         return deleted;
307       } else if (bucket.getStatus() == AccessorStatus.DELETED) {
308         throw new CcsDeletedException(BUCKET_STRING + technicalBucketName + " is already deleted");
309       } else {
310         throw new CcsOperationException("Bucket Status " + bucket.getStatus());
311       }
312     } catch (final CcsDbException e) {
313       throw new CcsOperationException("Database error on delete bucket " + technicalBucketName, e);
314     }
315   }
316 
317   private AccessorBucket updateAccessorBucketDelete(final String technicalBucketName, final AccessorBucket bucket)
318       throws CcsDbException {
319     try {
320       try (final var storageDriver = storageDriverFactory.getInstance()) {
321         // retrieving metadata from storage, would data be duplicated in database
322         storageDriver.bucketDelete(technicalBucketName);
323       }
324       return bucketRepository.updateBucketStatus(bucket, AccessorStatus.DELETED, null);
325     } catch (final DriverException | RuntimeException e) {
326       LOGGER.error(e.getMessage());
327       bucketRepository.updateBucketStatus(bucket, AccessorStatus.ERR_DEL, null);
328       throw new CcsOperationException(e.getMessage(), e);
329     }
330   }
