59    @Test
60    void testBucket() {
61      final var accessorBucket =
62          new AccessorBucket().setSite(AccessorProperties.getAccessorSite()).setCreation(Instant.now())
63              .setName(DaoAccessorBucketRepository.getBucketName(clientId, bucket))
64              .setId(DaoAccessorBucketRepository.getBucketTechnicalName(clientId, bucket))
65              .setStatus(AccessorStatus.READY);
66      final DaoAccessorBucket daoAccessorBucket = new TestDaoAccessorBucket();
67      daoAccessorBucket.fromDto(accessorBucket);
68      final var second = daoAccessorBucket.getDto();
69      Assertions.assertEquals(accessorBucket, second);
70      assertEquals(accessorBucket, accessorBucket);
71      final var secondDao = new TestDaoAccessorBucket();
72      secondDao.fromDto(second);
73      assertEquals(daoAccessorBucket, secondDao);
74      assertEquals(daoAccessorBucket.hashCode(), secondDao.hashCode());
75      assertEquals(DaoAccessorBucketRepository.getBucketName(clientId, secondDao.getId()), secondDao.getName());
76      assertEquals(DaoAccessorBucketRepository.getBucketName(clientId, secondDao.getName()), secondDao.getName());
77      assertEquals(DaoAccessorBucketRepository.getBucketTechnicalName(clientId, secondDao.getName()), secondDao.getId());
78      assertEquals(DaoAccessorBucketRepository.getPrefix(clientId) + secondDao.getName(), secondDao.getId());
79      assertEquals(DaoAccessorBucketRepository.getRealBucketName(clientId, secondDao.getId(), false), secondDao.getId());
80      assertEquals(DaoAccessorBucketRepository.getRealBucketName(clientId, secondDao.getName(), true), secondDao.getId());
81      assertTrue(daoAccessorBucket.toString().contains(AccessorProperties.getAccessorSite()));
82    }

171   @Test
172   void testObject() {
173     final var accessorObject =
174         new AccessorObject().setSite(AccessorProperties.getAccessorSite()).setCreation(Instant.now())
175             .setBucket(DaoAccessorBucketRepository.getBucketTechnicalName(clientId, bucket)).setName(object)
176             .setId(GuidLike.getGuid()).setStatus(AccessorStatus.READY);
177     final DaoAccessorObject daoAccessorObject = new TestDaoAccessorObject();
178     daoAccessorObject.fromDto(accessorObject);
179     final var second = daoAccessorObject.getDto();
180     Assertions.assertEquals(accessorObject, second);
181     final var secondDao = new TestDaoAccessorObject();
182     secondDao.fromDto(second);
183     assertEquals(daoAccessorObject, secondDao);
184     assertEquals(daoAccessorObject.hashCode(), secondDao.hashCode());
185     assertEquals(object, ParametersChecker.getSanitizedName(object));
186     assertEquals(object, ParametersChecker.getSanitizedName("/" + object));
187     assertEquals(object, ParametersChecker.getSanitizedName("//" + object));
188     assertEquals(object, ParametersChecker.getSanitizedName("\\\\" + object));
189     assertTrue(daoAccessorObject.toString().contains(AccessorProperties.getAccessorSite()));
190   }
