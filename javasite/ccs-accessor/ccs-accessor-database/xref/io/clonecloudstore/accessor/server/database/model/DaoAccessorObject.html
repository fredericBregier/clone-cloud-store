50  /**
51   * Object DAO
52   */
53  @MappedSuperclass
54  public abstract class DaoAccessorObject {
55    @Column(name = SITE, nullable = false, length = SITE_LENGTH)
56    private String site;
57    @Column(name = BUCKET, nullable = false, length = BUCKET_LENGTH)
58    private String bucket;
59    @Column(name = NAME, nullable = false, length = OBJECT_LENGTH)
60    private String name;
61    @Column(name = HASH, nullable = false, length = HASH_LENGTH)
62    private String hash;
63    @Column(name = STATUS, nullable = false, length = STATUS_LENGTH)
64    private AccessorStatus status;
65    @Column(name = CREATION, nullable = false)
66    private Instant creation;
67    @Column(name = EXPIRES)
68    private Instant expires;
69    @Column(name = SIZE)
70    private long size;
71    /**
72     * Reconciliation status is not used elsewhere than Reconciliation
73     */
74    @Column(name = RSTATUS)
75    private short rstatus;
76  
77    public abstract String getId();
78  
79    public abstract DaoAccessorObject setId(String id);
80  
81    @Transient
82    @JsonIgnore
83    public DaoAccessorObject fromDto(final AccessorObject dto) {
84      this.setId(dto.getId()).setSite(dto.getSite()).setBucket(dto.getBucket()).setName(dto.getName())
85          .setHash(dto.getHash()).setStatus(dto.getStatus()).setCreation(dto.getCreation()).setExpires(dto.getExpires())
86          .setSize(dto.getSize()).setMetadata(dto.getMetadata());
87      return this;
88    }
89  
90    /**
91     * Ignore Id and Site
92     *
93     * @return True if updated locally
94     */
95    @Transient
96    @JsonIgnore
97    public boolean updateFromDtoExceptIdSite(final AccessorObject dto) {
98      if (Objects.equals(dto.getBucket(), this.getBucket()) && Objects.equals(dto.getName(), this.getName()) &&
99          Objects.equals(dto.getHash(), this.getHash()) && Objects.equals(dto.getStatus(), this.getStatus()) &&
100         Objects.equals(dto.getCreation(), this.getCreation()) && Objects.equals(dto.getExpires(), this.getExpires()) &&
101         Objects.equals(dto.getSize(), this.getSize()) && Objects.equals(dto.getMetadata(), this.getMetadata())) {
102       return false;
103     }
104     this.setBucket(dto.getBucket()).setName(dto.getName()).setHash(dto.getHash()).setStatus(dto.getStatus())
105         .setCreation(dto.getCreation()).setExpires(dto.getExpires()).setSize(dto.getSize())
106         .setMetadata(dto.getMetadata());
107     return true;
108   }
109 
110   @Transient
111   @JsonIgnore
112   public AccessorObject getDto() {
113     return new AccessorObject().setId(this.getId()).setSite(this.getSite()).setBucket(this.getBucket())
114         .setName(this.getName()).setHash(this.getHash()).setStatus(this.getStatus()).setCreation(this.getCreation())
115         .setExpires(this.getExpires()).setSize(this.getSize()).setMetadata(this.getMetadata());
116   }
117 
118   public String getSite() {
119     return site;
120   }
121 
122   public DaoAccessorObject setSite(final String site) {
123     this.site = site;
124     return this;
125   }
126 
127   public String getBucket() {
128     return bucket;
129   }
130 
131   public DaoAccessorObject setBucket(final String bucket) {
132     ParametersChecker.checkSanityBucketName(bucket);
133     this.bucket = bucket;
134     return this;
135   }
136 
137   public String getName() {
138     return name;
139   }
140 
141   public DaoAccessorObject setName(final String name) {
142     ParametersChecker.checkSanityObjectName(name);
143     this.name = name;
144     return this;
145   }
146 
147   public String getHash() {
148     return hash;
149   }
150 
151   public DaoAccessorObject setHash(final String hash) {
152     ParametersChecker.checkSanityString(hash);
153     this.hash = hash;
154     return this;
155   }
156 
157   public AccessorStatus getStatus() {
158     return status;
159   }
160 
161   public DaoAccessorObject setStatus(final AccessorStatus status) {
162     ParametersChecker.checkParameterNullOnly("Status cannot be null", status);
163     this.status = status;
164     return this;
165   }
166 
167   public Instant getCreation() {
168     return creation;
169   }
170 
171   public DaoAccessorObject setCreation(final Instant creation) {
172     this.creation = SystemTools.toMillis(creation);
173     return this;
174   }
175 
176   public Instant getExpires() {
177     return expires;
178   }
179 
180   public DaoAccessorObject setExpires(final Instant expires) {
181     this.expires = SystemTools.toMillis(expires);
182     return this;
183   }
184 
185   public long getSize() {
186     return size;
187   }
188 
189   public DaoAccessorObject setSize(final long size) {
190     this.size = size;
191     return this;
192   }
193 
194   public abstract Map<String, String> getMetadata();
195 
196   @Transient
197   @JsonIgnore
198   public abstract String getMetadata(String key);
199 
200   @Transient
201   @JsonIgnore
202   public abstract DaoAccessorObject addMetadata(String key, String value);
203 
204   public abstract DaoAccessorObject setMetadata(Map<String, String> metadata);
205 
206   public short getRstatus() {
207     return rstatus;
208   }
209 
210   public DaoAccessorObject setRstatus(final short rstatus) {
211     this.rstatus = rstatus;
212     return this;
213   }
214 
215   @Override
216   public boolean equals(final Object o) {
217     if (this == o) {
218       return true;
219     }
220     if (o instanceof DaoAccessorObject that) {
221       return size == that.size && Objects.equals(getId(), that.getId()) && Objects.equals(site, that.site) &&
222           Objects.equals(bucket, that.bucket) && Objects.equals(name, that.name) && Objects.equals(hash, that.hash) &&
223           Objects.equals(status, that.status) && Objects.equals(creation, that.creation) &&
224           Objects.equals(expires, that.expires) && Objects.deepEquals(getMetadata(), that.getMetadata());
225     }
226     return false;
227   }
228 
229   @Override
230   public int hashCode() {
231     return Objects.hash(getId(), site, bucket, name, hash, status, creation, expires, size, getMetadata());
232   }
233 
234   @Override
235   public String toString() {
236     try {
237       return JsonUtil.getInstance().writeValueAsString(this);
238     } catch (final JsonProcessingException e) {
239       return "{" + "\"id\":\"" + getId() + "\"" + ", \"site\":\"" + site + "\"" + ", \"bucket\":\"" + bucket + "\"" +
240           ", \"name\":\"" + name + "\"" + ", \"hash\":\"" + hash + "\"" + ", \"status\":\"" + status + "\"" +
241           ", \"rstatus\":" + rstatus + ", \"creation\":" + creation + ", \"expires\":" + expires + ", \"size\":\"" +
242           size + "\"" + ", \"metadata\": {" +
243           getMetadata().entrySet().stream().map(d -> "\"%s\": \"%s\",".formatted(d.getKey(), d.getValue()))
244               .collect(Collectors.joining()) + "}}";
245     }
246   }
