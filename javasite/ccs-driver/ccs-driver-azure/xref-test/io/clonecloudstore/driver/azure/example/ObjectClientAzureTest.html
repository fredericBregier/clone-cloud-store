37  
38  import static io.clonecloudstore.driver.azure.DriverAzureProperties.DEFAULT_MAX_SIZE_NOT_PART;
39  import static org.junit.jupiter.api.Assertions.fail;
40  
41  @TestMethodOrder(MethodOrderer.MethodName.class)
42  @QuarkusTest
43  @TestProfile(AzureProfile.class)
44  class ObjectClientAzureTest extends ObjectClientAzureBase {
45  
46    @BeforeAll
47    static void setup() throws InterruptedException {
48      // Sometime, test failed on DriverApiFactory not setup
49      if (DriverApiRegistry.getDriverApiFactory() == null) {
50        // Ensure container is started
51        Thread.sleep(500);
52        DriverApiRegistry.setDriverApiFactory(CDI.current().select(DriverAzureApiFactory.class).get());
53      }
54      old = QuarkusProperties.serverComputeSha256();
55      QuarkusProperties.setServerComputeSha256(false);
56      try (final var digestInputStream = new MultipleActionsInputStream(getPseudoInputStreamForSha(bigLen))) {
57        digestInputStream.computeDigest(DigestAlgo.SHA256);
58        FakeInputStream.consumeAll(digestInputStream);
59        sha256 = digestInputStream.getDigestBase32();
60      } catch (IOException | NoSuchAlgorithmException e) {
61        fail(e);
62      }
63    }
64  
65    @AfterAll
66    static void endTests() {
67      DriverAzureProperties.setDynamicPartSize(DEFAULT_MAX_SIZE_NOT_PART);
68      QuarkusProperties.setServerComputeSha256(old);
69    }
70  }
