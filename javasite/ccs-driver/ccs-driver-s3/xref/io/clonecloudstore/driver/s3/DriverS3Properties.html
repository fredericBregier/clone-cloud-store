17  package io.clonecloudstore.driver.s3;
18  
19  import io.clonecloudstore.common.quarkus.properties.QuarkusProperties;
20  import io.clonecloudstore.common.quarkus.properties.QuarkusSystemPropertyUtil;
21  import io.quarkus.arc.Unremovable;
22  import jakarta.enterprise.context.ApplicationScoped;
23  import org.jboss.logging.Logger;
24  
25  /**
26   * Configurable values for S3
27   */
28  @ApplicationScoped
29  @Unremovable
30  public class DriverS3Properties {
31    private static final Logger LOGGER = Logger.getLogger(DriverS3Properties.class);
32    public static final int MAX_ITEMS = 1000;
33    public static final String SHA_256 = "sha256";
34    public static final int DEFAULT_MIN_PART_SIZE = 5 * 1024 * 1024;
35    public static final long DEFAULT_SIZE_NOT_PART = 256 * 1024 * 1024L;
36    public static final long DEFAULT_MAX_SIZE_NOT_PART = 5 * 1024 * 1024 * 1024L;
37    public static final int DEFAULT_MAX_PART_SIZE_INT = 2000 * 1024 * 1024;
38    public static final String CCS_DRIVER_S3_HOST = "ccs.driver.s3.host";
39    public static final String CCS_DRIVER_S3_KEY_ID = "ccs.driver.s3.keyId";
40    public static final String CCS_DRIVER_S3_KEY = "ccs.driver.s3.key";
41    public static final String CCS_DRIVER_S3_REGION = "ccs.driver.s3.region";
42    /**
43     * MultiPart size (minimum 5 MB, maximum 5 GB, default 256 MB)
44     */
45    public static final String CCS_DRIVER_S_3_MAX_PART_SIZE = "ccs.driver.s3.maxPartSize";
46    /**
47     * MultiPart size (minimum 5 MB, maximum ~2 GB): will be used to buffer InputStream if length is unknown, so take
48     * care of the Memory consumption associated (512 MB, default, will limit the total InputStream length to 5 TB
49     * since 10K parts)
50     */
51    public static final String CCS_DRIVER_S3_MAX_PART_SIZE_FOR_UNKNOWN_LENGTH =
52        "ccs.driver.s3.maxPartSizeForUnknownLength";
53    private static String s3Host = QuarkusSystemPropertyUtil.getStringConfig(CCS_DRIVER_S3_HOST, "");
54    private static String s3KeyId = QuarkusSystemPropertyUtil.getStringConfig(CCS_DRIVER_S3_KEY_ID, "");
55    private static String s3Key = QuarkusSystemPropertyUtil.getStringConfig(CCS_DRIVER_S3_KEY, "");
56    private static String s3Region = QuarkusSystemPropertyUtil.getStringConfig(CCS_DRIVER_S3_REGION, "");
57    private static long s3MaxPartSize = Math.min(
58        Math.max(QuarkusSystemPropertyUtil.getLongConfig(CCS_DRIVER_S_3_MAX_PART_SIZE, DEFAULT_SIZE_NOT_PART),
59            DEFAULT_MIN_PART_SIZE), DEFAULT_MAX_SIZE_NOT_PART);
60    private static int s3MaxPartSizeForUnknownLength = Math.min(Math.max(
61        QuarkusSystemPropertyUtil.getIntegerConfig(CCS_DRIVER_S3_MAX_PART_SIZE_FOR_UNKNOWN_LENGTH,
62            QuarkusProperties.getDriverMaxChunkSize()), DEFAULT_MIN_PART_SIZE), DEFAULT_MAX_PART_SIZE_INT);
63  
64    static {
65      QuarkusProperties.setDriverMaxChunkSize(s3MaxPartSizeForUnknownLength);
66    }
67  
68    private final DriverS3Register driverS3Register; // NOSONAR intentional
69  
70    public DriverS3Properties(final DriverS3Register driverS3Register) {
71      this.driverS3Register = driverS3Register;
72    }
73  
74    /**
75     * @return S3 Host
76     */
77    public static String getDriverS3Host() {
78      return s3Host;
79    }
80  
81    /**
82     * @return S3 KeyId
83     */
84    public static String getDriverS3KeyId() {
85      return s3KeyId;
86    }
87  
88    /**
89     * @return S3 Key
90     */
91    public static String getDriverS3Key() {
92      return s3Key;
93    }
94  
95    /**
96     * @return S3 Region
97     */
98    public static String getDriverS3Region() {
99      return s3Region;
100   }
101 
102   public static long getMaxPartSize() {
103     return s3MaxPartSize;
104   }
105 
106   public static int getMaxPartSizeForUnknownLength() {
107     return s3MaxPartSizeForUnknownLength;
108   }
109 
110   /**
111    * Used to change dynamically the setup
112    */
113   public static void setDynamicS3Parameters(final String host, final String keyId, final String key,
114                                             final String region) {
115     s3Host = host;
116     s3KeyId = keyId;
117     s3Key = key;
118     s3Region = region;
119     LOGGER.debugf("Change configuration S3: %s %s %s", s3Host, s3KeyId, s3Region);
120   }
121 
122   /**
123    * Used to change dynamically the setup
124    */
125   public static void setDynamicPartSize(final long size) {
126     s3MaxPartSize = Math.min(Math.max(size, DEFAULT_MIN_PART_SIZE), DEFAULT_MAX_SIZE_NOT_PART);
127   }
128 
129   /**
130    * Used to change dynamically the setup
131    */
132   public static void setDynamicPartSizeForUnknownLength(final int size) {
133     s3MaxPartSizeForUnknownLength = Math.min(Math.max(size, DEFAULT_MIN_PART_SIZE), DEFAULT_MAX_PART_SIZE_INT);
134     QuarkusProperties.setDriverMaxChunkSize(s3MaxPartSizeForUnknownLength);
135   }
136 }
