70    @Override
71    public Stream<StorageBucket> bucketsStream() throws DriverException {
72      // List first level buckets from S3
73      final var response = driverS3Helper.getS3Buckets(s3Client);
74      final var buckets = response.buckets();
75      final List<StorageBucket> directories = new ArrayList<>(buckets.size());
76      for (final var bucket : buckets) {
77        directories.add(driverS3Helper.fromBucket(bucket));
78      }
79      return directories.stream();
80    }

159   @Override
160   public void objectPrepareCreateInBucket(final StorageObject object, final InputStream inputStream)
161       throws DriverNotFoundException, DriverAlreadyExistException, DriverException { // NOSONAR Exception details
162     if (!driverS3Helper.existS3Bucket(s3Client, object.bucket())) {
163       throw new DriverNotFoundException(DriverS3Helper.BUCKET_DOES_NOT_EXIST + object.bucket());
164     }
165     checkExistingObjectOnStorage(object);
166     final var countDownLatch = new CountDownLatch(1);
167     final var transferStatus = new TransferStatus();
168     transferStatus.countDownLatch = countDownLatch;
169     transferStatus.exception = null;
170     transferStatus.size = 0;
171     storageObjectCreations.put(object.bucket() + '/' + object.name(), transferStatus);
172     STANDARD_EXECUTOR_SERVICE.execute(() -> {
173       try {
174         final var exc = objectCreatePreparedAsync(object, inputStream, countDownLatch, transferStatus);
175         if (exc != null) {
176           transferStatus.exception = exc;
177           LOGGER.error(exc, exc);
178         }
179         SystemTools.silentlyCloseNoException(inputStream);
180       } finally {
181         countDownLatch.countDown();
182       }
183     });
184     Thread.yield();
185   }
186 
187   private void checkExistingObjectOnStorage(final StorageObject object)
188       throws DriverAlreadyExistException, DriverException { // NOSONAR Exception details
189     if (driverS3Helper.existS3ObjectInBucket(s3Client, object.bucket(), object.name())) {
190       throw new DriverAlreadyExistException("Object already exists: " + object.bucket() + ":" + object.name());
191     }
192   }

194   private Exception objectCreatePreparedAsync(final StorageObject object, final InputStream inputStream,
195                                               final CountDownLatch countDownLatch,
196                                               final TransferStatus transferStatus) {
197     if (object.size() > 0 && object.size() <= DriverS3Properties.getMaxPartSize()) {
198       return objectCreatePreparedAsyncMonoPart(object, inputStream, countDownLatch, transferStatus);
199     } else {
200       return objectCreatePreparedAsyncMultiParts(object, inputStream, countDownLatch, transferStatus);
201     }
202   }
203 
204   private Exception objectCreatePreparedAsyncMultiParts(final StorageObject object, final InputStream inputStream,
205                                                         final CountDownLatch countDownLatch,
206                                                         final TransferStatus transferStatus) {
207     LOGGER.debugf("Start creation chunked: %s", object.name());
208     final var partSize =
209         Math.min(object.size() > 0 ? object.size() : DriverS3Properties.getMaxPartSizeForUnknownLength(),
210             DriverS3Properties.getMaxPartSizeForUnknownLength());
211     final var chunkInputStream = new ChunkInputStream(inputStream, object.size(), (int) partSize);
212     try (final var client = driverS3Helper.getClient()) {
213       final var multipartUploadHelper = new MultipartUploadHelper(client, object);
214       return chunkByChunkAsyncMultiParts(transferStatus, chunkInputStream, multipartUploadHelper);
215     } catch (final DriverException e) {
216       transferStatus.exception = e;
217       return e;
218     } finally {
219       SystemTools.silentlyCloseNoException(inputStream);
220       SystemTools.silentlyCloseNoException(chunkInputStream);
221       countDownLatch.countDown();
222     }
223   }
224 
225   private Exception chunkByChunkAsyncMultiParts(final TransferStatus transferStatus,
226                                                 final ChunkInputStream chunkInputStream,
227                                                 final MultipartUploadHelper multipartUploadHelper) {
228     try {
229       while (chunkInputStream.nextChunk()) {
230         final var chunkSize = chunkInputStream.getChunkSize();
231         LOGGER.debugf("Newt ChunkSize: %d", chunkSize);
232         multipartUploadHelper.partUpload(chunkInputStream, chunkSize);
233         transferStatus.size += chunkSize;
234         Thread.yield();
235       }
236       multipartUploadHelper.complete();
237       Thread.yield();
238       return null;
239     } catch (final Exception e) {
240       LOGGER.debugf("Error: %s", e.getMessage());
241       transferStatus.exception = e;
242       try {
243         LOGGER.debug("Cancel multipart");
244         multipartUploadHelper.cancel();
245       } catch (final DriverException e2) {
246         LOGGER.warn("Cancel in error", e2);
247       }
248       return e;
249     }
250   }
251 
252   private Exception objectCreatePreparedAsyncMonoPart(final StorageObject object, final InputStream inputStream,
253                                                       final CountDownLatch countDownLatch,
254                                                       final TransferStatus transferStatus) {
255     LOGGER.debugf("Start creation direct: %s", object.name());
256     try (final var client = driverS3Helper.getClient()) {
257       driverS3Helper.createS3ObjectInBucketNoCheck(client, object, inputStream);
258       transferStatus.size = object.size();
259       Thread.yield();
260       return null;
261     } catch (final DriverException e) {
262       LOGGER.errorf("Error: %s", e);
263       transferStatus.exception = e;
264       return e;
265     } finally {
266       SystemTools.silentlyCloseNoException(inputStream);
267       countDownLatch.countDown();
268     }
269   }

271   @Override
272   public StorageObject objectFinalizeCreateInBucket(final String bucket, final String object, final long realLen,
273                                                     final String sha256)
274       throws DriverNotFoundException, DriverAlreadyExistException, DriverException { // NOSONAR Exception details
275     final var transferStatus = storageObjectCreations.remove(bucket + '/' + object);
276     if (transferStatus == null) {
277       throw new DriverException("Request not found while finishing creation request: " + bucket + '/' + object);
278     }
279     try {
280       transferStatus.countDownLatch.await();
281     } catch (final InterruptedException e) {
282       transferStatus.exception = e;
283       Thread.currentThread().interrupt();
284     }
285     final var exception = transferStatus.exception;
286     if (exception != null) {
287       if (exception instanceof DriverException e) {
288         throw e;
289       }
290       throw new DriverException("Issue during creation", exception);
291     }
292     try {
293       return driverS3Helper.waitUntilObjectExist(s3Client, bucket, object, sha256);
294     } catch (final DriverNotAcceptableException e) {
295       throw new DriverException("Issue during waiting creation ending", e);
296     }
297   }

334   private class StorageObjectIterator implements Iterator<StorageObject> {
335     private final Iterator<S3Object> iterator;
336     private final String bucket;
337 
338     public StorageObjectIterator(final Iterator<S3Object> iterator, final String bucket) {
339       this.iterator = iterator;
340       this.bucket = bucket;
341     }
342 
343     @Override
344     public boolean hasNext() {
345       return iterator.hasNext();
346     }
347 
348     @Override
349     public StorageObject next() {
350       final var item = iterator.next();
351       try {
352         return driverS3Helper.fromS3Object(s3Client, bucket, item);
353       } catch (DriverException e) {
354         // Should not occur except if object is deleted in the middle
355         throw new DriverRuntimeException(e.getMessage(), e);
356       }
357     }
