38  @QuarkusTest
39  class StorageObjectTest {
40  
41    @Test
42    void checkBean() {
43      final var object = new StorageObject("name", "objectName", "hash", 1024, Instant.now());
44      final var object1 =
45          new StorageObject(object.bucket(), object.name(), object.hash(), object.size(), object.creationDate());
46      assertEquals(object1, object);
47      assertEquals(object1, object);
48      assertEquals(object1.hashCode(), object.hashCode());
49      assertEquals(object1, object1);
50      assertTrue(object1.toString().contains(object1.bucket()));
51      assertTrue(object1.toString().contains(object1.creationDate().toString()));
52      final Map<String, String> map = new HashMap<>();
53      map.put("key", "value");
54      final var object2 =
55          new StorageObject(object.bucket(), object.name(), object.hash(), object.size(), object.creationDate(), null,
56              map);
57      assertEquals(1, object2.metadata().size());
58      Log.info(object2);
59      assertThrows(CcsInvalidArgumentRuntimeException.class,
60          () -> new StorageObject("test@test", object.name(), object.hash(), object.size(), object.creationDate()));
61      assertThrows(CcsInvalidArgumentRuntimeException.class,
62          () -> new StorageObject(object.bucket(), "test@test", object.hash(), object.size(), object.creationDate()));
63  
64      StorageObject object3 = new StorageObject(null, null, null, 0, null);
65      assertNotEquals(object, object3);
66      assertEquals(-1, object3.hashCode());
67      object3 = new StorageObject(object.bucket(), null, null, 0, null);
68      assertNotEquals(object, object3);
69      object3 = new StorageObject(null, object.name(), null, 0, null);
70      assertNotEquals(object, object3);
71      assertFalse(object.equals(new Object()));
72    }
73  
74    @Test
75    void checkTypeBean() throws JsonProcessingException {
76      assertTrue(StorageType.NONE.ordinal() < StorageType.BUCKET.ordinal());
77      assertTrue(StorageType.BUCKET.ordinal() < StorageType.DIRECTORY.ordinal());
78      assertTrue(StorageType.DIRECTORY.ordinal() < StorageType.OBJECT.ordinal());
79      assertNotNull(JsonUtil.getInstance().writeValueAsString(StorageType.OBJECT));
80    }
81  }
