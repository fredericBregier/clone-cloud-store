59    @AfterEach
60    void afterEach() {
61      try (final var driver = factory.getInstance()) {
62        final var bucketStorageResponse = driver.bucketsIterator();
63        while (bucketStorageResponse.hasNext()) {
64          final var bucket = bucketStorageResponse.next();
65          Iterator<StorageObject> objectStorageResponse = null;
66          try {
67            objectStorageResponse = driver.objectsIteratorInBucket(bucket.bucket());
68          } catch (final Exception e) {
69            LOG.error(e.getMessage());
70          }
71          while (objectStorageResponse.hasNext()) {
72            final var object = objectStorageResponse.next();
73            try {
74              driver.objectDeleteInBucket(object.bucket(), object.name());
75            } catch (final Exception e) {
76              LOG.error(e.getMessage());
77            }
78          }
79          ;
80          try {
81            driver.bucketDelete(bucket.bucket());
82          } catch (final Exception e) {
83            LOG.error(e.getMessage());
84          }
85        }
86        ;
87        FakeDriverFactory.cleanUp();
88      } catch (final Exception e) {
89        LOG.error(e.getMessage());
90      }

222   @Test
223   void test3CreateAndDeleteObjectsInBucket() {
224     final var bucket = "test1";
225     final var directory1 = "dir1/";
226     final var object1 = directory1 + "object1";
227     final var directory2 = "dir2/";
228     final var object2 = directory2 + "object2";
229     final long len1 = 1000;
230     final long len2 = 2000;
231     final var storageBucket = new StorageBucket(bucket, Instant.now());
232     final var object = new StorageObject(bucket, object1, null, len1, Instant.now().plusSeconds(1));
233     try (final var driver = factory.getInstance()) {
234       assertFalse(driver.bucketExists(bucket));
235       assertEquals(StorageType.NONE, driver.directoryOrObjectExistsInBucket(bucket, directory1));
236       assertEquals(StorageType.NONE, driver.directoryOrObjectExistsInBucket(bucket, directory2));
237 
238       // Create Bucket
239       assertEquals(storageBucket.bucket(), driver.bucketCreate(storageBucket).bucket());
240       assertTrue(driver.bucketExists(bucket));
241       assertEquals(1, driver.bucketsCount());
242       assertEquals(0, driver.objectsCountInBucket(bucket, null, null, null));
243       assertEquals(0, driver.objectsStreamInBucket(bucket).count());
244       assertFalse(driver.objectsIteratorInBucket(bucket).hasNext());
245       assertEquals(StorageType.NONE, driver.directoryOrObjectExistsInBucket(bucket, directory1));
246       assertEquals(StorageType.NONE, driver.directoryOrObjectExistsInBucket(bucket, directory2));
247 
248       // Try reading nonexistent object from bucket
249       try {
250         driver.objectGetMetadataInBucket(bucket, object1);
251         fail("Should produces an exception");
252       } catch (final DriverNotFoundException e) {
253         LOG.info(e.getMessage());
254       }
255       try (final var inputStream = driver.objectGetInputStreamInBucket(bucket, object1)) {
256         inputStream.read();
257         fail("Should produces an exception");
258       } catch (final DriverNotFoundException e) {
259         LOG.info(e.getMessage());
260       } catch (final IOException e) {
261         LOG.warn(e.getMessage());
262       }
263       final var before = object.creationDate().minusSeconds(1);
264       final var after0 = object.creationDate().plusSeconds(1);
265       assertEquals(0, driver.objectsCountInBucket(bucket, null, null, null));
266       assertEquals(0, driver.objectsCountInBucket(bucket, "dir", null, null));
267       assertEquals(0, driver.objectsCountInBucket(bucket, null, before, null));
268       assertEquals(0, driver.objectsCountInBucket(bucket, null, null, before));
269       assertEquals(0, driver.objectsCountInBucket(bucket, null, null, after0));
270       assertEquals(0, driver.objectsCountInBucket(bucket, "dir", before, after0));
271 
272       assertEquals(0, driver.objectsStreamInBucket(bucket, null, null, null).count());
273       assertEquals(0, driver.objectsStreamInBucket(bucket, "dir", null, null).count());
274       assertEquals(0, driver.objectsStreamInBucket(bucket, null, before, null).count());
275       assertEquals(0, driver.objectsStreamInBucket(bucket, null, null, after0).count());
276       assertEquals(0, driver.objectsStreamInBucket(bucket, "dir", before, after0).count());
277       assertEquals(0, driver.objectsStreamInBucket(bucket, null, null, before).count());
278 
279       assertFalse(driver.objectsIteratorInBucket(bucket, null, null, null).hasNext());
280       assertFalse(driver.objectsIteratorInBucket(bucket, "dir", null, null).hasNext());
281       assertFalse(driver.objectsIteratorInBucket(bucket, null, before, null).hasNext());
282       assertFalse(driver.objectsIteratorInBucket(bucket, null, null, after0).hasNext());
283       assertFalse(driver.objectsIteratorInBucket(bucket, "dir", before, after0).hasNext());
284       assertFalse(driver.objectsIteratorInBucket(bucket, null, null, before).hasNext());
285 
286       // Create Object driver
287       assertEquals(StorageType.NONE, driver.directoryOrObjectExistsInBucket(bucket, object1));
288       driver.objectPrepareCreateInBucket(object, new FakeInputStream(len1, (byte) 'A'));
289       final var object3 = driver.objectFinalizeCreateInBucket(bucket, object1, len1, null);
290       assertEquals(object, object3);
291       final var after1 = object.creationDate().plusSeconds(1);
292       assertEquals(len1, object.size());
293       assertEquals(object1, object.name());
294       assertEquals(bucket, object.bucket());
295       assertEquals(1, driver.objectsCountInBucket(bucket, null, null, null));
296       assertEquals(1, driver.objectsCountInBucket(bucket, "dir", null, null));
297       assertEquals(1, driver.objectsCountInBucket(bucket, null, before, null));
298       assertEquals(0, driver.objectsCountInBucket(bucket, null, null, before));
299       assertEquals(1, driver.objectsCountInBucket(bucket, null, null, after1));
300       assertEquals(1, driver.objectsCountInBucket(bucket, "dir", before, after1));
301 
302       assertEquals(1, driver.objectsStreamInBucket(bucket, null, null, null).count());
303       driver.objectsStreamInBucket(bucket, null, null, null).findFirst()
304           .ifPresent(so -> assertEquals(object1, so.name()));
305       assertEquals(1, driver.objectsStreamInBucket(bucket, "dir", null, null).count());
306       assertEquals(1, driver.objectsStreamInBucket(bucket, null, before, null).count());
307       assertEquals(1, driver.objectsStreamInBucket(bucket, null, null, after1).count());
308       assertEquals(1, driver.objectsStreamInBucket(bucket, "dir", before, after1).count());
309       assertEquals(0, driver.objectsStreamInBucket(bucket, null, null, before).count());
310 
311       var iterator = driver.objectsIteratorInBucket(bucket, null, null, null);
312       assertTrue(iterator.hasNext());
313       assertEquals(object1, iterator.next().name());
314       assertFalse(iterator.hasNext());
315       iterator = driver.objectsIteratorInBucket(bucket, "dir", null, null);
316       assertTrue(iterator.hasNext());
317       iterator.next();
318       assertFalse(iterator.hasNext());
319       iterator = driver.objectsIteratorInBucket(bucket, null, before, null);
320       assertTrue(iterator.hasNext());
321       iterator.next();
322       assertFalse(iterator.hasNext());
323       iterator = driver.objectsIteratorInBucket(bucket, null, null, after1);
324       assertTrue(iterator.hasNext());
325       iterator.next();
326       assertFalse(iterator.hasNext());
327       iterator = driver.objectsIteratorInBucket(bucket, "dir", before, after1);
328       assertTrue(iterator.hasNext());
329       iterator.next();
330       assertFalse(iterator.hasNext());
331       iterator = driver.objectsIteratorInBucket(bucket, null, null, before);
332       assertFalse(iterator.hasNext());
333 
334       assertEquals(StorageType.OBJECT, driver.directoryOrObjectExistsInBucket(bucket, object1));
335       assertEquals(StorageType.DIRECTORY, driver.directoryOrObjectExistsInBucket(bucket, directory1));
336       assertEquals(StorageType.NONE, driver.directoryOrObjectExistsInBucket(bucket, directory2));
337       try {
338         final var storageObject = driver.objectGetMetadataInBucket(bucket, object1);
339         assertEquals(len1, storageObject.size());
340         assertEquals(object1, storageObject.name());
341         assertEquals(bucket, storageObject.bucket());
342       } catch (final DriverNotFoundException e) {
343         fail(e);
344       }
345       try {
346         final var inputStream = driver.objectGetInputStreamInBucket(bucket, object1);
347         final var len = FakeInputStream.consumeAll(inputStream);
348         inputStream.close();
349         assertEquals(len1, len);
350       } catch (final DriverNotFoundException | IOException e) {
351         fail(e);
352       }
353       // Try wrong Delete Bucket/Object
354       try {
355         driver.bucketDelete(bucket);
356         fail("Should produces an exception");
357       } catch (final DriverNotAcceptableException e) {
358         LOG.info(e.getMessage());
359       }
360       try {
361         driver.objectDeleteInBucket(bucket, object2);
362         fail("Should produces an exception");
363       } catch (final DriverNotFoundException e) {
364         LOG.info(e.getMessage());
365       } catch (final DriverException e) {
366         fail(e);
367       }
368 
369       // Try reading nonexistent object from bucket
370       try {
371         driver.objectGetMetadataInBucket(bucket, object2);
372         fail("Should produces an exception");
373       } catch (final DriverNotFoundException e) {
374         LOG.info(e.getMessage());
375       }
376       try (final var inputStream = driver.objectGetInputStreamInBucket(bucket, object2)) {
377         inputStream.read();
378         fail("Should produces an exception");
379       } catch (final DriverNotFoundException e) {
380         LOG.info(e.getMessage());
381       } catch (final IOException e) {
382         LOG.warn(e.getMessage());
383       }
384 
385       // Add second Object (wait 2s to keep time correct)
386       final var second = after1.plusSeconds(2);
387       assertEquals(StorageType.NONE, driver.directoryOrObjectExistsInBucket(bucket, object2));
388       var storageObject = new StorageObject(bucket, object2, null, len2, second);
389       driver.objectPrepareCreateInBucket(storageObject, new FakeInputStream(len2, (byte) 'A'));
390       driver.objectFinalizeCreateInBucket(bucket, object2, len2, null);
391       storageObject = driver.objectGetMetadataInBucket(bucket, object2);
392       assertEquals(len2, storageObject.size());
393       assertEquals(object2, storageObject.name());
394       assertEquals(bucket, storageObject.bucket());
395       assertEquals(2, driver.objectsCountInBucket(bucket));
396       assertEquals(2, driver.objectsCountInBucket(bucket, null, null, null));
397       assertEquals(2, driver.objectsCountInBucket(bucket, "dir", null, null));
398       assertEquals(2, driver.objectsCountInBucket(bucket, null, before, null));
399       assertEquals(0, driver.objectsCountInBucket(bucket, null, null, before));
400       assertEquals(1, driver.objectsCountInBucket(bucket, null, null, after1));
401       final var after2 = second.plusSeconds(1);
402       assertEquals(2, driver.objectsCountInBucket(bucket, null, null, after2));
403       assertEquals(2, driver.objectsCountInBucket(bucket, "dir", before, after2));
404       assertEquals(1, driver.objectsCountInBucket(bucket, "dir", after1, after2));
405       assertEquals(2, driver.objectsStreamInBucket(bucket, null, null, null).count());
406       assertEquals(2, driver.objectsStreamInBucket(bucket).count());
407       assertEquals(StorageType.OBJECT, driver.directoryOrObjectExistsInBucket(bucket, object2));
408       assertEquals(StorageType.DIRECTORY, driver.directoryOrObjectExistsInBucket(bucket, directory1));
409       assertEquals(StorageType.DIRECTORY, driver.directoryOrObjectExistsInBucket(bucket, directory2));
410       try {
411         storageObject = driver.objectGetMetadataInBucket(bucket, object2);
412         assertEquals(len2, storageObject.size());
413         assertEquals(object2, storageObject.name());
414         assertEquals(bucket, storageObject.bucket());
415       } catch (final DriverNotFoundException e) {
416         fail(e);
417       }
418       try {
419         final var inputStream = driver.objectGetInputStreamInBucket(bucket, object2);
420         final var len = FakeInputStream.consumeAll(inputStream);
421         inputStream.close();
422         assertEquals(len2, len);
423       } catch (final DriverNotFoundException | IOException e) {
424         fail(e);

429         driver.objectPrepareCreateInBucket(storageObject, new FakeInputStream(len2, (byte) 'A'));

432         LOG.info(e.getMessage());

449         driver.objectGetMetadataInBucket(bucket + "no", object2);

452         LOG.info(e.getMessage());

455         final var inputStream = driver.objectGetInputStreamInBucket(bucket + "no", object2);

473       assertEquals(bucket, driver.objectsStreamInBucket(bucket, null, null, null).findFirst().get().bucket());
474       try {
475         driver.bucketDelete(bucket);

485       assertEquals(StorageType.NONE, driver.directoryOrObjectExistsInBucket(bucket, directory1));

498     } catch (final Exception e) {

501     }
