40  import static io.clonecloudstore.driver.google.example.server.ApiService.NOT_ACCEPTABLE_NAME;

42  @RequestScoped
43  public class NativeStreamHandler extends NativeStreamHandlerAbstract<StorageObject, StorageObject> {
44    private static final Logger LOG = Logger.getLogger(NativeStreamHandler.class);
45    private volatile StorageObject pullAble;
46    private DriverApi driverApi;
47  
48    private void setDriverApi() {
49      if (driverApi == null) {
50        // No native CDI here
51        driverApi = DriverApiRegistry.getDriverApiFactory().getInstance();
52      }
53    }
54  
55    @Override
56    protected void postSetup() {
57      setDriverApi();
58      getCloser().add(driverApi);
59    }
60  
61    @Override
62    protected boolean checkDigestToCumpute(final StorageObject businessIn) {
63      return true;
64    }
65  
66    @Override
67    protected void checkPushAble(final StorageObject apiBusinessIn, final MultipleActionsInputStream inputStream) {
68      // Business code should come here (example: check through Object Storage that object does not exist yet, and if so
69      // add the consumption of the stream for the Object Storage object creation)
70      if (NOT_ACCEPTABLE_NAME.equals(apiBusinessIn.name())) {
71        throw new CcsNotAcceptableException("Not Acceptable test");
72      }
73      try {
74        driverApi.objectPrepareCreateInBucket(apiBusinessIn, inputStream);
75      } catch (final DriverNotFoundException e) {
76        LOG.debugf("Not Pushable since not found: %s/%s", apiBusinessIn.bucket(), apiBusinessIn.name());
77        throw new CcsNotExistException(e.getMessage(), e);
78      } catch (final DriverAlreadyExistException e) {
79        LOG.debugf("Not Pushable since already exists: %s/%s", apiBusinessIn.bucket(), apiBusinessIn.name());
80        throw new CcsAlreadyExistException(e.getMessage(), e);
81      } catch (final DriverException e) {
82        throw new CcsOperationException(e.getMessage(), e);
83      }
84    }
85  
86    @Override
87    protected StorageObject getAnswerPushInputStream(final StorageObject apiBusinessIn, final String finalHash,
88                                                     final long size) {
89      // Business code should come here (example: getting the StorageObject object)
90      // Hash from request
91      var hash = apiBusinessIn.hash();
92      if (finalHash != null) {
93        // Hash from NettyToInputStream (on the fly)
94        hash = finalHash;
95      }
96      try {
97        return driverApi.objectFinalizeCreateInBucket(apiBusinessIn.bucket(), apiBusinessIn.name(), size, hash);
98      } catch (final DriverNotFoundException e) {
99        throw new CcsNotExistException(e.getMessage(), e);
100     } catch (final DriverAlreadyExistException e) {
101       throw new CcsAlreadyExistException(e.getMessage(), e);
102     } catch (final DriverException e) {
103       throw new CcsOperationException(e.getMessage(), e);
104     }
105   }
106 
107   @Override
108   protected Map<String, String> getHeaderPushInputStream(final StorageObject apiBusinessIn, final String finalHash,
109                                                          final long size, final StorageObject apiBusinessOut) {
110     // Business code should come here (example: headers for object name, object size, ...)
111     final var map = getHeaderMap(apiBusinessIn);
112     final var creationDate = apiBusinessOut.creationDate();
113     if (creationDate != null) {
114       map.put(ApiConstants.X_CREATION_DATE, creationDate.toString());
115     }
116     // Hash from request
117     var hash = apiBusinessIn.hash();
118     if (apiBusinessOut.hash() != null) {
119       // Hash from prepared Output
120       hash = apiBusinessOut.hash();
121     }
122     if (hash == null) {
123       // Hash from NettyToInputStream (on the fly)
124       hash = finalHash;
125     }
126     if (hash != null) {
127       map.put(ApiConstants.X_HASH, hash);
128     }
129     if (size > 0) {
130       map.put(ApiConstants.X_LEN, Long.toString(size));
131     }
132     return map;
133   }
134 
135   @Override
136   protected boolean checkPullAble(final StorageObject apiBusinessIn, final MultiMap headers) {
137     // Business code should come here
138     // Put here business logic to check if the GET is valid (example: ObjectStorage check of existence of object)
139     try {
140       pullAble = driverApi.objectGetMetadataInBucket(apiBusinessIn.bucket(), apiBusinessIn.name());
141       LOG.debugf("Pullable: %s/%s = %s => %s", apiBusinessIn.bucket(), apiBusinessIn.name(), apiBusinessIn, pullAble);
142       return true;
143     } catch (final DriverNotFoundException e) {
144       pullAble = null;
145       LOG.debugf("Not Pullable: %s/%s ", apiBusinessIn.bucket(), apiBusinessIn.name());
146       return false;
147     } catch (final DriverException e) {
148       throw new CcsOperationException(e.getMessage(), e);
149     }
150   }
151 
152   @Override
153   protected InputStream getPullInputStream(final StorageObject apiBusinessIn) {
154     // Business code should come here (example: getting the Object Storage object stream)
155     try {
156       return driverApi.objectGetInputStreamInBucket(apiBusinessIn.bucket(), apiBusinessIn.name());
157     } catch (final DriverNotFoundException e) {
158       throw new CcsNotExistException(e.getMessage(), e);
159     } catch (final DriverException e) {
160       throw new CcsOperationException(e.getMessage(), e);
161     }
162   }
163 
164   @Override
165   protected Map<String, String> getHeaderPullInputStream(final StorageObject apiBusinessIn) {
166     // Business code should come here (example: headers for object name, object size...)
167     LOG.debugf("Pullable %s", pullAble);
168     if (pullAble != null) {
169       return getHeaderMap(pullAble);
170     }
171     try {
172       return getHeaderMap(driverApi.objectGetMetadataInBucket(apiBusinessIn.bucket(), apiBusinessIn.name()));
173     } catch (final DriverNotFoundException e) {
174       throw new CcsNotExistException(e.getMessage(), e);
175     } catch (final DriverException e) {
176       throw new CcsOperationException(e.getMessage(), e);
177     }
178   }
179 
180   @Override
181   protected Map<String, String> getHeaderError(final StorageObject apiBusinessIn, final int status) {
182     // Business code should come here (example: get headers in case of error as Object name, Bucket name...)
183     return getHeaderMap(apiBusinessIn);
184   }
185 
186   private static Map<String, String> getHeaderMap(final StorageObject apiBusinessIn) {
187     final Map<String, String> map = new HashMap<>();
188     map.put(ApiConstants.X_BUCKET, apiBusinessIn.bucket());
189     map.put(ApiConstants.X_OBJECT, apiBusinessIn.name());
190     final var creationDate = apiBusinessIn.creationDate();
191     if (creationDate != null) {
192       map.put(ApiConstants.X_CREATION_DATE, creationDate.toString());
193     }
194     final var hash = apiBusinessIn.hash();
195     if (hash != null) {
196       map.put(ApiConstants.X_HASH, hash);
197     }
198     if (apiBusinessIn.size() > 0) {
199       map.put(ApiConstants.X_LEN, Long.toString(apiBusinessIn.size()));
200     }
201     LOG.debugf("Debug Map: %s", map);
202     return map;
203   }
