60    public boolean bucketExists(String bucket) throws DriverException {
61      final var uni = getService().bucketExists(bucket);
62      var businessIn = new StorageObject(bucket, null, null, 0, null);
63      try (final var response = exceptionMapper.handleUniResponse(uni)) {
64        switch (response.getStatus()) {
65          case 200, 204 -> {
66            return true;
67          }
68        }
69        return false;
70      } catch (final CcsWithStatusException e) {
71        if (e.getStatus() == Response.Status.NOT_FOUND.getStatusCode()) {
72          return false;
73        }
74        throw new DriverException(e);
75      } catch (final CcsClientGenericException | CcsServerGenericException e) {
76        if (e.getStatus() == Response.Status.NOT_FOUND.getStatusCode()) {
77          return false;
78        }
79        throw new DriverException(e);
80      }
81    }

104   public void bucketDelete(String bucket)
105       throws DriverNotFoundException, DriverException, DriverNotAcceptableException {
106     final var uni = getService().bucketDelete(bucket);
107     var businessIn = new StorageObject(bucket, null, null, 0, null);
108     try (final var response = exceptionMapper.handleUniResponse(uni)) {
109       if (response.getStatus() == Response.Status.NOT_FOUND.getStatusCode()) {
110         throw new DriverNotFoundException("Bucket not found: " + bucket);

112       if (response.getStatus() == Response.Status.NOT_ACCEPTABLE.getStatusCode()) {
113         throw new DriverNotAcceptableException("Bucket not empty: " + bucket);
114       }
115     } catch (final CcsWithStatusException e) {
116       switch (e.getStatus()) {
117         case 404 -> throw new DriverNotFoundException(e);
118         case 406 -> throw new DriverNotAcceptableException(e);
119       }
120       throw new DriverException(e);
121     } catch (final CcsClientGenericException | CcsServerGenericException e) {
122       switch (e.getStatus()) {
123         case 404 -> throw new DriverNotFoundException(e);
124         case 406 -> throw new DriverNotAcceptableException(e);
125       }
126       throw new DriverException(e);

130   public StorageType objectOrDirectoryExists(String bucket, String objectOrDirectory) throws DriverException {
131     final var uni = getService().objectOrDirectoryExists(bucket, objectOrDirectory);
132     var businessIn = new StorageObject(bucket, objectOrDirectory, null, 0, null);
133     try (final var response = exceptionMapper.handleUniResponse(uni)) {
134       var type = (String) response.getHeaders().get(ApiConstants.X_TYPE).get(0);
135       if (ParametersChecker.isNotEmpty(type)) {
136         return StorageType.valueOf(type);
137       }
138       return StorageType.NONE;
139     } catch (CcsWithStatusException e) {
140       if (e.getStatus() == 404) {
141         return StorageType.NONE;
142       }
143       throw new DriverException(e);
144     } catch (final CcsClientGenericException | CcsServerGenericException e) {
145       if (e.getStatus() == 404) {
146         return StorageType.NONE;
147       }
148       throw new DriverException(e);
149     }

250   private Map<String, String> getHeaders(final StorageObject apiBusinessIn, final long len) {
251     // Here simple example of Headers for GET/POST InputStream
252     Map<String, String> map = new HashMap<>();
253     map.put(ApiConstants.X_LEN, Long.toString(len));
254     map.put(ApiConstants.X_BUCKET, apiBusinessIn.bucket());
255     map.put(ApiConstants.X_OBJECT, apiBusinessIn.name());
256     map.put(ApiConstants.X_HASH, apiBusinessIn.hash());
257     return map;
258   }

260   @Override
261   protected StorageObject getApiBusinessOutFromResponse(final Response response) {
262     try {
263       final var storageObject = response.readEntity(StorageObject.class);
264       if (storageObject != null) {
265         return storageObject;
266       }
267     } catch (final RuntimeException ignore) {
268       // Nothing
269     }
270     final var headers = response.getStringHeaders();
271     long len = 0;
272     var slen = headers.getFirst(ApiConstants.X_LEN);
273     if (ParametersChecker.isNotEmpty(slen)) {
274       len = Long.parseLong(slen);
275     }
276     Instant instant = null;
277     var sInstant = headers.getFirst(ApiConstants.X_CREATION_DATE);
278     if (ParametersChecker.isNotEmpty(sInstant)) {
279       instant = Instant.parse(sInstant);
280     }
281     var bucket = headers.getFirst(ApiConstants.X_BUCKET);
282     var sobject = headers.getFirst(ApiConstants.X_OBJECT);
283     return new StorageObject(bucket, sobject, headers.getFirst(ApiConstants.X_HASH), len, instant);
284   }
