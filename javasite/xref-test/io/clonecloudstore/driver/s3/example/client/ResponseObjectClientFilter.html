17  package io.clonecloudstore.driver.s3.example.client;
18  
19  import java.time.Instant;
20  
21  import io.clonecloudstore.common.quarkus.client.SimpleClientAbstract;
22  import io.clonecloudstore.common.standard.system.ParametersChecker;
23  import io.clonecloudstore.driver.api.model.StorageObject;
24  import jakarta.ws.rs.HttpMethod;
25  import jakarta.ws.rs.client.ClientResponseContext;
26  import org.jboss.logging.Logger;
27  import org.jboss.resteasy.reactive.client.spi.ResteasyReactiveClientRequestContext;
28  import org.jboss.resteasy.reactive.client.spi.ResteasyReactiveClientResponseFilter;
29  
30  import static io.clonecloudstore.common.quarkus.client.SimpleClientAbstract.X_OP_ID;
31  
32  public class ResponseObjectClientFilter implements ResteasyReactiveClientResponseFilter {
33    private static final Logger LOGGER = Logger.getLogger(ResponseObjectClientFilter.class);
34  
35    @Override
36    public void filter(final ResteasyReactiveClientRequestContext requestContext,
37                       final ClientResponseContext responseContext) {
38      SimpleClientAbstract.setMdcOpId((String) requestContext.getHeaders().getFirst(X_OP_ID));
39      final var method = requestContext.getMethod();
40      if (HttpMethod.GET.equalsIgnoreCase(method) || HttpMethod.POST.equalsIgnoreCase(method)) {
41        if (responseContext.getHeaders() != null) {
42          var headers = responseContext.getHeaders();
43          LOGGER.debugf("Headers: %s", headers);
44          long len = 0;
45          var slen = headers.getFirst(ApiConstants.X_LEN);
46          if (ParametersChecker.isNotEmpty(slen)) {
47            len = Long.parseLong(slen);
48          }
49          Instant instant = null;
50          var sInstant = headers.getFirst(ApiConstants.X_CREATION_DATE);
51          if (ParametersChecker.isNotEmpty(sInstant)) {
52            instant = Instant.parse(sInstant);
53          }
54          var bucket = headers.getFirst(ApiConstants.X_BUCKET);
55          var sobject = headers.getFirst(ApiConstants.X_OBJECT);
56          final var object = new StorageObject(bucket, sobject, headers.getFirst(ApiConstants.X_HASH), len, instant);
57          SimpleClientAbstract.setDtoFromHeaders(object);
58        }
59      }
60    }
61  }
