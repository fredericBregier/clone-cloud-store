34    @Test
35    void testCcsServerException() {
36      CcsServerGenericException serverInternalEx = new CcsOperationException("Internal Error Exception");
37      assertEquals(Module.UNKNOWN + " - Internal Error Exception", serverInternalEx.getMessage());
38      assertEquals(500, serverInternalEx.getResponse().getStatus());
39      assertEquals(Module.UNKNOWN, serverInternalEx.getModule());
40  
41      CcsServerGenericException serverInternalWithCauseEx =
42          new CcsOperationException("Internal Error Exception with cause", serverInternalEx);
43      assertEquals(Module.UNKNOWN + " - Internal Error Exception with cause", serverInternalWithCauseEx.getMessage());
44      assertEquals(Module.UNKNOWN + " - Internal Error Exception", serverInternalWithCauseEx.getCause().getMessage());
45      assertEquals(Module.UNKNOWN, serverInternalWithCauseEx.getModule());
46      assertEquals(500, serverInternalWithCauseEx.getResponse().getStatus());
47  
48      //Test generic server error exception mapper
49      //Consume Uni<Response> and use response to check status and error message
50      CcsServerGenericExceptionMapper exceptionMapper = new CcsServerGenericExceptionMapper();
51      Uni<Response> responseExceptionMapper = exceptionMapper.handleServerException(serverInternalEx);
52      Response response =
53          responseExceptionMapper.subscribe().withSubscriber(UniAssertSubscriber.create()).assertCompleted().getItem();
54      assertEquals(500, response.getStatus());
55      assertEquals("Internal Server Error", response.getStatusInfo().getReasonPhrase());
56    }

63      assertEquals(Module.UNKNOWN, alreadyExistException.getModule());

70      CcsNotAcceptableException notAcceptableException = new CcsNotAcceptableException("Not Acceptable Exception");
71      assertEquals(Module.UNKNOWN + " - Not Acceptable Exception", notAcceptableException.getMessage());

73      assertEquals(Module.UNKNOWN, notAcceptableException.getModule());

75      CcsNotAcceptableException notAcceptableException2 =

78      assertEquals(406, notAcceptableException2.getResponse().getStatus());

84      assertEquals(Module.UNKNOWN, notExistException.getModule());

86      CcsNotExistException notExistExceptionWithCause =

99          new CcsOperationException("Operation Exception with cause", operationException);

103     assertEquals(Module.UNKNOWN, operationExceptionWithCause.getModule());
104

107         operationExceptionWithCause2.getMessage());

109     assertEquals(Module.UNKNOWN + " - Operation Exception", operationExceptionWithCause2.getCause().getMessage());
110     assertEquals(Module.UNKNOWN, operationExceptionWithCause2.getModule());

113   @Test
114   void testExceptionMapper() {
115     final var exception = new Exception("test");
116     final var listStatus = List.of(Response.Status.NOT_FOUND, Response.Status.CONFLICT, Response.Status.GONE,
117         Response.Status.NOT_ACCEPTABLE, Response.Status.REQUEST_ENTITY_TOO_LARGE);
118     for (final var status : listStatus) {
119       var e = CcsServerGenericExceptionMapper.getCcsException(status.getStatusCode());
120       assertEquals(status.getStatusCode(), ((CcsClientGenericException) e).getStatus());
121       assertEquals(Module.UNKNOWN + " - " + status.getReasonPhrase(), e.getMessage());
122       assertNull(e.getCause());
123       e = CcsServerGenericExceptionMapper.getCcsException(status.getStatusCode(), "message", null);
124       assertEquals(status.getStatusCode(), ((CcsClientGenericException) e).getStatus());
125       assertEquals(Module.UNKNOWN + " - " + "message", e.getMessage());
126       assertNull(e.getCause());
127       e = CcsServerGenericExceptionMapper.getCcsException(status.getStatusCode(), "message", exception);
128       assertEquals(status.getStatusCode(), ((CcsClientGenericException) e).getStatus());
129       assertEquals(Module.UNKNOWN + " - " + "message", e.getMessage());
130       assertEquals(exception, e.getCause());
131       switch (status) {
132         case NOT_FOUND -> assertInstanceOf(CcsNotExistException.class, e);
133         case CONFLICT -> assertInstanceOf(CcsAlreadyExistException.class, e);
134         case GONE -> assertInstanceOf(CcsDeletedException.class, e);
135         case NOT_ACCEPTABLE -> assertInstanceOf(CcsNotAcceptableException.class, e);
136         default -> assertInstanceOf(CcsClientGenericException.class, e);
137       }
138     }
139     final var listStatusServer = List.of(Response.Status.INTERNAL_SERVER_ERROR, Response.Status.NOT_IMPLEMENTED);
140     for (final var status : listStatusServer) {
141       var e = CcsServerGenericExceptionMapper.getCcsException(status.getStatusCode());
142       assertEquals(status.getStatusCode(), ((CcsServerGenericException) e).getStatus());
143       assertEquals(Module.UNKNOWN + " - " + status.getReasonPhrase(), e.getMessage());
144       assertNull(e.getCause());
145       e = CcsServerGenericExceptionMapper.getCcsException(status.getStatusCode(), "message", null);
146       assertEquals(status.getStatusCode(), ((CcsServerGenericException) e).getStatus());
147       assertEquals(Module.UNKNOWN + " - " + "message", e.getMessage());
148       assertNull(e.getCause());
149       e = CcsServerGenericExceptionMapper.getCcsException(status.getStatusCode(), "message", exception);
150       assertEquals(status.getStatusCode(), ((CcsServerGenericException) e).getStatus());
151       assertEquals(Module.UNKNOWN + " - " + "message", e.getMessage());
152       assertEquals(exception, e.getCause());
153       switch (status) {
154         case INTERNAL_SERVER_ERROR -> assertInstanceOf(CcsOperationException.class, e);
155         default -> assertInstanceOf(CcsServerGenericException.class, e);
156       }
157     }
158   }
