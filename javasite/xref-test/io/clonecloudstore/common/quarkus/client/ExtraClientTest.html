17  package io.clonecloudstore.common.quarkus.client;
18  
19  import java.net.URI;
20  
21  import io.clonecloudstore.common.quarkus.client.example.ApiClientFactory;
22  import io.clonecloudstore.common.quarkus.client.example.SimpleApiClientFactory;
23  import io.clonecloudstore.common.standard.properties.StandardProperties;
24  import io.clonecloudstore.test.stream.FakeInputStream;
25  import io.clonecloudstore.test.stream.VoidOutputStream;
26  import io.quarkus.test.junit.QuarkusTest;
27  import jakarta.inject.Inject;
28  import org.jboss.logging.Logger;
29  import org.junit.jupiter.api.Test;
30  
31  import static org.junit.jupiter.api.Assertions.assertEquals;
32  import static org.junit.jupiter.api.Assertions.assertThrows;
33  
34  @QuarkusTest
35  class ExtraClientTest {
36    private static final Logger LOG = Logger.getLogger(ApiClientTest.class);
37  
38    @Inject
39    ApiClientFactory factory;
40    @Inject
41    SimpleApiClientFactory simpleFactory;
42  
43    @Test
44    void testClientAbstract() {
45      final var uri = factory.getUri();
46      try {
47        try (final var client = factory.newClient(URI.create("http://127.0.0.1:8081/base"))) {
48          assertEquals("http://127.0.0.1:8081/base", factory.getUri().toString().trim());
49          assertThrows(Exception.class, () -> client.postInputStream("name", new FakeInputStream(10), 10));
50          assertThrows(Exception.class, () -> client.getInputStreamBusinessOut("name", 0));
51        }
52        try (final var client = factory.newClient(URI.create("http://127.0.0.1:8081/"))) {
53          assertThrows(Exception.class,
54              () -> client.postInputStream("INTERRUPT", new FakeInputStream(1000 * StandardProperties.getBufSize()), 0));
55          assertThrows(Exception.class,
56              () -> client.getInputStreamBusinessOut("INTERRUPT", 1000 * StandardProperties.getBufSize()).inputStream()
57                  .transferTo(new VoidOutputStream()));
58        }
59        factory.prepare(false, "localhost", 8081, "/test");
60        assertEquals(URI.create("http://localhost:8081/test"), factory.getUri());
61      } finally {
62        factory.prepare(uri);
63      }
64    }
65  
66    @Test
67    void testSimpleClientAbstract() {
68      final var uri = simpleFactory.getUri();
69      try {
70        try (final var client = simpleFactory.newClient(URI.create("http://127.0.0.1:8081/base"))) {
71          assertEquals(URI.create("http://127.0.0.1:8081/base"), client.getUri());
72        }
73        try (final var client = simpleFactory.newClient(URI.create("http://127.0.0.1:8081/"))) {
74          assertEquals(URI.create("http://127.0.0.1:8081/"), client.getUri());
75        }
76        simpleFactory.prepare(false, "127.0.0.1", 8081, "path");
77        assertEquals(URI.create("http://127.0.0.1:8081/path"), simpleFactory.getUri());
78        simpleFactory.prepare(URI.create("base"));
79        assertEquals(URI.create("http://127.0.0.1:8081/base"), simpleFactory.getUri());
80        simpleFactory.prepare(false, "localhost", 8081, "");
81        assertEquals(URI.create("http://localhost:8081/"), simpleFactory.getUri());
82      } finally {
83        simpleFactory.prepare(uri);
84      }
85    }
86  }
