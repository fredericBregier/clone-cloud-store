17  package io.clonecloudstore.accessor.model;
18  
19  import java.time.Instant;
20  
21  import io.clonecloudstore.common.standard.exception.CcsInvalidArgumentRuntimeException;
22  import io.quarkus.test.junit.QuarkusTest;
23  import org.junit.jupiter.api.Assertions;
24  import org.junit.jupiter.api.Test;
25  
26  import static org.junit.jupiter.api.Assertions.assertEquals;
27  import static org.junit.jupiter.api.Assertions.assertFalse;
28  import static org.junit.jupiter.api.Assertions.assertNotEquals;
29  import static org.junit.jupiter.api.Assertions.assertTrue;
30  
31  @QuarkusTest
32  public class AccessorBucketTest {
33  
34    public static final String WRONG_CHAR = "wrong\b";
35    public static final String WRONG_CDATA = "wrong<![CDATA[]]/>";
36  
37    @Test
38    void converter() {
39      final var accessorBucket = new AccessorBucket();
40      accessorBucket.setName("bucket").setStatus(AccessorStatus.READY).setCreation(Instant.now())
41          .setExpires(Instant.now().plusSeconds(60)).setSite("site");
42      accessorBucket.setId("id-" + accessorBucket.getName());
43      final var accessorBucket1 = accessorBucket.cloneInstance();
44      assertEquals(accessorBucket, accessorBucket1);
45      assertEquals(accessorBucket, accessorBucket);
46      Assertions.assertTrue(accessorBucket.toString().contains("site"));
47      accessorBucket1.setSite("newsite");
48      Assertions.assertNotEquals(accessorBucket, accessorBucket1);
49    }
50  
51    @Test
52    void statusTest() {
53      assertEquals(AccessorStatus.ERR_DEL.name(), AccessorStatus.toString(AccessorStatus.ERR_DEL));
54      assertEquals(AccessorStatus.UNKNOWN.name(), AccessorStatus.toString(AccessorStatus.UNKNOWN));
55      assertEquals(AccessorStatus.UPLOAD, AccessorStatus.fromStatusCode(AccessorStatus.UPLOAD.getStatus()));
56      assertEquals(AccessorStatus.READY, AccessorStatus.fromStatusCode(AccessorStatus.READY.getStatus()));
57      assertEquals(AccessorStatus.DELETING, AccessorStatus.fromStatusCode(AccessorStatus.DELETING.getStatus()));
58      assertEquals(AccessorStatus.DELETED, AccessorStatus.fromStatusCode(AccessorStatus.DELETED.getStatus()));
59      assertEquals(AccessorStatus.ERR_UPL, AccessorStatus.fromStatusCode(AccessorStatus.ERR_UPL.getStatus()));
60      assertEquals(AccessorStatus.ERR_DEL, AccessorStatus.fromStatusCode(AccessorStatus.ERR_DEL.getStatus()));
61      assertEquals(AccessorStatus.UNKNOWN, AccessorStatus.fromStatusCode(AccessorStatus.UNKNOWN.getStatus()));
62      assertEquals(AccessorStatus.UNKNOWN, AccessorStatus.fromStatusCode((short) -1));
63    }
64  
65    @Test
66    void wrongValues() {
67      final var accessorBucket = new AccessorBucket();
68      accessorBucket.setSite("correctSite");
69      try {
70        accessorBucket.setName(WRONG_CHAR);
71        Assertions.fail("Should failed");
72      } catch (final CcsInvalidArgumentRuntimeException ignored) {
73      }
74      try {
75        accessorBucket.setName(WRONG_CDATA);
76        Assertions.fail("Should failed");
77      } catch (final CcsInvalidArgumentRuntimeException ignored) {
78      }
79      try {
80        accessorBucket.setName("BadUpperCase");
81        Assertions.fail("Should failed");
82      } catch (final CcsInvalidArgumentRuntimeException ignored) {
83      }
84      try {
85        accessorBucket.setName("toolong".repeat(10));
86        Assertions.fail("Should failed");
87      } catch (final CcsInvalidArgumentRuntimeException ignored) {
88      }
89      accessorBucket.setName("correct-bucket-0");
90    }
91  
92    @Test
93    void checkEquals() {
94      final var accessorBucket = new AccessorBucket();
95      final var accessorBucket2 = new AccessorBucket();
96      final var accessorObject = new AccessorObject();
97      assertFalse(accessorBucket.equals(accessorObject));
98      assertNotEquals(accessorBucket.hashCode(), accessorObject.hashCode());
99      assertTrue(accessorBucket.equals(accessorBucket));
100     assertTrue(accessorBucket.equals(accessorBucket2));
101     assertEquals(accessorBucket.hashCode(), accessorBucket2.hashCode());
102     accessorBucket.setId("idbucket");
103     assertFalse(accessorBucket.equals(accessorBucket2));
104     assertNotEquals(accessorBucket.hashCode(), accessorBucket2.hashCode());
105     accessorBucket2.setId("idbucket");
106     assertTrue(accessorBucket.equals(accessorBucket2));
107     assertEquals(accessorBucket.hashCode(), accessorBucket2.hashCode());
108     accessorBucket.setSite("id");
109     assertFalse(accessorBucket.equals(accessorBucket2));
110     assertNotEquals(accessorBucket.hashCode(), accessorBucket2.hashCode());
111     accessorBucket2.setSite("id");
112     assertTrue(accessorBucket.equals(accessorBucket2));
113     assertEquals(accessorBucket.hashCode(), accessorBucket2.hashCode());
114     accessorBucket.setName("idbucket");
115     assertFalse(accessorBucket.equals(accessorBucket2));
116     assertNotEquals(accessorBucket.hashCode(), accessorBucket2.hashCode());
117     accessorBucket2.setName("idbucket");
118     assertTrue(accessorBucket.equals(accessorBucket2));
119     assertEquals(accessorBucket.hashCode(), accessorBucket2.hashCode());
120     accessorBucket.setStatus(AccessorStatus.READY);
121     assertFalse(accessorBucket.equals(accessorBucket2));
122     assertNotEquals(accessorBucket.hashCode(), accessorBucket2.hashCode());
123     accessorBucket2.setStatus(AccessorStatus.READY);
124     assertTrue(accessorBucket.equals(accessorBucket2));
125     assertEquals(accessorBucket.hashCode(), accessorBucket2.hashCode());
126     accessorBucket.setCreation(Instant.now());
127     assertFalse(accessorBucket.equals(accessorBucket2));
128     assertNotEquals(accessorBucket.hashCode(), accessorBucket2.hashCode());
129     accessorBucket2.setCreation(accessorBucket.getCreation());
130     assertTrue(accessorBucket.equals(accessorBucket2));
131     assertEquals(accessorBucket.hashCode(), accessorBucket2.hashCode());
132     accessorBucket.setExpires(Instant.now());
133     assertFalse(accessorBucket.equals(accessorBucket2));
134     assertNotEquals(accessorBucket.hashCode(), accessorBucket2.hashCode());
135     accessorBucket2.setExpires(accessorBucket.getExpires());
136     assertTrue(accessorBucket.equals(accessorBucket2));
137     assertEquals(accessorBucket.hashCode(), accessorBucket2.hashCode());
138   }
