26  import io.clonecloudstore.common.standard.exception.CcsWithStatusException;
27  import io.clonecloudstore.driver.s3.DriverS3Properties;
28  import io.clonecloudstore.test.resource.MinioKafkaProfile;
29  import io.clonecloudstore.test.resource.s3.MinIoResource;
30  import io.clonecloudstore.test.stream.FakeInputStream;
31  import io.quarkus.test.junit.QuarkusTest;
32  import io.quarkus.test.junit.TestProfile;
33  import jakarta.inject.Inject;
34  import org.jboss.logging.Logger;
35  import org.junit.jupiter.api.BeforeAll;
36  import org.junit.jupiter.api.Test;
37  
38  import static org.junit.jupiter.api.Assertions.assertEquals;
39  import static org.junit.jupiter.api.Assertions.assertThrows;
40  
41  @QuarkusTest
42  @TestProfile(MinioKafkaProfile.class)
43  class AccessorObjectResourceNoMongoTest {
44    private static final Logger LOG = Logger.getLogger(AccessorObjectResourceNoMongoTest.class);
45    public static final String BUCKET_NAME = "testbucket";
46    public static final String DIR_NAME = "dir/";
47    public static final String OBJECT = DIR_NAME + "testObject";
48  
49    @Inject
50    AccessorBucketApiFactory factoryBucket;
51    @Inject
52    AccessorObjectApiFactory factory;
53    private static String clientId = null;
54  
55    @BeforeAll
56    static void setup() {
57      clientId = UUID.randomUUID().toString();
58  
59      // Bug fix on "localhost"
60      var url = MinIoResource.getUrlString();
61      if (url.contains("localhost")) {
62        url = url.replace("localhost", "127.0.0.1");
63      }
64      DriverS3Properties.setDynamicS3Parameters(url, MinIoResource.getAccessKey(), MinIoResource.getSecretKey(),
65          MinIoResource.getRegion());
66    }
67  
68    @Test
69    void createBucketAndObject() throws CcsWithStatusException {
70      final var finalBucketName = DaoAccessorBucketRepository.getBucketTechnicalName(clientId, BUCKET_NAME);
71      createBucketAndObject(BUCKET_NAME);
72    }
73  
74    void createBucketAndObject(final String bucketName) {
75      try (final var client = factoryBucket.newClient()) {
76        assertEquals(500,
77            assertThrows(CcsWithStatusException.class, () -> client.createBucket(bucketName, clientId)).getStatus());
78        LOG.info("CreateBucket");
79      }
80      try (final var client = factory.newClient()) {
81        // 404 since Bucket does not exists
82        final var accessorObject = new AccessorObject().setBucket(bucketName).setName(OBJECT).setSize(100);
83        assertEquals(500, assertThrows(CcsWithStatusException.class,
84            () -> client.createObject(accessorObject, clientId, new FakeInputStream(100))).getStatus());
85        LOG.info("CreateObject");
86      }
87      // Without creating object
88      try (final var client = factory.newClient()) {
89        assertEquals(500, assertThrows(CcsWithStatusException.class,
90            () -> client.checkObjectOrDirectory(bucketName, OBJECT, clientId)).getStatus());
91        LOG.info("CheckObject");
92      }
93      try (final var client = factory.newClient()) {
94        assertEquals(500, assertThrows(CcsWithStatusException.class,
95            () -> client.checkObjectOrDirectory(bucketName, DIR_NAME, clientId)).getStatus());
96      }
97      try (final var client = factory.newClient()) {
98        assertEquals(500, assertThrows(CcsWithStatusException.class,
99            () -> client.checkObjectOrDirectory(bucketName, OBJECT, clientId)).getStatus());
100     }
101     try (final var client = factory.newClient()) {
102       assertEquals(500, assertThrows(CcsWithStatusException.class,
103           () -> client.checkObjectOrDirectory(bucketName, DIR_NAME, clientId)).getStatus());
104     }
105     try (final var client = factory.newClient()) {
106       assertEquals(500, assertThrows(CcsWithStatusException.class,
107           () -> client.getObjectInfo(bucketName, OBJECT, clientId)).getStatus());
108     }
109     try (final var client = factory.newClient()) {
110       assertEquals(500,
111           assertThrows(CcsWithStatusException.class, () -> client.getObject(bucketName, OBJECT, clientId)).getStatus());
112     }
113     try (final var client = factory.newClient()) {
114       assertEquals(500, assertThrows(CcsWithStatusException.class,
115           () -> client.listObjects(bucketName, clientId, new AccessorFilter().setNamePrefix(DIR_NAME))).getStatus());
116     }
117     try (final var client = factory.newClient()) {
118       assertEquals(500, assertThrows(CcsWithStatusException.class,
119           () -> client.deleteObject(bucketName, OBJECT, clientId)).getStatus());
120     }
121     try (final var client = factoryBucket.newClient()) {
122       assertEquals(500,
123           assertThrows(CcsWithStatusException.class, () -> client.deleteBucket(bucketName, clientId)).getStatus());
124     }
125   }
126 }
