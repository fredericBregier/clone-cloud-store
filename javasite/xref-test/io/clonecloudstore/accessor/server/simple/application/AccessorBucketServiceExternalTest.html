48  @QuarkusTest
49  @TestProfile(MinioProfile.class)
50  class AccessorBucketServiceExternalTest {
51    @Inject
52    AccessorBucketService service;
53    @Inject
54    DriverApiFactory storageDriverFactory;
55    String clientId = null;
56  
57    @BeforeAll
58    static void setup() {
59      // Setup Minio S3 Driver
60      // Bug fix on "localhost"
61      var url = MinIoResource.getUrlString();
62      if (url.contains("localhost")) {
63        url = url.replace("localhost", "127.0.0.1");
64      }
65      DriverS3Properties.setDynamicS3Parameters(url, MinIoResource.getAccessKey(), MinIoResource.getSecretKey(),
66          MinIoResource.getRegion());
67    }
68  
69    @BeforeEach
70    public void cleanBeforeTest() {
71      //Generate fake client id
72      clientId = UUID.randomUUID().toString();
73    }
74  
75    @Test
76    void createBucket() {
77      try {
78        final var bucketName = "bucketcreatetest";
79        final var bucketTechnicalName = AccessorBucketResource.getRealBucketName(clientId, bucketName);
80  
81        //Create bucket
82        final var bucket = service.createBucket(clientId, bucketTechnicalName);
83        Assertions.assertEquals(bucketName, bucket.getName());
84        Assertions.assertEquals(bucketTechnicalName, bucket.getId());
85        Assertions.assertEquals(AccessorStatus.READY, bucket.getStatus());
86  
87        //Check can't create 2 bucket for one client.
88        assertThrows(CcsAlreadyExistException.class, () -> service.createBucket(clientId, bucketTechnicalName));
89  
90        //Check can't create bucket with Maj
91        final var bucketNameMaj = "bucketCreateTest";
92        assertThrows(CcsOperationException.class, () -> service.createBucket(clientId, bucketNameMaj));
93  
94        //Check can't create bucket with special char
95        final var bucketNameSpecialChar = "bucket*";
96        assertThrows(CcsOperationException.class, () -> service.createBucket(clientId, bucketNameSpecialChar));
97  
98  
99        //Check min limit (min allowed 3)
100       final var bucketNameMin = "b";
101       assertThrows(CcsOperationException.class, () -> service.createBucket(clientId, bucketNameMin));
102 
103       //Check max limit (max allowed 31)
104       final var bucketNameMax = "uguoktjpshccqaqgyeiekwocfpupbhblvgkykuaosnjhrsylpqawndmjxw";
105       final var technicalId = AccessorBucketResource.getRealBucketName(clientId, bucketNameMax);
106       final Exception exceptionMax =
107           assertThrows(CcsOperationException.class, () -> service.createBucket(clientId, technicalId));
108       assertNotNull(exceptionMax);
109 
110     } catch (final Exception e) {
111       throw new RuntimeException(e);
112     }
113   }
114 
115   @Test
116   void getBucket() {
117     final var unknownBucketTechnicalName = AccessorBucketResource.getRealBucketName(clientId, "unknownbucket");
118     //Get Bucket not exist
119     final Exception exceptionMin =
120         assertThrows(CcsNotExistException.class, () -> service.getBucket(unknownBucketTechnicalName, clientId));
121     assertNotNull(exceptionMin);
122     assertFalse(service.checkBucket(unknownBucketTechnicalName));
123     assertFalse(AccessorProperties.isRemoteRead());
124     assertFalse(AccessorProperties.isFixOnAbsent());
125 
126     //Create a bucket and get Bucket
127     final var bucketName = "testgetbucket";
128     final var bucketTechnicalName = AccessorBucketResource.getRealBucketName(clientId, bucketName);
129     final AccessorBucket bucket;
130     service.createBucket(clientId, bucketTechnicalName);
131     bucket = service.getBucket(bucketTechnicalName, clientId);
132     assertEquals(bucketName, bucket.getName());
133     assertEquals(AccessorStatus.READY, bucket.getStatus());
134     assertTrue(service.checkBucket(bucketTechnicalName));
135 
136     //Check read bucket deleted
137     service.deleteBucket(clientId, bucketTechnicalName);
138 
139     assertThrows(CcsNotExistException.class, () -> service.getBucket(bucketTechnicalName, clientId));
140     assertFalse(service.checkBucket(bucketTechnicalName));
141   }
142 
143   @Test
144   void getBuckets() throws CcsOperationException {
145     //Create buckets
146     final var bucketName = "testgetbuckets";
147 
148     final var prefixBucketTechnicalName = AccessorBucketResource.getRealBucketName(clientId, bucketName);
149     service.createBucket(clientId, prefixBucketTechnicalName + "1");
150     service.createBucket(clientId, prefixBucketTechnicalName + "2");
151     service.createBucket(clientId, prefixBucketTechnicalName + "3");
152 
153     final var buckets = service.getBuckets(clientId);
154     assertEquals(3, buckets.size());
155     for (final var bucket : buckets) {
156       Assertions.assertEquals(AccessorStatus.READY, bucket.getStatus());
157     }
158   }
159 
160   @Test
161   void deleteBucket() {
162     final var bucketName = "bucketdeletedtest";
163     final AccessorBucket bucket;
164     {
165       final var bucketTechnicalName = AccessorBucketResource.getRealBucketName(clientId, bucketName);
166       bucket = service.createBucket(clientId, bucketTechnicalName);
167       assertEquals(bucketName, bucket.getName());
168       assertEquals(bucketTechnicalName, bucket.getId());
169       assertEquals(AccessorStatus.READY, bucket.getStatus());
170     }
171     //Delete bucket already
172     {
173       final var bucketTechnicalName = AccessorBucketResource.getRealBucketName(clientId, bucketName);
174       final var bucketDeleted = service.deleteBucket(clientId, bucketTechnicalName);
175       Assertions.assertEquals(AccessorStatus.DELETED, bucketDeleted.getStatus());
176       final Exception exceptionDeleted =
177           assertThrows(CcsDeletedException.class, () -> service.deleteBucket(clientId, bucketTechnicalName));
178       assertNotNull(exceptionDeleted);
179       //Delete bucket already deleted
180       try (final var storageDriver = storageDriverFactory.getInstance()) {
181         final Exception exceptionDeletedStorage =
182             assertThrows(DriverNotFoundException.class, () -> storageDriver.bucketDelete(bucketDeleted.getId()));
183         assertNotNull(exceptionDeletedStorage);
184       }
185       //Delete unknown Bucket
186       final var unknownBucketTechnicalName = AccessorBucketResource.getRealBucketName(clientId, "unknownBucket");
187       final Exception exceptionDeleted2 =
188           assertThrows(CcsDeletedException.class, () -> service.deleteBucket(clientId, unknownBucketTechnicalName));
189       assertNotNull(exceptionDeleted2);
190     }
191   }
