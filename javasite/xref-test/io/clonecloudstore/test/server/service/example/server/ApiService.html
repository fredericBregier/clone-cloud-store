30  import io.smallrye.mutiny.Uni;

45  import org.jboss.resteasy.reactive.RestPath;
46  
47  import static io.clonecloudstore.test.server.service.example.client.ApiConstants.X_CREATION_DATE;
48  import static io.clonecloudstore.test.server.service.example.client.ApiConstants.X_LEN;
49  import static io.clonecloudstore.test.server.service.example.client.ApiConstants.X_NAME;
50  
51  @Path(ApiConstants.API_ROOT)
52  @NoCache
53  public class ApiService extends StreamServiceAbstract<ApiBusinessIn, ApiBusinessOut, ServerNativeStreamHandler> {
54    public static final long LEN = 50 * 1024 * 1024;
55    private static final Logger LOG = Logger.getLogger(ApiService.class);
56  
57    // Example of REST API out of any InputStream usage
58    @Path(ApiConstants.API_COLLECTIONS + "/{name}")
59    @HEAD
60    public Uni<Response> checkName(final String name) {
61      return Uni.createFrom().emitter(em -> {
62        // Business code should come here
63        try {
64          em.complete(Response.noContent().header(X_NAME, name).header(X_LEN, Long.toString(LEN))
65              .header(X_CREATION_DATE, Instant.now().toString()).build());
66        } catch (final CcsClientGenericException | CcsServerGenericException e) {
67          em.complete(createErrorResponse(e));
68        } catch (final Exception e) {
69          LOG.error(e.getMessage(), e);
70          em.complete(createErrorResponse(e));
71        }
72      });
73    }
74  
75    // REST API for receiving InputStream from client
76    @Path(ApiConstants.API_COLLECTIONS)
77    @POST
78    @Consumes(MediaType.APPLICATION_OCTET_STREAM)
79    @Produces(MediaType.APPLICATION_JSON)
80    @Blocking
81    public Uni<Response> createObject(final HttpServerRequest request, @Context final Closer closer,
82                                      final InputStream inputStream,
83                                      @DefaultValue("name") @RestHeader(X_NAME) final String name,
84                                      @DefaultValue("0") @RestHeader(X_LEN) final long len) {
85      // Business code should come here
86      final var businessIn = new ApiBusinessIn();
87      businessIn.name = name;
88      businessIn.len = len;
89      return createObject(request, closer, businessIn, businessIn.len, null, false, inputStream);
90    }
91  
92    // REST API for sending InputStream back to client
93    @Path(ApiConstants.API_COLLECTIONS + "/{business}")
94    @GET
95    @Produces(MediaType.APPLICATION_OCTET_STREAM)
96    @Blocking
97    public Uni<Response> readObject(@RestPath final String business, @DefaultValue("0") @RestHeader(X_LEN) final long len,
98                                    final HttpServerRequest request, @Context final Closer closer) {
99      // Business code should come here
100     final var businessIn = new ApiBusinessIn();
101     businessIn.name = business;
102     // Fake LEN
103     businessIn.len = len > 0 ? len : LEN;
104     return readObject(request, closer, businessIn, false);
105   }
106 
107   // Example of REST API out of any InputStream usage but same URI (different accept through Produces annotation)
108   @Path(ApiConstants.API_COLLECTIONS + "/{business}")
109   @GET
110   @Produces(MediaType.APPLICATION_JSON)
111   public Uni<Response> getObjectMetadata(@RestPath final String business) {
112     return Uni.createFrom().emitter(em -> {
113       // Business code should come here
114       try {
115         final var businessOut = new ApiBusinessOut();
116         businessOut.name = business;
117         businessOut.len = LEN;
118         businessOut.creationDate = Instant.now();
119         em.complete(Response.ok(businessOut).build());
120       } catch (final CcsClientGenericException | CcsServerGenericException e) {
121         em.complete(createErrorResponse(e));
122       } catch (final Exception e) {
123         LOG.error(e.getMessage());
124         em.complete(createErrorResponse(e));
125       }
126     });
127   }
128 }
