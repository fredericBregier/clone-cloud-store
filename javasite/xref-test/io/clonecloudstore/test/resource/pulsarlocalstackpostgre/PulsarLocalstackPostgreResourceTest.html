64    @Test
65    @Transactional
66    void checkDb() {
67      assertNotNull(PostgresResource.getJdbcUrl());
68      System.out.println(PostgresResource.getJdbcUrl());
69      assertEquals(0, repository.count());
70      final var personEntity = new PgPersonEntity();
71      personEntity.setName("name");
72      repository.persistAndFlush(personEntity);
73      assertEquals(1, repository.count());
74      final var personEntity1 = repository.findByName("name");
75      assertEquals(personEntity.getName(), personEntity1.getName());
76      assertEquals(personEntity.getId(), personEntity1.getId());
77      repository.delete(personEntity);
78      assertEquals(0, repository.count());
79    }

98    @Test
99    void s3Check() throws URISyntaxException {
100     new EmptyClass();
101     final var bucket = "namebucket";
102     var url = LocalStackResource.getUrlString();
103     if (url.contains("localhost")) {
104       url = url.replace("localhost", "127.0.0.1");
105     }
106     try (final var s3Client = S3Client.builder().endpointOverride(new URI(url)).credentialsProvider(
107             StaticCredentialsProvider.create(
108                 AwsBasicCredentials.create(LocalStackResource.getAccessKey(), LocalStackResource.getSecretKey())))
109         .region(Region.of(LocalStackResource.getRegion())).build()) {
110       final var bucketRequestExist = HeadBucketRequest.builder().bucket(bucket).build();
111       try {
112         s3Client.headBucket(bucketRequestExist).sdkHttpResponse().isSuccessful();
113         fail("Should raised an exception");
114       } catch (final NoSuchBucketException ignored) {
115         // Ignore
116       }
117       final var bucketRequest = CreateBucketRequest.builder().bucket(bucket).build();
118       final var response = s3Client.createBucket(bucketRequest);
119       assertTrue(response.sdkHttpResponse().isSuccessful());
120       final var bucketRequestWait = HeadBucketRequest.builder().bucket(bucket).build();
121       // Wait until the bucket is created
122       final var s3Waiter = s3Client.waiter();
123       final var waiterResponse = s3Waiter.waitUntilBucketExists(bucketRequestWait);
124       assertFalse(waiterResponse.matched().response().isEmpty() ||
125           !waiterResponse.matched().response().get().sdkHttpResponse().isSuccessful());
126       try {
127         s3Client.headBucket(bucketRequestExist).sdkHttpResponse().isSuccessful();
128       } catch (final NoSuchBucketException e) {
129         fail(e);
130       }
131       final var responseDelete = s3Client.deleteBucket(DeleteBucketRequest.builder().bucket(bucket).build());
132       assertTrue(response.sdkHttpResponse().isSuccessful());
133     }
134   }
